1 /**
  2  * For use while testing classes that make AJAX requests to a remote server.
  3  *
  4  * For example, if you are testing a bit of code that includes the following:
  5  *
  6  * <pre>
  7  * <code class="language-javascript">
  8  * new bbq.ajax.JSONRequest({
  9  *     url: "/foo/bar",
 10  *     args: {
 11  *          bar: "baz"
 12  *     },
 13  *     onSuccess: function(serverResponse, json) {
 14  *          alert("server said: " + json.baz);
 15  *     }
 16  * });
 17  * </code>
 18  * </pre>
 19  *
 20  * To handle this sort of request, you'd do the following before the AJAX
 21  * call is invoked:
 22  *
 23  * <pre>
 24  * <code class="language-javascript">
 25  * bbq.ajax.MockAJAXRequest["/foo/bar"] = function(args) {
 26  *      return new bbq.ajax.MockJSONResponse({response: {
 27  *          baz: "qux"
 28  *      }});
 29  * }
 30  * </code>
 31  * </pre>
 32  *
 33  * This will result in the onSuccess handler being called immediately.
 34  *
 35  * @class bbq.ajax.MockAJAXRequest
 36  */
 37 bbq.ajax.MockAJAXRequest = {
 38 
 39 };

 41 // Overwrite method that sends the AJAX request
 42 bbq.ajax.AJAXRequest.prototype._sendRequest = function() {
 43 	var result;
 44 
 45 	try {
 46 		if(bbq.ajax.MockAJAXRequest[this.options.url]) {
 47 			var handler = bbq.ajax.MockAJAXRequest[this.options.url];
 48 
 49 			result = handler(this.options.args);
 50 
 51 			this._onSuccess(result);
 52 		}
 53 	} catch(e) {
 54 
 55 	}
 56 
 57 	this._onFailure(result);
 58 };
