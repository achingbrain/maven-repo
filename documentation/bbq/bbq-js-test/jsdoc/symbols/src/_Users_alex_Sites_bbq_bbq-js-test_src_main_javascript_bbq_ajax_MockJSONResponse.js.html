1 /**
  2  * To be used in concert with bbq.ajax.MockAJAXRequest.  Instances of this
  3  * class should be returned by anonymous functions added to the
  4  * bbq.ajax.MockAJAXRequest object.
  5  *
  6  * @class bbq.ajax.MockJSONResponse
  7  * @param options
  8  *      headers: Object
  9  *      response: Object
 10  * @see bbq.ajax.MockAJAXRequest
 11  */
 12 bbq.ajax.MockJSONResponse = Class.create({
 13 	_headers: null,
 14 
 15 	initialize: function(options) {
 16 		this._headers = {
 17 			"Content-Type": "application/json",
 18 			"X-BBQ-ResponseType": 100
 19 		};
 20 
 21 		if(options.headers) {
 22 			for(var key in options.headers) {
 23 				if(options.headers[key]) {
 24 					this._headers[key] = options.headers[key];
 25 				}
 26 			}
 27 		}
 28 
 29 		this.responseText = options.response.toJSON();
 30 	},
 31 
 32 	getHeader: function(key) {
 33 		return this._headers[key];
 34 	},
 35 
 36 	getResponseHeader: function(key) {
 37 		return this.getHeader(key);
 38 	}
 39 });
