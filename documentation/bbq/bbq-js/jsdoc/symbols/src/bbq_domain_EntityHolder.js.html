 24 	},

 31 	},
 32

 34 	 * @param {Number} index
 35 	 */

 38 	},

 41 	 * @param * object An object or a GUID
 42 	 * @return {Number}

 46

 48 			if(entity.equals(object) || entity.getId() == object) {
 49 				index = i;
 50 
 51 				throw $break;
 52 			}
 53 		}.bind(this));
 54 
 55 		return index;
 56 	},
 57

 59 	 * @return {Array} array of elements

 65 	/**
 66 	 * @param {Object} element
 67 	 */
 68 	add: function(element) {
 69 		if(element instanceof this.options.type) {
 70 			// passed entity
 71 			if(this.indexOf(element) != -1) {
 72 				return element;
 73 			}
 74 		} else if(Object.isString(element) || Object.isNumber(element)) {
 75 			// passed identifier
 76 			element = new this.options.type({data: {id: element}});
 77 		} else {
 78 			// passed object data
 79 			element = new this.options.type({data: element});
 80 		}
 81 
 82 		return this.put(element, this._entities.length);
 83 	},
 84 
 85 	/**
 86 	 * @param {Object} element
 87 	 * @param {Number} index
 88 	 */

 91 			element = new this.options.type({data: element});
 92 		}
 93 
 94 		//Log.info(index);
 95 		this._entities[index] = element;
 96 
 97 		return element;
 98 	},

111 				//delete this._arrayIndexes["index-" + entityKey];

113 			}
114 		}
