  3 /**
  4  * This class is designed to be used with the org.bbqjs.spring.ajax.RequestForwarder class
  5  * and attepts to work around the same origin policy of most browsers in terms of AJAX.
  6  *
  7  * The policy means that you can only dispatch an AJAX request to the server from which
  8  * the JavaScript dispatching the request was loaded.  So if you deliver your code from
  9  * http://www.example.org, you cannot make a request to http://api.google.com.
 10  *
 11  * The RequestForwarder class will forward on the request to the location specified in the
 12  * url property of the options argument, even if it's a different domain.
 13  *
 14  * Other than that it behaves in the same way as bbq.ajax.SoapRequest
 15  *
 16  * <pre>
 17  * <code class="language-javascript">
 18  *
 19  * var request = new bbq.ajax.ForwardingSoapRequest({
 20  *     url: "http://api.google.com",
 21  *     onSuccess: function(serverRequest, json) {
 22  *
 23  *     }
 24  *     ... // more args similar to bbq.ajax.SoapRequest
 25  * });
 26  *
 27  * </code>
 28  * </pre>
 29  *
 30  * @class bbq.ajax.ForwardingJSONRequest
 31  * @extends bbq.ajax.SoapRequest
 32  */
 33 bbq.ajax.ForwardingSoapRequest = Class.create(bbq.ajax.SoapRequest, {
 34 	initialize: function($super, options) {
 35 		options.forwardTo = options.url;
 36 		options.url = "/forward";
 37 		
 38 		$super(options);
 39 	},
 40 	
 41 	_createRequestHeaders: function($super) {
 42 		var headers = $super();
 43 		headers["X-BBQ-Forward-To"] = this.options.forwardTo;
 44 		
 45 		return headers;
 46 	},
 47 	
 48 	onSuccess: function(serverResponse) {
 49 		if(serverResponse.getResponseHeader("X-responseType") == -100) {
 50 			this._callHandler("onError", $A(arguments));
 51 		} else {
 52 			this._callHandler("onSuccess", $A(arguments));
 53 		}
 54 	}
 55 });
