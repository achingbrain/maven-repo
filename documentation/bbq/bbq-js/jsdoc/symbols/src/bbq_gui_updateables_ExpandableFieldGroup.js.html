1 include(bbq.gui.updateables.UpdateableFieldGroup);
  2 include(bbq.gui.button.AddButton);
  3 
  4 /**
  5  * @class bbq.gui.updateables.ExpandableFieldGroup
  6  * @extends bbq.gui.updateables.UpdateableFieldGroup
  7  */
  8 bbq.gui.updateables.ExpandableFieldGroup = new Class.create(bbq.gui.updateables.UpdateableFieldGroup, {
  9 	_tableBody: null,
 10 	_newItemButton: null,
 11 	_updateFunction: null,
 12 	_callbackKey: null,
 13 	
 14 	/**
 15 	 * @param {Object} options
 16 	 * @example
 17 	 * Todo an example here
 18 	 */
 19 	initialize: function($super, options) {
 20 		$super(options);
 21 		
 22 		this.setRootNode("table");
 23 		this.addClass("ExpandableFieldGroup");
 24 		this._newItemButton = new bbq.gui.button.AddButton({onclick: this.addItem.bindAsEventListener(this), attributes: {style: {"display": "none"}}});
 25 		this.appendChild(DOMUtil.createTextElement("thead", DOMUtil.createTextElement("tr", DOMUtil.createTextElement("th", this._newItemButton))));
 26 		this._tableBody  = this.appendChild(DOMUtil.createElement("tbody"));
 27 		
 28 		if(this.options.entities) {
 29 			// use pre-existing entities
 30 			this.options.entities.each(function(entity) {
 31 				this.addPreExistingItem(entity);
 32 			}.bind(this));
 33 		} else if(this.options.propertyDisplay) {
 34 			// use properties on passed entity
 35 			this.options.propertyDisplay.entity.getProperty(this.options.propertyDisplay.key).each(function(entity) {
 36 				this.addPreExistingItem(entity);
 37 			}.bind(this));
 38 			
 39 			// ensure that we will display current data when an object has been updated
 40 			this._callbackKey = this.options.propertyDisplay.entity.registerListener("onDataLoaded", this._updateDisplay.bind(this));
 41 		}
 42 	},
 43 	
 44 	_updateDisplay: function(options) {
 45 		if(!this.getEditMode()) {
 46 			this._fields = [];
 47 			this._fieldOptions = [];
 48 			
 49 			this.options.propertyDisplay.entity.getProperty(this.options.propertyDisplay.key).each(function(entity) {
 50 				this.addPreExistingItem(entity);
 51 			}.bind(this));
 52 			
 53 			this.render();
 54 		}
 55 	},
 56 	
 57 	deregisterChildListeners: function() {
 58 		if(this.options.propertyDisplay) {
 59 			Log.info("removing listener");
 60 			this.options.propertyDisplay.entity.deRegisterListener("onDataLoaded", this._callbackKey);
 61 		}
 62 	},
 63 	
 64 	render: function() {
 65 		DOMUtil.emptyNode(this._tableBody);
 66 		
 67 		this._fields.each(function(field) {
 68 			if(field._deleted) {
 69 				return;
 70 			}
 71 			
 72 			this._tableBody.appendChild(DOMUtil.createTextElement("tr", DOMUtil.createTextElement("td", field)));
 73 			field.setEditMode(this._editMode);
 74 		}.bind(this));
 75 	},
 76 	
 77 	/**
 78 	 * @param {integer} editMode
 79 	 */
 80 	setEditMode: function($super, editMode) {
 81 		$super(editMode);
 82 		
 83 		this._editMode = editMode;
 84 		this._newItemButton.setStyle("display", (editMode ? "block" : "none"));
 85 	},
 86 	
 87 	addItem: function() {
 88 		try {
 89 			var newItem = this.options.newItemMethod();
 90 			this.addField(newItem, {name: this.options.title + "-" + this._fields.length});
 91 			
 92 			var row = document.createElement("tr");
 93 			var cell = document.createElement("td");
 94 			newItem.appendTo(cell);
 95 			row.appendChild(cell);
 96 			this._tableBody.appendChild(row);
 97 			
 98 			newItem.setEditMode(this._editMode);
 99 		} catch(e) {
100 			Log.error("Error adding item", e);
101 		}
102 		
103 		return false;
104 	},
105 	
106 	/**
107 	 * @param {Object} item
108 	 */
109 	addPreExistingItem: function(item) {
110 		var newItem = this.options.newItemMethod.call(this, {initialValue: item});
111 		this.addField(newItem, {name: this.options.title + "-" + this._fields.length});
112 	},
113 	
114 	clearError: function() {
115 		this._fields.each(function(field) {
116 			field.clearError();
117 		})
118 	},
119 	
120 	getValue: function() {
121 		return this._fields.collect(function(field) {
122 			return field && field.deleted && field.getValue instanceof Function ? null : field.getValue();
123 		}).compact();
124 	},
125 	
126 	/**
127 	 * @param {boolean} disabled
128 	 */
129 	setDisabled: function($super, disabled) {
130 		$super(disabled);
131 		
132 		this._newItemButton.setDisabled(disabled);
133 	},
134 	
135 	cancelEdit: function($super) {
136 		// remove empty fields
137 		for(var i = 0; i < this._fields.length; i++) {
138 			if(this._fields[i].getValue() == null) {
139 				this._fields.splice(i, 1);
140 				i--;
141 			}
142 		}
143 		
144 		$super();
145 		
146 		this.render();
147 	},
148 	
149 	changed: function() {
150 		var changed = bbq.gui.updateables.UpdateableFieldGroup.prototype.changed.call(this);
151 		
152 		this._fields.each(function(field) {
153 			if(field._deleted) {
154 				changed = true;
155 			}
156 		});
157 		
158 		return  changed;
159 	}
160 });
