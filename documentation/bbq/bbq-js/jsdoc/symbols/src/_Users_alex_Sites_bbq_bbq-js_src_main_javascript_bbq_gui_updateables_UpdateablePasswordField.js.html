  3 /**
  4  * @class bbq.gui.updateables.UpdateablePasswordField
  5  * @extends bbq.gui.updateables.UpdateableTextField
  6  */
  7 bbq.gui.updateables.UpdateablePasswordField = new Class.create(bbq.gui.updateables.UpdateableTextField, {
  8 	
  9 	initialize: function($super, options) {
 10 		$super(options);
 11 		
 12 		this.addClass("UpdateablePasswordField");
 13 	},
 14 	
 15 	/**
 16 	 * @private
 17 	 */
 18 	_createEditField: function() {
 19 		this.inputField = bbq.gui.updateables.UpdateableTextField.prototype._createEditField.call(this);
 20 		this.inputField.type = "password";
 21 		this.inputField.value = "";
 22 		
 23 		Event.observe(this.inputField, "keypress", this._capslockWatcher.bindAsEventListener(this));
 24 		
 25 		return this.inputField;
 26 	},
 27 	
 28 	createViewField: function() {
 29 		if(this._getDisplayValue() && this._getDisplayValue().length > 0) {
 30 			return document.createTextNode("**********");
 31 		}
 32 		
 33 		return document.createTextNode(" ");
 34 	},
 35 	
 36 	/**
 37 	 * @param {boolean} supressErrorWarning
 38 	 * @return Returns the current field value 
 39 	 * @type {Object}
 40 	 */
 41 	getValue: function(supressErrorWarning) {
 42 		if(this.inputField && this.inputField.value == "" && this._getOriginalValue()) {
 43 			return -1;
 44 		}
 45 		
 46 		return bbq.gui.updateables.UpdateableTextField.prototype.getValue.apply(this, arguments);
 47 	},
 48 	
 49 	/**
 50 	 * @private
 51 	 */
 52 	_checkForCapsLock: function(shiftKey) {
 53 		try {
 54 			if(Browser.InternetExplorer) {
 55 				if(this._passwordFocused) {
 56 					var lastChar = this.inputField.value.charAt(this.inputField.value.length - 1);
 57 					
 58 					if(lastChar && ((lastChar.toLowerCase() == lastChar && shiftKey) || (lastChar.toUpperCase() == lastChar && !shiftKey))) {
 59 						DOMUtil.addClass(this.inputField, "UpdateablePasswordFieldCapsLockWarning");
 60 						this._capsLockOn = true;
 61 					} else {
 62 						DOMUtil.removeClass(this.inputField, "UpdateablePasswordFieldCapsLockWarning");
 63 						this._capsLockOn = false;
 64 					}
 65 				}
 66 			}
 67 		} catch(e) {
 68 			Log.error("Error checking for caps lock", e);
 69 		}
 70 	},
 71 	
 72 	_capslockWatcher: function(event) {
 73 		if(Browser.InternetExplorer) {
 74 			// IE for once is actually useful in reporting caps lock key presses
 75 			
 76 			if(event.keyCode == 20) { // caps lock
 77 				if(this._capsLockOn) {
 78 					this._capsLockOn = false;
 79 					DOMUtil.removeClass(this.inputField, "UpdateablePasswordFieldCapsLockWarning");
 80 				} else {
 81 					this._capsLockOn = true;
 82 					DOMUtil.addClass(this.inputField, "UpdateablePasswordFieldCapsLockWarning");
 83 				}
 84 			} else if(event.keyCode >= 65 && event.keyCode <= 90) {
 85 				setTimeout(this._checkForCapsLock.bind(this, event.shiftKey), 100);
 86 			}
 87 		} else if(Browser.Mozilla) {
 88 			// Mozilla/Firefox type browsers don't report caps lock keypresses so we have to be a little more sneaky
 89 			
 90 			//Log.info("char code = " + event.charCode + " key code = " + event.keyCode + " shift key = " + event.shiftKey);
 91 			
 92 			if(event.charCode == 0) {
 93 				// non-character code
 94 				return;
 95 			}
 96 			
 97 			this._capsLockOn = false;
 98 			var keyPressed = String.fromCharCode(event.charCode);
 99 			
100 			if(!isNaN(keyPressed/1)) {
101 				// a number - which can't be upper or lower case so we can't detect caps lock
102 				return;
103 			}
104 			
105 			//Log.info(" key pressed: " + keyPressed + " upper: " + keyPressed.toUpperCase());
106 			
107 			if(!event.shiftKey) {
108 				// if the key pressed is upper case then the caps lock key is down
109 				if(keyPressed == keyPressed.toUpperCase()) {
110 					this._capsLockOn = true;
111 				}
112 			}
113 			
114 			if(this._capsLockOn) {
115 				DOMUtil.addClass(this.inputField, "UpdateablePasswordFieldCapsLockWarning");
116 			} else {
117 				DOMUtil.removeClass(this.inputField, "UpdateablePasswordFieldCapsLockWarning");
118 			}
119 		}
120 	}
121 });
