  4 /**
  5  * @class Language
  6  */
  7 Language = {
  8 	options: {},
  9 	_language: {},
 10 	_languageLoaded: false,
 11 	
 12 	/**
 13 	 * @param {Object} options
 14 	 * @param {String} [options.section] This corresponds to the name of a language file defined in /WEB-INF/language
 15 	 * @param {Function} [options.postLoad] A callback function invoked once the language file has been loaded
 16 	 */
 17 	load: function(options) {
 18 		Language.options = options ? options : {};
 19 		
 20 		new bbq.ajax.JSONRequest({
 21 			url: "/language/get",
 22 			onSuccess: Language._gotLanguage.bind(Language), 
 23 			args: (options && options.section ? {section: options.section} : null)});
 24 	},
 25 	
 26 	/**
 27 	 * @private
 28 	 */
 29 	_gotLanguage: function(serverResponse, language) {
 30 		Language._language = language;
 31 		Language._languageLoaded = true;
 32 		
 33 		if(Language.options.postLoad instanceof Function) {
 34 			Language.options.postLoad.call(Language);
 35 		}
 36 	},
 37 
 38 	/**
 39 	 * Get a translated language string for a key
 40 	 *
 41 	 * @example
 42 	 * <pre><code class="language-javascript">
 43 	 * Language.get("foo.bar");
 44 	 * </code></pre>
 45 	 * 
 46 	 * @param key
 47 	 * @returns {String}
 48 	 */
 49 	get: function(key) {
 50 		if(Language._language && Language._language[key]) {
 51 			return Language._language[key];
 52 		}
 53 		
 54 		Log.error("Could not find language key " + key);
 55 		
 56 		return "unknown";
 57 	},
 58 
 59 	/**
 60 	 * Returns an array of language strings
 61 	 * 
 62 	 * @param	{String}	key
 63 	 * @return	{Array}
 64 	 */
 65 	getArray: function(string) {
 66 		if(!Language._language) {
 67 			return [];
 68 		}
 69 		
 70 		var output = [];
 71 		
 72 		for(var i = 0; true; i++) {
 73 			var key = string + i;
 74 			
 75 			if(Object.isUndefined(Language._language[key])) {
 76 				return output;
 77 			}
 78 			
 79 			output.push(Language._language[key]);
 80 		}
 81 	},
 82 	
 83 	/**
 84 	 * Allows for language formatting so that variable value placement in strings is not dependant on the flow of the English language.
 85 	 * 
 86 	 * In the language file, declare this sort of thing:
 87 	 *
 88 	 * <pre><code class="language-xml">
 89 	 * <?xml version="1.0" encoding="utf-8"?>
 90 	 * <!DOCTYPE properties SYSTEM "http://java.sun.com/dtd/properties.dtd">
 91 	 * <properties>
 92 	 *      <entry key="generic.mystring">This string has {marker} marker</entry>
 93 	 * </properties>
 94 	 * </code></pre>
 95 	 * 
 96 	 * Then in the javascript, call:
 97 	 *
 98 	 * <pre><code class="language-javascript">
 99 	 * Language.getFormatted("generic.mystring", {marker: "a nice little"});
100 	 * </code></pre>
101 	 * 
102 	 * And the following will be output:
103 	 *
104 	 * <pre><code>
105 	 * "This string has a nice little marker"
106 	 * </code></pre>
107 	 * 
108 	 * @param {String} language
109 	 * @param {Object} keys
110 	 */
111 	getFormatted: function(language, keys) {
112 		var outputDomNode = false;
113 
114 		for(var key in keys) {
115 			if(!Object.isString(keys[key]) && !Object.isNumber(keys[key])) {
116 				outputDomNode = true;
117 			}
118 		}
119 
120 		if(outputDomNode) {
121 			return Language._getFormattedNode(language, keys);
122 		} else {
123 			return Language._getFormattedString(language, keys);
124 		}
125 	},
126 
127 	_getFormattedString: function(language, keys) {
128 		var string = Language.get(language);
129 
130 		try {
131 			for (var key in keys) {
132 				string = string.replace(new RegExp("\{" + key + "\}", "gi"), keys[key]);
133 			}
134 		} catch(e) {
135 			Log.error("Error formatting language string", e);
136 		}
137 
138 		return string;
139 	},
140 
141 	_getFormattedNode: function(language, keys) {
142 		var output = document.createDocumentFragment();
143 		var string = Language.get(language);
144 
145 		try {
146 			for (var key in keys) {
147 				var index = string.indexOf("{" + key + "}");
148 
149 				output.appendChild(document.createTextNode(string.substring(0, index)));
150 
151 				if(Object.isString(keys[key])) {
152 					// String
153 					output.appendChild(document.createTextNode(keys[key]));
154 				} else if(keys[key] instanceof bbq.gui.GUIWidget) {
155 					// GUIWidget
156 					keys[key].appendTo(output);
157 				} else {
158 					// DOM node
159 					output.appendChild(keys[key]);
160 				}
161 
162 				string = string.substring(index + ("{" + key + "}").length);
163 			}
164 		} catch(e) {
165 			Log.error("Error formatting language string", e);
166 		}
167 
168 		return output;
169 	},
170 
171 	/**
172 	 * @returns {boolean} True if the language file has been loaded
173 	 */
174 	isLoaded: function() {
175 		return Language._languageLoaded ? true : false;
176 	}
