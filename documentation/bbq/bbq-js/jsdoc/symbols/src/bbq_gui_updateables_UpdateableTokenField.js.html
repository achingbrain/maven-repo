1 include(bbq.gui.updateables.UpdateableTextField);
  2 include(bbq.gui.token.TextToken);
  3 include(bbq.util.BBQUtil);
  4 
  5 /**
  6  * @class bbq.gui.updateables.UpdateableTokenField
  7  * @extends bbq.gui.updateables.UpdateableTextField 
  8  */
  9 bbq.gui.updateables.UpdateableTokenField = new Class.create(bbq.gui.updateables.UpdateableTextField, {
 10 	_noItemsText: null,
 11 	
 12 	/**
 13 	 * @param {Object} options
 14 	 * @example
 15 	 * Todo an example here
 16 	 * 
 17 	 */
 18 	initialize: function($super, options) {
 19 		$super(options);
 20 		
 21 		this.addClass("UpdateableTokenField");
 22 	},
 23 	
 24 	_setInitialValues: function() {
 25 		if(!Object.isUndefined(this.options.propertyDisplay)) {
 26 			this._setCurrentValue(this.options.propertyDisplay.entity.getProperty(this.options.propertyDisplay.key));
 27 		} else {
 28 			var value = [];
 29 			
 30 			if(this.options.initialValue instanceof Array) {
 31 				this.options.initialValue.each(function(keyword) {
 32 					value.push(keyword);
 33 				}.bind(this));
 34 			}
 35 			
 36 			this._setCurrentValue(value);
 37 		}
 38 		
 39 		this._setOriginalValue(BBQUtil.cloneObject(this._getCurrentValue()));
 40 	},
 41 	
 42 	/**
 43 	 * Called after the object the property display belongs to has loaded it's values
 44 	 */
 45 	_updateInitialValues: function() {
 46 		this._setCurrentValue(this.options.propertyDisplay.entity.getProperty(this.options.propertyDisplay.key));
 47 		this._setOriginalValue(this.options.propertyDisplay.entity.getProperty(this.options.propertyDisplay.key));
 48 	},
 49 	
 50 	_createViewFieldFromPropertyDisplay: function() {
 51 		return this.options.propertyDisplay.entity.getPropertyDisplay({property: this.options.propertyDisplay.key, formatter: function(words) {
 52 			return this.createViewField();
 53 		}.bind(this)});
 54 	},
 55 	
 56 	createEditField: function() {
 57 		this.inputField = bbq.gui.updateables.UpdateableTextField.prototype.createEditField.apply(this);
 58 		this.inputField.value = this._getCurrentValue().join(", ").strip();
 59 		
 60 		return this.inputField;
 61 	},
 62 	
 63 	createViewField: function() {
 64 		if(!this._getCurrentValue() || this._getCurrentValue().length == 0) {
 65 			return this._noItemsText;
 66 		}
 67 		
 68 		var documentFragment = document.createDocumentFragment();
 69 		
 70 		this._getCurrentValue().each(function(word) {
 71 			var textToken = new bbq.gui.token.TextToken({text: word});
 72 			textToken.appendTo(documentFragment);
 73 		});
 74 		
 75 		return documentFragment;
 76 	},
 77 	
 78 	_validateInput: function(input, currentValues) {
 79 		Log.error("UpdateableTokenField#_validateInput not overridden");
 80 	},
 81 	
 82 	updateLocalValue: function() {
 83 		if(this.options.inlineInstruction && this.inputField.value == this.options.inlineInstruction) {
 84 			// no value has been entered
 85 			this._setCurrentValue([]);
 86 		} else {
 87 			// create array from entered words
 88 			var words = []; 
 89 			
 90 			this.inputField.value.split(",").each(function(word) {
 91 				word = word.strip();
 92 				
 93 				// make sure we don't return duplicate words
 94 				if(this._validateInput(word, words)) {
 95 					words.push(word);
 96 				}
 97 			}.bind(this));
 98 			
 99 			this._setCurrentValue(words);
100 		}
101 		
102 		this.notifyListeners("onValueChanged");
103 		this.checkForRequired();
104 	},
105 	
106 	checkForRequired: function() {
107 		if(this.getEditMode()) {
108 			if(this.options.required && (!this._getCurrentValue() || this._getCurrentValue().length == 0)) {
109 				this.addClass("error");
110 				return false;
111 			} else {
112 				this.setError(false);
113 			}
114 		}
115 		return true;
116 	}
117 });
