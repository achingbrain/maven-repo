 30 			viewPortSize.width = window.innerWidth,
 31 			viewPortSize.height = window.innerHeight
 32 		} else if (typeof document.documentElement != 'undefined' && typeof document.documentElement.clientWidth != 'undefined' && document.documentElement.clientWidth != 0) {

101  * 
102  * @param	{Object}	object
103  * @return	{Object}
104  */
105 	cloneObject: function(object) {
106 		if(object) {
107 			var objectClone = new object.constructor();
108 			
109 			for(var property in object) {
110 				if(property != "controller" && property != "currentPage") {
111 					if(typeof object[property] == 'object') {
112 						objectClone[property] = BBQUtil.cloneObject(object[property]);
113 					} else {
114 						objectClone[property] = object[property];
115 					}
116 				}
117 			}
118 		
119 		return objectClone;
120 		}

204 	/**
205 	 * @param {string} guid
206 	 * @return {boolean}
207 	 */
208 	isGUID: function(guid) {
209 		if(!guid) {
210 			return false;
211 		}
212 		
213 		if(!guid.toLowerCase) {
214 			return false;
215 		}
216 		
217 		guid = guid.toLowerCase();
218 		
219 		if(guid.length != 36) {
220 			return false;
221 		}
222 		
223 		for(var i = 0; i < 36; i++) {
224 			if(i == 8 || i == 13 || i == 18 || i == 23) {
225 				if(guid.charAt(i) != '-') {
226 					return false;
227 				}
228 			} else if(!guid.charAt(i).match(/[0-9|a-f]/i)) {
229 				return false;
230 			}
231 		}
232 		
233 		return true;
234 	},
235 	/**
236 	 * @param {string} text
237 	 * @param {string} attributes
238 	 * @return {Node}
239 	 */
240 	formatText: function(text, attributes) {
241 		var output = DOMUtil.createElement("p", attributes);
242 				
243 		var lines = text.strip().split(/\n/);
244 		
245 		var linkRegex = /(\b(https?|ftp|file):\/\/[-A-Z0-9+&@#\/%?=~_|!:,.;]*[-A-Z0-9+&@#\/%=~_|])/ig;
246 		
247 		// go through line by line
248 		for(var i = 0; i < lines.length; i++) {
249 			var words = lines[i].strip().split(" ");
250 			
251 			for(var n = 0; n < words.length; n++) {
252 				// search for URLy text
253 				if(words[n].match(linkRegex)) {
254 					output.appendChild(DOMUtil.createTextElement("a", words[n], {href: words[n], target: "_new"}));
255 				} else {
256 					output.appendChild(document.createTextNode(words[n]));
257 				}
258 				
259 				// more words to go, append a space
260 				if(n < words.length - 1) {
261 					output.appendChild(document.createTextNode(" "));
262 				}
263 			}
264 			
265 			// more lines to go, append a line break
266 			if(i < lines.length - 1) {
267 				output.appendChild(document.createElement("br"));
268 			}
269 		}
270 		
271 		return output;
272 	},

316 	urlDecode: function(string) {
317 		var histogram = {};
318 		var replacer = function(search, replace, str) {
319 		var tmp_arr = [];
320 			tmp_arr = str.split(search);
321 			return tmp_arr.join(replace);
322 		};
323     	
324     	// The histogram is identical to the one in urlencode.
325 		histogram["'"]   = '%27';
326 		histogram['(']   = '%28';
327 		histogram[')']   = '%29';
328 		histogram['*']   = '%2A';
329 		histogram['~']   = '%7E';
330 		histogram['!']   = '%21';
331 		histogram['%20'] = '+';
332 		
333 		for (replace in histogram) {
334 			search = histogram[replace]; // Switch order when decoding
335 			string = replacer(search, replace, string) // Custom replace. No regexing   
336 		}
337 		
338 		// End with decodeURIComponent, which most resembles PHP's encoding functions
339 		string = decodeURIComponent(string);
340  		return string;
341 	},

374 	 * @return {String}

376 	formatFileSize: function(bytes) {
377 		if(bytes < Math.pow(2,10)) {

379 		}else if(bytes >= Math.pow(2,10) && bytes < Math.pow(2,20)) {
380 			return (Math.round(bytes/Math.pow(2,10))) + 'KB';
381 		}else if(bytes >= Math.pow(2,20) && bytes < Math.pow(2,30)) {
382 			return  (Math.round(bytes/Math.pow(2,20))) + 'MB';
383 		}else if(bytes >= Math.pow(2,30) && bytes < Math.pow(2,40)) {
384 			return  (Math.round(bytes/Math.pow(2,30))) + 'GB';
385 		}else if(bytes >= Math.pow(2,40)) {
386 			return  (Math.round(bytes/Math.pow(2,40))) + 'TB';
387 		}

390 	findClassName: function(parent, child, path) {
391 		for (var key in parent) {
392 			// ignore anything that will cause an infinite loop or security error
393 			if (parent[key] == parent || key == "history" || key == "globalStorage" || key == "argumentNames") {
394 				continue;
395 			}
396 
397 			var firstCharacter = key.substring(0, 1);
398 
399 			// first character is upper case, we have a class
400 			if (firstCharacter == firstCharacter.toUpperCase()) {
401 				try {
402 					if (child instanceof parent[key]) {
403 						return path + "." + key;
404 					}
405 				} catch(e) {
406 					// instanceof operator is a bit picky about it's operands
407 				}
408 			} else {
409 				// key is not a class name so continue walking the tree
410 				var matches = BBQUtil.findClassName(parent[key], child, path + "." + key);
411 
412 				if (matches) {
413 					return matches;
414 				}
415 			}
416 		}
417 	},
