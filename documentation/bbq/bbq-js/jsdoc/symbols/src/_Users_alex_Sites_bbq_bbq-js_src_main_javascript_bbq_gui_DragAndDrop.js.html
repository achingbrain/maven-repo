1 include(bbq.util.Log);
  2 
  3 /**
  4  * @class bbq.gui.DragAndDrop
  5  */
  6 bbq.gui.DragAndDrop = new Class.create({
  7 	_droppable: null,
  8 
  9 	initialize: function() {
 10 		try {
 11 
 12 		} catch(e) {
 13 			Log.error("Error constructing DragAndDrop", e);
 14 		}
 15 	},
 16 
 17 	_onDragStart: function(event) {
 18 		//Log.info("Noticed drag start");
 19 
 20 		this._droppable = null;
 21 
 22 		var element = Event.findElement(event);
 23 
 24 		if (!element.owner) {
 25 			return;
 26 		}
 27 
 28 		var owner = element.owner();
 29 
 30 		if (owner.getRootNode().draggable) {
 31 			this._droppable = owner;
 32 		}
 33 
 34 		if (owner.draggableStarted) {
 35 			owner.draggableStarted();
 36 		}
 37 
 38 		// FireFox and Chrome won't honour drag and drop unless some data is set
 39 		event.dataTransfer.setData('Text', "wtf bbq?!");
 40 	},
 41 
 42 	_onDrag: function(event) {
 43 		//Log.info("Noticed drag");
 44 	},
 45 
 46 	_onDragOver: function(event) {
 47 		//Log.info("Noticed drag over");
 48 
 49 		var element = Event.findElement(event);
 50 
 51 		// find drop targets
 52 		this._findDropTargets(element).each(function(target) {
 53 			if (target.dropTargetEnter) {
 54 				target.dropTargetEnter(this._droppable);
 55 			}
 56 		}.bind(this));
 57 
 58 		// must cancel this event in order for drop to file
 59 		Event.stop(event);
 60 	},
 61 
 62 	_onDragEnter: function(event) {
 63 		//Log.info("Noticed drag enter");
 64 
 65 		var element = Event.findElement(event);
 66 
 67 		// find drop targets
 68 		this._findDropTargets(element).each(function(target) {
 69 			if (target.dropTargetEnter) {
 70 				target.dropTargetEnter(this._droppable);
 71 			}
 72 		}.bind(this));
 73 
 74 		// must cancel this event in order for drop to file
 75 		Event.stop(event);
 76 	},
 77 
 78 	_onDragLeave: function(event) {
 79 		//Log.info("Noticed drag leave");
 80 
 81 		var element = Event.findElement(event);
 82 
 83 		// find drop targets
 84 		this._findDropTargets(element).each(function(target) {
 85 			if (target.dropTargetLeave) {
 86 				target.dropTargetLeave(this._droppable);
 87 			}
 88 		}.bind(this));
 89 	},
 90 
 91 	_onDrop: function(event) {
 92 		//Log.info("Noticed drop");
 93 
 94 		var element = Event.findElement(event);
 95 
 96 		// find drop targets
 97 		this._findDropTargets(element).each(function(target) {
 98 			if (target.dropTargetDropped) {
 99 				target.dropTargetDropped(this._droppable);
100 			}
101 		}.bind(this));
102 
103 		Event.stop(event);
104 	},
105 
106 	_onDragEnd: function(event) {
107 		//Log.info("Noticed drag end");
108 
109 		this._droppable = null;
110 
111 		var element = Event.findElement(event);
112 
113 		if (!element.owner) {
114 			return;
115 		}
116 
117 		var owner = element.owner();
118 
119 		if (owner.draggableStopped) {
120 			owner.draggableStopped();
121 		}
122 	},
123 
124 	makeDraggable: function(widget) {
125 		widget.setAttribute("draggable", true);
126 		widget.addClass("draggable");
127 
128 		widget.setStyle("-moz-user-select", "none");
129 		widget.setStyle("-khtml-user-select", "none");
130 		widget.setStyle("-webkit-user-select", "none");
131 		widget.setStyle("user-select", "none");
132 		widget.setStyle("-khtml-user-drag", "element");
133 
134 		Event.observe(widget.getRootNode(), "dragstart", this._onDragStart.bindAsEventListener(this));
135 		Event.observe(widget.getRootNode(), "dragend", this._onDragEnd.bindAsEventListener(this));
136 	},
137 
138 	makeDropTarget: function(widget) {
139 		widget.addClass("DragAndDropManager_dropTarget");
140 
141 		Event.observe(widget.getRootNode(), "dragover", this._onDragOver.bindAsEventListener(this));
142 		Event.observe(widget.getRootNode(), "dragenter", this._onDragEnter.bindAsEventListener(this));
143 		Event.observe(widget.getRootNode(), "dragleave", this._onDragLeave.bindAsEventListener(this));
144 		Event.observe(widget.getRootNode(), "drop", this._onDrop.bindAsEventListener(this));
145 	},
146 
147 	_findDropTargets: function(node) {
148 		var output = [];
149 
150 		// find drop target
151 		var parentNode = node;
152 
153 		while (parentNode) {
154 			if (parentNode.owner) {
155 				var owner = parentNode.owner();
156 
157 				if (owner.hasClass("DragAndDropManager_dropTarget") && owner.dropTargetWillAccept && owner.dropTargetWillAccept(this._droppable)) {
158 					output.push(owner);
159 				}
160 			}
161 
162 			parentNode = parentNode.parentNode;
163 		}
164 
165 		return output;
166 	}
167 });
