 85 	/**
 86 	 * @private
 87 	 * 
 88 	 * Recusively walks the tree building a gui representation
 89 	 */
 90 	_buildLevel: function(listHolder, entities, levelIndex) {
 91 		// loop through each entity
 92 		entities.each(function(entity){
 93 			try {
 94 				var listTitle = document.createElement("li");
 95 				listTitle.entity = entity;
 96 				listTitle.onclickcallback = this._listConstructionData[levelIndex].onclick;
 97 				
 98 				if(this._listConstructionData[levelIndex].customClass && this._listConstructionData[levelIndex].customClass instanceof Function) {
 99 					var customClass = this._listConstructionData[levelIndex].customClass(entity);
100 					
101 					if(customClass) {
102 						DOMUtil.addClass(listTitle, this._listConstructionData[levelIndex].customClass(entity));
103 					}
104 				}
105 				
106 				DOMUtil.addClass(listTitle, "CollapseExpandListTitle");
107 				var startCollapsed = false;
108 				
109 				this._entityIdToNode[entity[this.options.entityKey]] = listTitle;
110 				
111 				if(entity[this.options.entityKey] == this._currentlySelectedID) {
112 					DOMUtil.addClass(listTitle, "down");
113 				}
114 				
115 				// get child elements
116 				var childElements = [];
117 				
118 				if(this._listConstructionData[levelIndex+1] && this._listConstructionData[levelIndex+1].propertyName && entity[this._listConstructionData[levelIndex+1].propertyName]) {
119 					
120 					if(entity[this._listConstructionData[levelIndex+1].propertyName] instanceof Function) {
121 						childElements = entity[this._listConstructionData[levelIndex+1].propertyName]();
122 					} else {
123 						childElements = entity[this._listConstructionData[levelIndex+1].propertyName];
124 					}
125 				}
126 				
127 				if(this._listConstructionData[levelIndex].collapsible) { // add collapse/expand button
128 					if(childElements.length > 0) {
129 						if(typeof(this._folderOpenStatus[entity[this.options.entityKey]]) != "undefined") {
130 							startCollapsed = this._folderOpenStatus[entity[this.options.entityKey]];
131 						} else {
132 							startCollapsed = this._listConstructionData[levelIndex].startCollapsed(entity);
133 						}
134 						
135 						listTitle.collapsible = true;
136 						listTitle.collapsed = startCollapsed;
137 						
138 						this._folderOpenStatus[entity[this.options.entityKey]] = startCollapsed;
139 						
140 						listTitle.appendChild(DOMUtil.createTextElement("a", Language.get("bbq.gui.tree.CollapsableTreeList.collapseexpand"), {
141 							href: ".",
142 							onclick: this._folderClicked.bindAsEventListener(this),
143 							title: Language.get("bbq.gui.tree.CollapsableTreeList.collapseexpand"),
144 							className: "CollapseExpand  CollapseExpand" + (startCollapsed ? "Collapsed" : "Expanded"),
145 							collapsed: startCollapsed,
146 							entity: entity
147 						}));
148 					} else { // no children, add placeholder
149 						listTitle.appendChild(DOMUtil.createElement("span", {
150 							className: "CollapseExpand  CollapseExpandCollapserPlaceHolder"
151 						}));
152 					}
153 				}
154 				
155 				listTitle.appendChild(DOMUtil.createTextElement("a", this._listConstructionData[levelIndex].text(entity), {
156 						href: ".",
157 						onclick: this._elementClicked.bindAsEventListener(this, this._listConstructionData[levelIndex].onclick),
158 						entity: entity, 
159 						className: "CollapseExpandFolderTitle"
160 					}));
161 				
162 				var list = document.createElement("ul");
163 				list.entity = entity;
164 				list.appendChild(listTitle);
165 				
166 				// check for child elements
167 				if(childElements && childElements.length > 0) {
168 					var listElement = DOMUtil.createElement("li", {className: "CollapsableTreeListContainerList"});
169 					this._buildLevel(listElement, childElements, levelIndex+1);
170 					list.appendChild(listElement);
171 				
172 					if(startCollapsed) {
173 						list.getElementsByClassName("CollapsableTreeListContainerList")[0].style.display = "none";
174 					}
175 				}
176 				
177 				listHolder.appendChild(list);
178 			} catch(e) {
179 				Log.error("Error building level", e);
180 			}
181 		}.bind(this));
182 	},

195 	/**
196 	 * @private
197 	 * 
198 	 * @example
199 	 * Shows/hides a tree level.  The tree is structured thus:
200 	 * 
201 	 * <code>
202 	 * <ul>
203 	 *   <li>title 1</li>
204 	 *   <li class="CollapsableTreeListContainerList">
205 	 *     <ul>
206 	 *       <li>element 1</li>
207 	 *       <li>element 2</li>
208 	 *     </ul>
209 	 *   </li>
210 	 * </ul>
211 	 * <ul>
212 	 *   <li>title 2</li>
213 	 *   <li class="CollapsableTreeListContainerList">
214 	 *     <ul>
215 	 *       <li>element a</li>
216 	 *       <li>element b</li>
217 	 *     </ul>
218 	 *   </li>
219 	 * </ul>
220 	 * </code>
221 	 * 
222 	 * The passed element is the ul element that contains an li with the class CollapsableTreeListContainerList
223 	 * 
224 	 */
225 	_toggleFolder: function(element, listToCollapse) {
226 		try {
227 			var entity = element.entity;
228 			
229 			// reverse collapsed state
230 			element.collapsed = !element.collapsed;
231 			element.parentNode.parentNode.collapsible = true;
232 			element.parentNode.parentNode.collapsed = element.collapsed;
233 			
234 			// store locally to make it persist beyond the next render
235 			this._folderOpenStatus[entity[this.options.entityKey]] = element.collapsed;
236 			
237 			//set css classes
238 			DOMUtil.removeClass(element, (element.collapsed ? "CollapseExpandExpanded" : "CollapseExpandCollapsed"));
239 			DOMUtil.addClass(element, (element.collapsed ? "CollapseExpandCollapsed" : "CollapseExpandExpanded"));
240 			
241 			// show/hide element
242 			Effect["Blind" + (element.collapsed ? "Up" : "Down")](listToCollapse, {duration: 0.15});
243 			
244 			// store collapsed state
245 			new bbq.ajax.JSONRequest({url: "/backend/setFolderCollapsedStatus", args: {entity: element.entity.id, closed: element.collapsed}});		
246 		} catch(e) {
247 			Log.error("Error toggling folder", e);
248 		}
249 	},

295 	/**
296 	 * @param {Event} event
297 	 */
298 	processKeypress: function(event) {
299 		//Log.info("looking for " + this._currentlySelectedID);
300 		var nodes = this._getNodes(this._currentlySelectedID);
301 		
302 		// process keypress
303 		if(event.keyCode == Event.KEY_DOWN) { // if we can go one further down the list, do it but skip the contents of closed folders
304 			if(nodes.next) {
305 				var nextNode = this._getNextFolder(nodes, "next");
306 				
307 				if(nextNode && nextNode.onclickcallback) {
308 					this.setItemDownWithId(nextNode.entity[this.options.entityKey]);
309 					nextNode.onclickcallback.call(this, nextNode.entity);
310 				}
311 			}
312 		} else if(event.keyCode == Event.KEY_UP) {  // if we can go one further up the list, do it but skip the contents of closed folders
313 			if(nodes.prev) {
314 				var prevNode = this._getNextFolder(nodes, "previous");
315 				
316 				if(prevNode && prevNode.onclickcallback) {
317 					this.setItemDownWithId(prevNode.entity[this.options.entityKey]);
318 					prevNode.onclickcallback.call(this, prevNode.entity);
319 				}
320 			}
321 		} else if(event.keyCode == Event.KEY_RIGHT) { // if the currently selected level is collapsible and collapsed, expand it
322 			var folderLink = nodes.current.getElementsByTagName("a")[0];
323 			
324 			if(nodes.current.collapsible && folderLink.collapsed) {
325 				this._toggleFolder(folderLink, Element.getElementsByClassName(nodes.current.parentNode, "CollapsableTreeListContainerList")[0]);
326 			}
327 		} else if(event.keyCode == Event.KEY_LEFT) { // if the currently selected level is collapsible and expanded, collapse it
328 			var folderLink = nodes.current.getElementsByTagName("a")[0];
329 			
330 			if(nodes.current.collapsible && !folderLink.collapsed) {
331 				this._toggleFolder(folderLink, Element.getElementsByClassName(nodes.current.parentNode, "CollapsableTreeListContainerList")[0]);
332 			}
333 		}
334 		
335 		return true;
336 	},

338 	/**
339 	 * @param {Node} node
340 	 */
341 	_goodNode: function(node) {
342 		//Log.warn("testing " + node.tagName);
343 		
344 		while(node.tagName == "UL" || node.tagName == "LI") {
345 			//Log.warn("testing " + node.tagName);
346 			
347 			if(node.entity && this._folderOpenStatus[node.entity[this.options.entityKey]]) {
348 				if(DOMUtil.hasClass(node, "CollapseExpandListTitle")) { // allow highlighting of collapsed treetops
349 					return true;
350 				}
351 				
352 				//Log.warn("node hidden");
353 				return false;
354 			}
355 			
356 			node = node.parentNode;
357 		}
358 		
359 		//Log.warn("node visible");
360 		return true;
361 	},

363 	/**
364 	 * Returns the next/previous folder that isn't collapsed
365 	 * @param {Array} nodes
366 	 * 		startNode	One of the ul nodes that the list is made up of
367 	 * 		nextNode	The next li node in the series - may or may not be a
368 	 * @param	{String}	direction	previous or next.  Defaults to next
369 	 */
370 	_getNextFolder: function(nodes, direction) {
371 		direction = direction == "previous" ? direction : "next";
372 		
373 		if(direction == "previous") {
374 				for(var i = nodes.index - 1; i >= 0; i--) {
375 					if(this._goodNode(nodes.nodes[i])) {
376 						return nodes.nodes[i];
377 					}
378 			}
379 		} else {
380 				for(var i = nodes.index +1; i < nodes.nodes.length; i++) {
381 					if(this._goodNode(nodes.nodes[i])) {
382 						return nodes.nodes[i];
383 					}
384 				}
385 		}
386 		
387 		return false;
388 	},

390 	_getNodes: function(id) {
391 		var childNodes = [];
392 		
393 		$A(this.getRootNode().getElementsByTagName("li")).each(function(node){
394 			if(node.entity) {
395 				childNodes.push(node);
396 			}
397 		});
398 		
399 		var output = {
400 			prev: false,
401 			current: false,
402 			next: false,
403 			nodes: childNodes,
404 			index: -1
405 		};
406 		
407 		for(var i = 0; i < childNodes.length; i++) {
408 			if(childNodes[i].entity) {
409 				if(childNodes[i].entity[this.options.entityKey] == id) {
410 					output.index = i;
411 					output.prev = childNodes[i-1];
412 					output.current = childNodes[i];
413 					output.next = childNodes[i+1];
414 					break;
415 				}
416 			}
417 		}
418 		
419 		return output;
420 	}
