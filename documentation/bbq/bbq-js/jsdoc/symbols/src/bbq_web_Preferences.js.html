1 include(bbq.util.PersistenceUtil);
  2 
  3 /**
  4  * User preferences mechanism.
  5  * 
  6  * Uses HTML5 localStorage where available otherwise
  7  * falls back to cookies.
  8  * 
  9  * <code>
 10  * var foo = {bar: "baz"};
 11  * 
 12  * Preferences.set("wibble", foo);
 13  * 
 14  * ... time passes ...
 15  * 
 16  * var foo = Preferences.get("wibble");
 17  * 
 18  * alert(foo.bar);
 19  * 
 20  * </code>
 21  *
 22  * @class Preferences
 23  */
 24 Preferences = {
 25 	/**
 26 	 * Holds a Preferences implementation.  Should support three methods:
 27 	 * 
 28 	 * set(key:String, value:String):Void
 29 	 * get(key:String):String
 30 	 * del(key:String)
 31 	 * 
 32 	 * Value will be JSON encoded prior to being set and decocde after get.
 33 	 */
 34 	implementation: null,
 35 	
 36 	/**
 37 	 * Pass in a value to persist under the passed key
 38 	 */
 39 	set: function(key, value) {
 40 		if(value) {
 41 			this.implementation.set(key, PersistenceUtil.serialize(value));
 42 		} else {
 43 			this.implementation.del(key);
 44 		}
 45 	},
 46 	
 47 	/**
 48 	 * Pass in a key to retrieve the stored value with an option value to set as the default
 49 	 * in case the value stored is undefined.
 50 	 */
 51 	get: function(key, defaultValue) {
 52 		var value = this.implementation.get(key);
 53 
 54 		if(value == null && !Object.isUndefined(defaultValue)) {
 55 			Preferences.set(key, defaultValue);
 56 
 57 			return defaultValue;
 58 		}
 59 		
 60 		// if a string returned, decode it
 61 		if(Object.isString(value)) {
 62 			return PersistenceUtil.deserialize(value);
 63 		}
 64 		
 65 		return value;
 66 	},
 67 	
 68 	/**
 69 	 * Deletes the value stored under the passed key
 70 	 */
 71 	del: function(key) {
 72 		this.implementation.del(key);
 73 	},
 74 	
 75 	/**
 76 	 * Preferences implementation that uses HTML5 storage.
 77 	 * 
 78 	 * Used when the browser supports it.
 79 	 */
 80 	_localStorage: {
 81 		set: function(key, value) {
 82 			localStorage.setItem(key, value);
 83 		},
 84 		
 85 		get: function(key) {
 86 			return localStorage.getItem(key);
 87 		},
 88 		
 89 		del: function(key) {
 90 			localStorage.removeItem(key);
 91 		}
 92 	},
 93 	
 94 	/**
 95 	 * Preferences implementation that uses cookie storage.
 96 	 * 
 97 	 * Used when the browser doesn't support HTML5.
 98 	 */
 99 	_cookies: {
100 		set: function(key, value) {
101 			Cookie.set(key, value);
102 		}, 
103 		
104 		get: function(key) {
105 			return Cookie.get(key);
106 		}, 
107 		
108 		del: function(key) {
109 			Cookie.del(key);
110 		}
111 	}
112 };
113 
114 // set preferences implementation according to browser support
115 if(window.localStorage) {
116 	// use HTML5 session storage
117 	Preferences.implementation = Preferences._localStorage;
118 } else {
119 	// fallback to cookies
120 	Preferences.implementation = Preferences._cookies;
121 }
