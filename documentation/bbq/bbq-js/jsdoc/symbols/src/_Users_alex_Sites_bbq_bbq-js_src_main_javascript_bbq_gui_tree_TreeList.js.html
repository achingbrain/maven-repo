1 include(bbq.gui.GUIWidget);
  2 include(bbq.util.BBQUtil);
  3 
  4 /**
  5  * @class bbq.gui.tree.TreeList
  6  * @extends bbq.gui.GUIWidget 
  7  */
  8 bbq.gui.tree.TreeList = new Class.create(bbq.gui.GUIWidget, {
  9 	
 10 	_entities: null,
 11 	_selectedID: null,
 12 	_selectedIndex: null,
 13 	
 14 	/**
 15 	 * @param {Object} options
 16 	 * @example
 17 	 * Todo an example
 18 	 */
 19 	initialize: function($super, options) {
 20 		$super(options);
 21 		
 22 		this._entities = [];
 23 		
 24 		this.setRootNode("ul");
 25 		this.addClass("TreeList");
 26 	},
 27 	
 28 	/**
 29 	 * @param {Object} content
 30 	 * @param {Object} parentNode
 31 	 * @param {string} elementID 
 32 	 */	
 33 	addEntity: function(text, onclick, id) {
 34 		this._entities.push({
 35 			content: text,
 36 			callback: onclick,
 37 			id: id
 38 		});
 39 	},
 40 	
 41 	render: function() {
 42 		this.empty();
 43 		
 44 		this._entities.each(function(entity) {
 45 			var listElement = DOMUtil.createTextElement("li", entity.callback ? DOMUtil.createTextElement("a", entity.content, {href: ".", onclick: this._processClick.bindAsEventListener(this, entity)}) : entity.content);
 46 			
 47 			if(entity.id == this._selectedID) {
 48 				DOMUtil.addClass(listElement, "down");
 49 			}
 50 			
 51 			this.appendChild(listElement);
 52 		}.bind(this));
 53 	},
 54 	
 55 	_processClick: function(event, entity) {
 56 		if(FocusWatcher) {
 57 			FocusWatcher.setKeypressCallbackObject(this);
 58 		}
 59 		
 60 		try {
 61 			var element = BBQUtil.clearFocus(event);
 62 		
 63 			if(entity.callback) {
 64 				entity.callback();
 65 			}
 66 		} catch(e) {
 67 			Log.error("Error processing click callback", e);
 68 		}
 69 		
 70 		this.setDown(entity);
 71 		
 72 		return false;
 73 	},
 74 	
 75 	setDown: function(entity) {
 76 		if(entity) {
 77 			this.clearDown();
 78 			
 79 			var id;
 80 			
 81 			if(entity.getId && entity.getId instanceof Function) {
 82 				id = entity.getId();
 83 			} else if(typeof(entity.id) != "undefined") {
 84 				id = entity.id;
 85 			} else {
 86 				id = entity;
 87 			}
 88 			
 89 			var listElements = $A(this.getRootNode().getElementsByTagName("li"));
 90 			
 91 			this._entities.each(function(currentEntity, index) {
 92 				if(currentEntity.id == id) {
 93 					DOMUtil.addClass(listElements[index], "down");
 94 					this._selectedID = id;
 95 					this._selectedIndex = index;
 96 				}
 97 			}.bind(this));
 98 		}
 99 	},
100 	
101 	clearDown: function() {
102 		this.removeClass("down", true);
103 		this._selectedID = null;
104 		this._selectedIndex = null;
105 	},
106 	
107 	acceptFocus: function() {
108 		this.addClass("hasFocus");
109 	},
110 	
111 	loseFocus: function() {
112 		this.removeClass("hasFocus");
113 	},
114 	
115 	processKeypress: function(event) {
116 		var nextEntity = false;
117 		
118 		// process keypress
119 		if(event.keyCode == Event.KEY_DOWN) {
120 			if(this._entities[this._selectedIndex + 1]) {
121 				nextEntity = this._entities[this._selectedIndex + 1];
122 			}
123 		} else if(event.keyCode == Event.KEY_UP) {
124 			if(this._entities[this._selectedIndex - 1]) {
125 				nextEntity = this._entities[this._selectedIndex - 1];
126 			}
127 		}
128 		
129 		if(nextEntity) {
130 			this._processClick(null, nextEntity);
131 		}
132 		
133 		return true;
134 	}
135 });
