1 include(bbq.gui.GUIWidget);
  2 include(bbq.gui.updateables.InputException);
  3 include(bbq.web.FocusWatcher);

  5 /**
  6  * @class bbq.gui.updateables.UpdateableField
  7  * @extends bbq.gui.GUIWidget
  8  */
  9 bbq.gui.updateables.UpdateableField = new Class.create(bbq.gui.GUIWidget, {
 10 	
 11 	editMode: null,
 12 	disabled: null,
 13 	displayValue: null,
 14 	currentValue: null,
 15 	originalValue: null,
 16 	shortenDisplayField: null,
 17 	_changed: null,
 18 	_entityDataLoadedCallbackKey: null,
 19 	_entityDataUpdatedCallbackKey: null,
 20 	
 21 	/**
 22 	 * @param {Object} options
 23 	 * @example
 24 	 * Supports the following options:
 25 	 * 
 26 	 * options: {
 27 	 * 		initialValue:	mixed							// Some sort of value
 28 	 * 		propertyDisplay: Object					// {key: String, entity: Object}  If passed, ignores initialValue and will use #getPropertyDisplay(key, formatter) on the passed object for display value instead 
 29 	 * 		required:	boolean
 30 	 * 		errorText: String								// thrown as an exception if the field has no value
 31 	 * 		shortenDisplayField: boolean
 32 	  * }
 33 	 */
 34 	initialize: function($super, options) {
 35 		$super(options);
 36 		
 37 		this.setRootNode("span");
 38 		
 39 		this.shortenDisplayField = this.options.shortenDisplayField;
 40 		this.editMode = false;
 41 		
 42 		this._setInitialValues();
 43 		
 44 		this.disabled = false;
 45 		this._changed = false
 46 		this.addClass("UpdateableField");
 47 		
 48 		if(!Object.isUndefined(this.options.propertyDisplay)) {
 49 			// set up property display watchers
 50 			this.setPropertyDisplay(this.options.propertyDisplay.key, this.options.propertyDisplay.entity);
 51 		}
 52 		
 53 		// flag this field as updated when value is changed
 54 		this.registerListener("onValueChanged", function() {
 55 			this._changed = true;
 56 		}.bind(this));
 57 	},
 58 	
 59 	setPropertyDisplay: function(key, entity) {
 60 		if(Object.isUndefined(this.options.propertyDisplay)) {
 61 			this.options.propertyDisplay = {};
 62 		} else {
 63 			if(this._entityDataLoadedCallbackKey) {
 64 				// if we were already watching an object for updates, stop watching it
 65 				this.options.propertyDisplay.entity.deRegisterListener("onDataLoaded", this._entityDataLoadedCallbackKey);
 66 			}
 67 			
 68 			if(this._entityDataUpdatedCallbackKey) {
 69 				// if we were already watching an object for updates, stop watching it
 70 				this.options.propertyDisplay.entity.deRegisterListener("onUpdate", this._entityDataUpdatedCallbackKey);
 71 			}
 72 		}
 73 		
 74 		this.options.propertyDisplay.key = key;
 75 		this.options.propertyDisplay.entity = entity;
 76 		
 77 		// register for repeat notifications to update display when entity is remotely updated
 78 		this._entityDataLoadedCallbackKey = this.options.propertyDisplay.entity.registerListener("onDataLoaded", function() {
 79 			this._updateInitialValues();
 80 			this.render();
 81 		}.bind(this));
 82 		
 83 		// register for repeat notifications to update display when entity is remotely updated
 84 		this._entityDataUpdatedCallbackKey = this.options.propertyDisplay.entity.registerListener("onUpdate", function() {
 85 			this._updateInitialValues();
 86 			this.render();
 87 		}.bind(this));
 88 		
 89 		if(entity.dataLoaded()) {
 90 			this._updateInitialValues();
 91 			this.render();
 92 		}
 93 	},
 94 	
 95 	_setInitialValues: function() {
 96 		if(!Object.isUndefined(this.options.propertyDisplay)) {
 97 			
 98 		} else {
 99 			if(Object.isUndefined(this.options.initialValue) || this.options.initialValue == null) {
100 				this.options.initialValue = "";
101 			}
102 			
103 			this._setCurrentValue(this.options.initialValue);
104 			this._setOriginalValue(this.options.initialValue);
105 			this._setDisplayValue(this.options.initialValue);
106 		}
107 	},
108 	
109 	_updateInitialValues: function() {
110 		this._setCurrentValue(this.options.propertyDisplay.entity.getProperty(this.options.propertyDisplay.key));
111 		this._setOriginalValue(this.options.propertyDisplay.entity.getProperty(this.options.propertyDisplay.key));
112 		this._setDisplayValue(this.options.propertyDisplay.entity.getProperty(this.options.propertyDisplay.key));
113 	},
114 	
115 	/*
116 	 * resets the internal field value to that prior editing
117 	 */
118 	cancelEdit: function() {
119 		this._setCurrentValue(this.originalValue);
120 		this._setDisplayValue(this.originalValue);
121 	},
122 	
123 	createEditField: function() {
124 		Log.info("Non overridden createEditField() method called");
125 	},
126 	
127 	/*
128 	 * creates view field
129 	 */
130 	createViewField: function() {
131 		if(typeof(this.options.propertyDisplay) != "undefined") {
132 			return this._createViewFieldFromPropertyDisplay();
133 		}
134 		
135 		if(this._getDisplayValue() && this._getDisplayValue().length > 0) {
136 			return document.createTextNode(this._getTruncatedValue());
137 		}
138 		
139 		return document.createTextNode(" ");
140 	},
141 	
142 	_getCurrentValue: function() {
143 		if(typeof(this.options.propertyDisplay) != "undefined" && typeof(this.currentValue) == "undefined") {
144 			this.currentValue = this.options.propertyDisplay.entity.getProperty(this.options.propertyDisplay.key);
145 		}
146 		
147 		return this.currentValue;
148 	},
149 	
150 	_setCurrentValue: function(value) {
151 		this.currentValue = value;
152 	},
153 	
154 	_getOriginalValue: function() {
155 		if(typeof(this.options.propertyDisplay) != "undefined") {
156 			return this.options.propertyDisplay.entity.getProperty(this.options.propertyDisplay.key);
157 		}
158 		
159 		return this.originalValue;
160 	},
161 	
162 	_setOriginalValue: function(value) {
163 		this.originalValue = value;
164 	},
165 	
166 	_getDisplayValue: function() {
167 		if(typeof(this.options.propertyDisplay) != "undefined" && !this.displayValue) {
168 			this.displayValue = this.options.propertyDisplay.entity.getProperty(this.options.propertyDisplay.key);
169 		}
170 		
171 		return this.displayValue;
172 	},
173 	
174 	_setDisplayValue: function(value) {
175 		this.displayValue = value;
176 	},
177 	
178 	_createViewFieldFromPropertyDisplay: function() {
179 		return this.options.propertyDisplay.entity.getPropertyDisplay({property: this.options.propertyDisplay.key, formatter: this.options.propertyDisplay.formatter});
180 	},
181 	
182 	_getTruncatedValue: function() {
183 		if(this.shortenDisplayField && this._getDisplayValue().length && this._getDisplayValue().length > 20) {
184 			return  this._getDisplayValue().substr(0, 9) + "..." + this._getDisplayValue().substr((this._getDisplayValue().length - 8), 8);
185 		}
186 		
187 		return this._getDisplayValue();
188 	},
189 	
190 	/**
191 	 * @param {Object} value
192 	 */
193 	setValue: function(value) {
194 		this._setCurrentValue(value);
195 		this._setDisplayValue(value);
196 		this.render();
197 	},
198 	
199 	/**
200 	 * @param {boolean} supressErrorWarning
201 	 * @return Returns the current field value 
202 	 * @type {Object}
203 	 */
204 	getValue: function(supressErrorWarning) {
205 		if(!supressErrorWarning && !this.checkForRequired()) {
206 			this.triggerError();
207 		}
208 		
209 		return this._getCurrentValue();
210 	},
211 	
212 	triggerError: function() {
213 		this.setError(true);
214 		this.notifyListeners("onError");
215 		throw new bbq.gui.updateables.InputException(this.options.errorText ? this.options.errorText : "field needs value");
216 	},
217 	
218 	/**
219 	 * draws the field either as a form for editing or text for display
220 	 */
221 	render: function() {
222 		this.empty();
223 		
224 		if(this.editMode) { // create edit node
225 			this.appendChild(this.createEditField());
226 			
227 			if(this.options.notes) {
228 				this.appendChild(DOMUtil.createTextElement("span", this.options.notes, {className: "UpdateableField_Notes"}));
229 			}
230 		} else { // create view node
231 			var view = this.createViewField();
232 			
233 			if(Object.isString(view)) {
234 				this.appendChild(document.createTextNode(view));
235 			} else {
236 				this.appendChild(view);
237 			}
238 		}
239 		
240 		if(this._getCurrentValue() != this._getOriginalValue()) {
241 			this.checkForRequired();
242 		}
243 	},
244 	
245 	/**
246 	 * saves the current field value internally
247 	 * 
248 	 */
249 	saveEdit: function() {
250 		this._setDisplayValue(this._getCurrentValue());
251 		this._setOriginalValue(this._getCurrentValue());
252 	},
253 	
254 	/**
255 	 * @param {boolean} disabled
256 	 */
257 	setDisabled: function(disabled) {
258 		if(this.disabled != disabled) {
259 			this.disabled = disabled;
260 			this.render();
261 		}
262 	},
263 	
264 	/**
265 	 * @param {integer} editMode
266 	 */
267 	setEditMode: function(editMode) {
268 		this.editMode = editMode;
269 		this.render();
270 	},
271 	
272 	getEditMode: function() {
273 		return this.editMode;
274 	},
275 	
276 	/**
277 	 * updates current value from input field
278 	 */
279 	updateLocalValue: function() {
280 		this._setCurrentValue(this.inputField.value);
281 		this.checkForRequired();
282 		this.notifyListeners("onValueChanged");
283 	},
284 	
285 	/**
286 	 * check if updateable field is required.  Returns false if we are required and do not have a value, true otherwise
287 	 */
288 	checkForRequired: function() {
289 		if(this.getEditMode()) {
290 			if(this.options.required && (!this._getCurrentValue() || (this._getCurrentValue().strip && this._getCurrentValue().strip() == ""))) {
291 				return false;
292 			}
293 		}
294 		
295 		return true;
296 	},
297 	
298 	/**
299 	 * @private
300 	 */
301 	_clearFocusWatcher: function() {
302 		FocusWatcher.setKeypressCallbackObject(this);
303 	},
304 	
305 	/**
306 	 * clears an error by removing the "error" class from the DOM
307 	 */
308 	clearError: function() {
309 		this.removeClass("error");
310 	},
311 	
312 	setError: function(error) {
313 		this.removeClass("error");
314 		
315 		if(error) {
316 			this.addClass("error");
317 		}
318 	},
319 	
320 	focus: function() {
321 		this.inputField.focus();
322 	},
323 	
324 	blur: function() {
325 		this.inputField.blur();
326 	},
327 	
328 	changed: function() {
329 		return this._changed;
330 	}
331 });
