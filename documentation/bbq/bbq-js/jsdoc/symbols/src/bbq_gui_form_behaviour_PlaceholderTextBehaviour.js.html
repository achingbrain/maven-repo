1 include(bbq.web.Browser);
  2 
  3 /**
  4  * Supports the following options:
  5  *
  6  * options: {
  7  *      text: String                                     // the placeholder text
  8  * }
  9  *
 10  * @class bbq.gui.form.behaviour.PlaceholderTextBehaviour
 11  */
 12 bbq.gui.form.behaviour.PlaceholderTextBehaviour = new Class.create({
 13 	options: null,
 14 
 15 	initialize: function(options) {
 16 		this.options = options;
 17 	},
 18 
 19 	setField: function(field) {
 20 		// respect HTML5 style placeholder
 21 		if(Browser.forms.placeholderText) {
 22 			field.setAttribute("placeholder", this.options.text);
 23 
 24 			return;
 25 		}
 26 
 27 		// shim in placeholder for non-HTML5 browsers
 28 		if(!field.getRootNode().value) {
 29 			field.getRootNode().value = this.options.text;
 30 			field.addClass("placeholderText");
 31 		}
 32 
 33 		Event.observe(field.getRootNode(), "focus", function() {
 34 			if(field.getRootNode().value == this.options.text) {
 35 				field.getRootNode().value = "";
 36 				field.removeClass("placeholderText");
 37 			}
 38 		}.bind(this));
 39 
 40 		Event.observe(field.getRootNode(), "blur", function() {
 41 			if(!field.getRootNode().value) {
 42 				field.getRootNode().value = this.options.text;
 43 				field.addClass("placeholderText");
 44 			}
 45 		}.bind(this));
 46 	}
 47 });
