 17 	/**
 18 	 * A table that loads data from somewhere. 
 19 	 * 
 20 	 * The URL passed as part of the options object should return in the following format:
 21 	 * 
 22 	 *  { numEntities: int, entities: {"en7": data, "en8": data, "en9": data,...}}     // <-  enN is zero indexed!
 23 	 * 
 24 	 * @param	Object		options					Options for the whole table
 25 	 * 
 26 	 * Supports the following options:
 27 	 * 
 28 	 * options: {
 29 	 * 		rowHeight: int																	// roughly the expected height of one row.
 30 	 * 		createEntity: function(data, index)								// should return one new entity
 31 	 * 		url: String																			// where to load from
 32 	 * 		args: Object																		// arguments to pass to the url
 33 	 * 		loadingText: String															// shown while doin the inital load
 34 	 * 		noEntitiesText:	String														// shown when there is nothing to show
 35 	 * 		defaultActionColumn: int												// the index of the column who's onclick handler should be invoked when the user keypresses over the rows
 36 	 * 		columnData = [{
 37 	 * 			text: function(entity, index, cell, ScrollLoadTable)	// Should return either a string, a node or a GUIButton
 38 	 * 			shorten: int																	// The maximum length of the field value - nb. this will only be applied if a string is returned from text()
 39 	 * 			sortField: boolean														// If we should allow sorting by this column
 40 	 * 			ignoreDownState: bool												// Will not set the down state for the row if a link in this column is clicked
 41 	 * 			onclick: function(entity)												// Invoked if the text is clicked
 42 	 * 			ondblclick: function(entity)										// Invoked if the text is double clicked
 43 	 * 			className: function(entity)										// Should return a string or null to get added as a CSS class to the td element
 44 	 * 			
 45 	  * 	}, {...}, {...}]
 46 	 * }
 47 	 * 
 48 	 * N.B. The down state referred to below means that a specified tr element will have a CSS class titled "down" applied to it.
 49 	 * 
 50 	 * Supports observers for the following events
 51 	 * 
 52 	 * onEntitiesLoad
 53 	 * onEmptyList
 54 	 * onEntitiesLoaded
 55 	 * onSort
 56 	 * onBeforeSort
 57 	 * onAfterSort
 58 	 * 
 59 	 */
 60 	initialize: function($super, options) {
 61 		$super(options);
 62 		
 63 		this.addClass("ScrollLoadTable");
 64 		this._table = DOMUtil.createElement("table", {className: "ScrollLoadTable"});
 65 		
 66 		if(this.options.headerRows) {
 67 			this._tableHead = this._table.appendChild(DOMUtil.createElement("thead"));
 68 		}
 69 		
 70 		if(typeof(this.options.defaultActionColumn) == "undefined") {
 71 			this.options.defaultActionColumn = 0;
 72 		}
 73 		
 74 		this._tableBody = this._table.appendChild(DOMUtil.createElement("tbody"));
 75 		
 76 		this._selectedIndex = -1;
 77 		this._selectedKey = false;
 78 	},

 80 	/**
 81 	 * Parses through the entity list and creates the table
 82 	 */
 83 	render: function() {
 84 		var scrollTop = 0;
 85 		
 86 		if(this._scrollableHolder) {
 87 			scrollTop = this._scrollableHolder.getScrollTop();
 88 		}
 89 		
 90 		this.empty();
 91 		
 92 		var eLength = this._entities.getLength();
 93 		
 94 		if(this._initialLoad == true) {
 95 			// show loading text
 96 			this.appendChild(new bbq.gui.LoadingNotification({text: this.options.loadingText}));
 97 		} else if(eLength == 0) {
 98 			// show "no items" type notification
 99 			this.appendChild(DOMUtil.createTextElement("p", this.options.noEntitiesText ? this.options.noEntitiesText : Language.get("bbq.gui.table.ScrollLoadTable.emptylist"), {className: "ScrollLoadTableEmptyList"}));
100 		} else {
101 			try {
102 				// create header rows
103 				if(this.options.headerRows) {
104 					DOMUtil.emptyNode(this._tableHead);
105 					this._tableHead.appendChild(DOMUtil.createTextElement("tr", this._generateHeaderRows()));
106 				}
107 				
108 				DOMUtil.emptyNode(this._tableBody);
109 				
110 				// create one row for each entity
111 				for(var i = 0; i < eLength; i++) {
112 					this._tableBody.appendChild(this._generateRow(i, this._entities.getElement(i)));
113 				}
114 				
115 				// add style to last row
116 				if(eLength > 0) {
117 					DOMUtil.addClass(this._tableBody.childNodes.item(this._tableBody.childNodes.length - 1), "lastRow");
118 				}
119 				
120 				this.appendChild(this._table);
121 			} catch(e) {
122 				Log.error("Error while rendering", e);
123 			}
124 		}
125 		
126 		if(this._scrollableHolder) {
127 			this._scrollableHolder.setScrollTop(scrollTop);
128 		}
129 	},
130 	
131 	_generateHeaderRows: function() {
132 		var cells = [];
133 		
134 		for(var i = 0; i < this.options.headerRows.length; i++) {
135 			var cell = document.createElement("th");
136 			
137 			if(!this.options.columnData[i]["sortField"] || typeof(this.options.columnData[i]["sortField"]) == "undefined") {
138 				// not a header that the data is supposed to be sortable by
139 				cell.appendChild(document.createTextNode(this.options.headerRows[i]));
140 			} else {
141 				// create link that when clicked sorts the table
142 				var anchor = DOMUtil.createTextElement("a", this.options.headerRows[i], {href: ".", onclick: this.sortByIndex.bind(this, i, this.options.args.sortDirection == foundation.constants.FOUNDATION_DATABASE_PAGINATOROPTIONS_SORTDESC ? foundation.constants.FOUNDATION_DATABASE_PAGINATOROPTIONS_SORTASC : foundation.constants.FOUNDATION_DATABASE_PAGINATOROPTIONS_SORTDESC)});
143 							
144 				// are we currently sorted by this header?
145 				if(this.options.args.sortBy == i) {
146 					DOMUtil.addClass(anchor, "sortedBy" + ((this.options.args.sortDirection == foundation.constants.FOUNDATION_DATABASE_PAGINATOROPTIONS_SORTDESC) ? "DESC" : "ASC"));
147 				}
148 				
149 				cell.appendChild(anchor);
150 			}
151 			
152 			if(this.options.columnData[i] && this.options.columnData[i]["iconColumn"]) {
153 				DOMUtil.addClass(cell, "iconColumnHeader");
154 			}
155 			// add cell to row
156 			cells.push(cell);
157 		}
158 		
159 		return cells;
160 	},
161 	
162 	_generateRow: function(index, entity) {
163 		var row = document.createElement("tr");
164 		
165 		if(!entity) { // invalid item
166 			row = this._handleInvalidItem(row, index);
167 		} else if(this.options.createRow instanceof Function) { // custom item
168 			row = this.options.createRow(row, entity, index);
169 			row.owningEntity = entity;
170 		} else { // create row from column data
171 			row = 	this._handleValidItem(row, index, entity);
172 			
173 			if(this.isSelectedEntity(entity)) {
174 				DOMUtil.addClass(row, "down");
175 			}
176 		}
177 		
178 		if(index % 2 == 0) {
179 			DOMUtil.addClass(row, "oddRow");
180 		} else {
181 			DOMUtil.addClass(row, "evenRow");
182 		}
183 		
184 		return row;
185 	},
186 	
187 	_handleValidItem: function(row, rowIndex, entity) {
188 		var cData = this.options.columnData;
189 		
190 		for(var n = 0, nCount=this.options.columnData.length; n < nCount; n++) { // process one entity (eg. one row)
191 			var cell = row.appendChild(DOMUtil.createElement("td", {className: "column" + n}));
192 			
193 			if(this.options.columnData[n].iconColumn) {
194 				DOMUtil.addClass(cell, "ScrollLoadTableIconColumn");
195 			}
196 			
197 			if(!(this.options.columnData[n].text instanceof Function)) {
198 				Log.error("Please pass an instance of Function to ScrollLoadTable to generate cell text");
199 				return;
200 			}
201 			
202 			var cellContent = this.options.columnData[n].text(entity, n, cell, this);
203 			
204 			if(cellContent) {
205 				if(cellContent.appendTo) {
206 					// passed GUIWidget
207 					cellContent.appendTo(cell);
208 				} else {
209 					// passed text
210 					if(this.options.columnData[n].onclick || this.options.columnData[n].ondblClick) {
211 						// make text clickable
212 						var link = DOMUtil.createTextElement("a", cellContent, {
213 							href: ".",
214 							onclick: this._processEvent.bindAsEventListener(this, entity),
215 							ondblclick: this._processEvent.bindAsEventListener(this, entity)
216 						});
217 						
218 						if(Object.isElement(cellContent)) {
219 							Event.observe(cellContent, "click", function() {
220 								link.blur();
221 								return true;
222 							});
223 						}
224 						
225 						cellContent = link;
226 						
227 						if(this.options.columnData[n].onclick) {
228 							this._storeCallback(entity, "click", this.options.columnData[n].onclick.bind(this, entity));
229 						}
230 						
231 						if(this.options.columnData[n].ondblclick) {
232 							this._storeCallback(entity, "dblclick", this.options.columnData[n].ondblclick.bind(this, entity));
233 						}
234 					}
235 					
236 					if(Object.isString(cellContent)) {
237 						cellContent = document.createTextNode(cellContent);
238 					}
239 					
240 					cell.appendChild(cellContent);
241 				}
242 			
243 				if(this.options.columnData[n]["className"]) {
244 					var className;
245 					
246 					if(className = this.options.columnData[n]["className"](entity)) {
247 						//Log.info("custom class " + className);
248 						DOMUtil.addClass(cellContent, className);
249 					}
250 				}
251 			}
252 		}
253 		return row;
254 	},

256 	/**
257 	 * Sets the down state on the row containing the passed element
258 	 * 
259 	 * @param	{mixed}	object	Either an entity contained within this table or an ID
260 	 */
261 	setDown: function(entity) {
262 		this.notifyListeners("onBeforeSelectedEntityChange");
263 		if(!entity) {
264 			this.clearDown();
265 		} else {
266 			var index = this._entities.indexOf(entity);
267 			
268 			if(index != this._selectedIndex) {
269 				this.clearDown();
270 				this._selectedIndex = index;
271 				this._selectedKey =  this._getEntityKey(entity);
272 				
273 				var rows = $A(this._tableBody.getElementsByTagName("tr"));
274 				DOMUtil.addClass(rows[this._selectedIndex], "down");
275 				
276 				bbq.gui.ScrollLoadGUIWidget.prototype.setDown.call(this, entity);
277 			}
278 		}
279 		this.notifyListeners("onAfterSelectedEntityChange");
280 	},
281 	
282 	/**
283 	 * Removes the down state from all table rows
284 	 */
285 	clearDown: function() {
286 		this._selectedIndex = -1;
287 		this._selectedKey = false;
288 		DOMUtil.removeClass(this.getRootNode(), "down", true);
289 	},

299 	processKeypress: function(event) {
300 		var nextEntity = false;
301 		
302 		// process keypress
303 		if(event.keyCode == Event.KEY_DOWN) {
304 			nextEntity = this._entities.getElement(this._selectedIndex + 1);
305 		} else if(event.keyCode == Event.KEY_UP) {
306 			nextEntity = this._entities.getElement(this._selectedIndex - 1);
307 		}
308 		
309 		if(nextEntity) {
310 			this.setDown(nextEntity);
311 			this._invokeCallback(nextEntity, "click");
312 		}
313 		
314 		return true;
315 	},
316 	
317 	isSelectedEntity: function(entity) {
318 		if(this._selectedKey && entity) {
319 			return this._selectedKey ==  this._getEntityKey(entity);
320 		}
321 		
322 		return false;
323 	},

326 		row.appendChild(DOMUtil.createTextElement("td", new bbq.gui.LoadingNotification(), {className: "column0", colSpan: this.options.columnData.length, style: {height: this.options.rowHeight + "px"}}));
327 		return row;

331 		if(this.getRootNode().getElementsByTagName("thead")[0]) {
332 			return Element.getHeight(this.getRootNode().getElementsByTagName("thead")[0]);
333 		}
334 		
335 		return 0;

338 	_workOutMovement: function() {	
339 		if(this._scrollingTimeout) {
340 			clearTimeout(this._scrollingTimeout);
341 		}
342 		
343 		// if we've no scrollableholder, we won't know how many to load
344 		if(!this._scrollableHolder) {
345 			return;
346 		}
347 		
348 		var offset = this._scrollableHolder.getRootNode().scrollTop - this.getHeaderHeight();
349 		
350 		if(offset < 0) {
351 			offset = 0;
352 		}
353 		
354 		// div height or table height if not scrolling
355 		var visibleHeight = Element.getHeight(this._scrollableHolder.getRootNode());
356 		
357 		if(visibleHeight == 0) {
358 			return;
359 		}
360 		
361 		var startRow = Math.floor(offset/this.options.rowHeight);
362 		var endRow = startRow + Math.floor(visibleHeight/this.options.rowHeight);
363 		
364 		if(this._lastRange.start != startRow || this._lastRange.end != endRow || this._initialLoad === true || this._forceLoad) {
365 			this._lastRange.start = startRow;
366 			this._lastRange.end = endRow;
367 			
368 			// load missing details
369 			//Log.info("calling loadDetails for rows " + (startRow) + " to " + (endRow + 1) + " this._forceLoad = " + this._forceLoad);
370 			this._loadDetails(startRow, endRow + 1);
371 		}
