1 include(bbq.gui.GUIWidget);
  2 include(bbq.util.BBQUtil);
  3 
  4 /**
  5  * @class bbq.gui.Resizer
  6  * @extends bbq.gui.GUIWidget 
  7  */
  8 bbq.gui.Resizer = new Class.create(bbq.gui.GUIWidget, {
  9 	
 10 	_disabled: null,
 11 	_screenSize: null,
 12 	_lastMousePosition: null,
 13 	_dimension: null,
 14 	_axis: null,
 15 	_mouseIsDown: null,
 16 	
 17 	/**
 18 	 * @param {mixed} options
 19 	 * @example
 20 	 * Supports the following options:
 21 	 * 
 22 	 * options {
 23 	 * 		div1: node,						The div one side of the dragger
 24 	 * 		div2: node,						The div on the other side of the dragger
 25 	 * 		div1Min: int,					The minimum size of div1
 26 	 * 		div2Min: int,					The minimum size of div2
 27 	 * 		horizontal: bool			Whether the dragging motion is horizontal or vertical
 28 	 * }
 29 	 */
 30 	initialize: function($super, options) {
 31 		$super(options);
 32 		
 33 		this.setRootNode("div");
 34 		this.addClass("Resizer");
 35 		this.addClass(this.options.vertical ? "VerticalResizer" : "HorizontalResizer");
 36 		this._disabled = false;
 37 		
 38 		if(this.options.vertical) {
 39 			this._dimension = "height";
 40 			this._axis = "y";
 41 		} else {
 42 			this._dimension = "width";
 43 			this._axis = "x";
 44 		}
 45 		
 46 		Event.observe(this.getRootNode(), "mousedown", function(event) {
 47 			this._mouseIsDown = true;
 48 			this._startResize(event);
 49 		}.bindAsEventListener(this));
 50 		
 51 		Event.observe(document, "mouseup", function(event) {
 52 			this._mouseIsDown = false;
 53 		}.bindAsEventListener(this));
 54 		
 55 		Event.observe(this.getRootNode(), "mouseover", function(event) {
 56 			if(this._mouseIsDown) {
 57 				this._startResize(event);
 58 			}
 59 		}.bindAsEventListener(this));
 60 	},
 61 	
 62 	/**
 63 	 * @param {boolean} disabled
 64 	 */
 65 	setDisabled: function(disabled) {
 66 		this._disabled = disabled;
 67 		this[(disabled ? "add" : "remove") + "Class"]("ResizerDisabled");
 68 	},
 69 	
 70 	_startResize: function(event) {
 71 		if(!this._disabled) {
 72 			this._lastMousePosition = {
 73 				x: Event.pointerX(event),
 74 				y: Event.pointerY(event)
 75 			};
 76 			
 77 			Event.stop(event);
 78 			document.onmousemove = this._resizeIt.bindAsEventListener(this);
 79 			document.onmouseup = this._stopResize.bindAsEventListener(this);
 80 			document.onselectstart = function() {
 81 				return false;
 82 			};
 83 			
 84 			this._screenSize = BBQUtil.getWindowSize();
 85 			this.addClass("ResizerOver");
 86 		}
 87 	},
 88 			
 89 	_stopResize: function() {
 90 		document.onmousemove = null;
 91 		document.onmouseup = null;
 92 		document.onselectstart = null;
 93 		this._lastMousePosition = {};
 94 		
 95 		this.removeClass( "ResizerOver");
 96 	},
 97 	
 98 	_getPanelDimension: function(panel, dimension) {
 99 		var value;
100 		
101 		if(panel.getRootNode instanceof Function) {
102 			value = panel.getStyle(dimension);
103 		} else {
104 			value = panel.style[dimension];
105 		}
106 		
107 		return parseInt(value.replace("px", ""));
108 	},
109 	
110 	_resizeIt: function(event) {
111 		var currentMousePosition = {
112 			x: Event.pointerX(event),
113 			y: Event.pointerY(event)
114 		};
115 		
116 		var d = currentMousePosition[this._axis] - this._lastMousePosition[this._axis];
117 		
118 		var newDiv1Dim = this._getPanelDimension(this.options.div1, this._dimension) + d;
119 		var newDiv2Dim = this._getPanelDimension(this.options.div2, this._dimension) - d;
120 		
121 		if(this.options.div1.owner) {
122 			this.options.div1.owner().triggerEvent("onWillResize");
123 		}	
124 		
125 		if(this.options.div2.owner) {
126 			this.options.div2.owner().triggerEvent("onWillResize");
127 		}	
128 		
129 		if(newDiv1Dim > this.options.div1Min && newDiv2Dim > this.options.div2Min) {
130 			DOMUtil.setStyle(this.options.div1, this._dimension, newDiv1Dim + "px");
131 		}
132 		
133 		if(this.options.div1.owner) {
134 			this.options.div1.owner().triggerEvent("onResize");
135 			
136 			if(this.options.div1.owner().resize) {
137 				this.options.div1.owner().resize();
138 			}
139 		}	
140 		
141 		if(this.options.div2.owner) {
142 			this.options.div2.owner().triggerEvent("onResize");
143 			
144 			if(this.options.div2.owner().resize) {
145 				this.options.div2.owner().resize();
146 			}
147 		}	
148 		
149 		LayoutManager.resizePanels();
150 		this._lastMousePosition = currentMousePosition;
151 		this.notifyListeners("onResize");
152 		Event.stop(event);
153 	}
154 		
155 });
