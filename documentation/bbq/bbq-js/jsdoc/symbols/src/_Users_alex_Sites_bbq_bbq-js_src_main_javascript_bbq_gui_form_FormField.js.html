1 include(bbq.gui.GUIWidget);
  2 include(bbq.util.Log);
  3 
  4 /**
  5  * Supports the following options:
  6  *
  7  * options: {
  8  *      value:  Object      // an initial value
  9  * }
 10  *
 11  * Dispatches the following notifications:
 12  *
 13  * onError
 14  *
 15  * @class bbq.gui.form.FormField
 16  * @extends bbq.gui.GUIWidget
 17  */
 18 bbq.gui.form.FormField = new Class.create(bbq.gui.GUIWidget, {
 19 	_behaviours: null,
 20 	_preTransformValidators: null,
 21 	_postTransformValidators: null,
 22 	_transformer: null,
 23 
 24 	initialize: function($super, args) {
 25 		try {
 26 			$super(args);
 27 
 28 			this._behaviours = [];
 29 			this._preTransformValidators = [];
 30 			this._postTransformValidators = [];
 31 
 32 			this.setRootNode("input");
 33 			this.addClass("FormField");
 34 
 35 			if(this.options.onChange) {
 36 				this.registerListener("onChange", this.options.onChange);
 37 			}
 38 
 39 			if(this.options.valueTransformer) {
 40 				this.setValueTransformer(this.options.valueTransformer);
 41 			}
 42 
 43 			if(this.options.name) {
 44 				this.setAttribute("name", this.options.name);
 45 			}
 46 
 47 			this.registerListener("onFocus", function() {
 48 				FocusWatcher.setKeypressCallbackObject(this);
 49 			}.bind(this));
 50 
 51 			this.registerListener("onBlur", function() {
 52 				FocusWatcher.setKeypressCallbackObject(null);
 53 			}.bind(this));
 54 		} catch(e) {
 55 			Log.error("Error constructing FormField", e);
 56 		}
 57 	},
 58 
 59 	setRootNode: function($super, rootNode) {
 60 		$super(rootNode);
 61 
 62 		this.getRootNode().onfocus = this.notifyListeners.bind(this, "onFocus");
 63 		this.getRootNode().onblur = this.notifyListeners.bind(this, "onBlur");
 64 		this.getRootNode().onchange = this.notifyListeners.bind(this, "onChange");
 65 
 66 		if(!Object.isUndefined(this.options.value)) {
 67 			this._setRawValue(this.options.value);
 68 		}
 69 	},
 70 
 71 	_getRawValue: function() {
 72 		return this.getRootNode().value;
 73 	},
 74 
 75 	_setRawValue: function(value) {
 76 		if(this._transformer && this._transformer.deTransform) {
 77 			value = this._transformer.deTransform(value);
 78 		}
 79 
 80 		this.getRootNode().value = value;
 81 	},
 82 
 83 	/**
 84 	 * Returns the value contained within this field.
 85 	 */
 86 	getValue: function() {
 87 		var value = this._getRawValue();
 88 
 89 		return this._validateAndTransform(value);
 90 	},
 91 
 92 	getName: function() {
 93 		return this.options.name;
 94 	},
 95 
 96 	getUnvalidatedValue: function() {
 97 		var value = this._getRawValue();
 98 
 99 		return this._transform(value);
100 	},
101 
102 	_validateAndTransform: function(value) {
103 		this.removeClass("FormField_error");
104 
105 		// run pre-transform validators
106 		this._preValidate(value);
107 
108 		// transform the value if necessary
109 		value = this._transform(value);
110 
111 		// run post transform validators
112 		this._postValidate(value);
113 
114 		// return our value
115 		return value;
116 	},
117 
118 	_transform: function(value) {
119 		// if we have a value transformer, transform the value
120 		if (this._transformer) {
121 			value = this._transformer.transform(value);
122 		}
123 
124 		return value;
125 	},
126 
127 	_preValidate: function(value) {
128 		this._preTransformValidators.each(function(validator) {
129 			var result = validator.validate(value);
130 
131 			if (result) {
132 				var error = {error: result, field: this};
133 
134 				this.addClass("FormField_error");
135 				this.notifyListeners("onError", error);
136 
137 				throw error;
138 			}
139 		}.bind(this));
140 	},
141 
142 	_postValidate: function(value) {
143 		this._postTransformValidators.each(function(validator) {
144 			var result = validator.validate(value);
145 
146 			if (result) {
147 				var error = {error: result, field: this};
148 
149 				this.addClass("FormField_error");
150 				this.notifyListeners("onError", error);
151 
152 				throw error;
153 			}
154 		}.bind(this));
155 	},
156 
157 	setValue: function(value) {
158 		this._setRawValue(value);
159 
160 		this._validateAndTransform(value);
161 	},
162 
163 	addValidator: function(validator) {
164 		if(!validator) {
165 			Log.error("Invalid validator!");
166 
167 			return;
168 		}
169 
170 		if(!validator.validate) {
171 			Log.error("Validator should implement a validate method");
172 
173 			return;
174 		}
175 
176 		if (validator.isPostTransformValidator && validator.isPostTransformValidator()) {
177 			this._postTransformValidators.push(validator);
178 		} else {
179 			this._preTransformValidators.push(validator);
180 		}
181 	},
182 
183 	addBehaviour: function(behaviour) {
184 		if (!behaviour) {
185 			Log.error("Invalid behaviour!");
186 		}
187 
188 		if (!behaviour.setField) {
189 			Log.error("Behaviour should implement a setField method");
190 
191 			return;
192 		}
193 
194 		behaviour.setField(this);
195 
196 		this._behaviours.push(behaviour);
197 	},
198 
199 	setValueTransformer: function(transformer) {
200 		if (!transformer) {
201 			Log.error("Invalid transformer!");
202 		}
203 
204 		if (!transformer.transform) {
205 			Log.error("Transformer should implement a transform method");
206 
207 			return;
208 		}
209 		
210 		this._transformer = transformer;
211 	},
212 
213 	loseFocus: function() {
214 		this.removeClass("FormField_focused");
215 	},
216 
217 	acceptFocus: function() {
218 		this.addClass("FormField_focused");
219 	}
220 });
