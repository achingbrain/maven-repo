  6 /**
  7  * Uses the SWFObject library to wrapper an embedded flash movie.
  8  *
  9  * @class bbq.web.SwfEmbed
 10  * @extends bbq.gui.GUIWidget 
 11  */
 12 bbq.web.SwfEmbed = new Class.create(bbq.gui.GUIWidget, {
 13 	// unique identifier used by the movie to call methods on us
 14 	_id: null,
 15 	_hasLoaded: false,
 16 
 17 	/**
 18 	 * @param {Object} options
 19 	 * @example
 20 	 * Supports the following options:
 21 	 * 
 22 	 * options {
 23 	 * 		swf: string							// Path to the flash movie
 24 	 * 		attributes: Object,				// Attributes to set on the Object node
 25 	 * 		variables: {						// Variables to pass to the movie in key: value pairs
 26 	 * 			key: value,
 27 	 * 			...
 28 	 * 		},
 29 	 * 		nocache: boolean				// Whether or not to append a cache buster
 30 	 *		wmode: string					// Specify the wmode to use
 31 	 * }
 32 	 * 
 33 	 * Supports the following events:
 34 	 * 
 35 	 * onLoad
 36 	 */
 37 	initialize: function($super, options) {
 38 		$super(options);
 39 		
 40 		this._id = BBQUtil.generateGUID().replace(/-/g, "");
 41 
 42 		bbq.web.SwfEmbed.instances[this._id] = this;
 43 		
 44 		if(this.options.nocache) {
 45 			this.options.swf += "?nocache=" + BBQUtil.generateGUID();
 46 		}
 47 
 48 		var params = {};
 49 		params["allowScriptAccess"] = "always";
 50 		params["swliveconnect"] = "true";
 51 		params["allowFullScreen"] = this.options.allowFullScreen ? "true" : "false";
 52 
 53 		if(this.options.wmode) {
 54 			params["wmode"] = this.options.wmode;
 55 		}
 56 
 57 		var flashVars = {};
 58 
 59 		if(this.options.variables instanceof Object) {
 60 			//Log.info("Setting flashVars on SWFObject");
 61 			for (var key in this.options.variables) {
 62 				flashVars[key] = this.options.variables[key];
 63 			}
 64 		}
 65 
 66 		flashVars["pageObject"] = this._id;
 67 
 68 		// swfobject demands that the element to be replaced is in the DOM :(
 69 		var node = DOMUtil.createElement("div");
 70 		node.id = this._id;
 71 		document.body.appendChild(node);
 72 
 73 		swfobject.embedSWF(this.options.swf, this._id, "100%", "100%", "10", null, flashVars, params, this.options.attributes, function(e) {
 74 			if(e.success) {
 75 				this.setRootNode(e.ref);
 76 				this.getRootNode().id = this._id;
 77 			} else {
 78 				Log.error("Failed to embed SWF " + this.options.swf);
 79 				Log.dir(e);
 80 			}
 81 		}.bind(this));
 82 	},
 83 
 84 	loaded: function() {
 85 		return this._hasLoaded;
 86 	},
 87 	
 88 	/**
 89 	 * @param {Node} node
 90 	 */
 91 	render: function() {
 92 		if(!Browser.Flash) {
 93 			var errorNotification = new bbq.gui.error.NoFlash();
 94 			errorNotification.appear();
 95 			
 96 			return;
 97 		}
 98 	},
 99 	
100 	/**
101 	 * @private
102 	 */
103 	_loaded: function() {
104 		//Log.info("Getting flash object from DOM with ID " + this.options.swf.id);
105 		
106 		if(!this._hasLoaded) {
107 			if($(this._id)) {
108 				this._hasLoaded = true;
109 				//Log.info("notifying onLoad listeners");
110 				this.notifyListeners("onLoad");
111 			} else {
112 				//Log.info("DOM node not loaded yet");
113 				setTimeout(this._loaded.bind(this), 500);
114 			}
115 		}
116 	},
117 	
118 	/**
119 	 * @param {string} methodName
120 	 * @param {Object} args
121 	 */
122 	callFlashFunction: function(methodName, args) {
123 		//Log.info("Calling flash function " + methodName + " with args " + Object.toJSON(args));
124 
125 		var rootNode = this.getRootNode();
126 
127 		if(rootNode[methodName]) {
128 			//Log.info("Found method " + methodName + " on Flash object - " + typeof(this._flashObject[methodName]));
129 			if(args) {
130 				return rootNode[methodName](args);
131 			}
132 			
133 			return rootNode[methodName]();
134 		}
135 		
136 		Log.warn("Could not find method " + methodName + " on Flash object");
137 	}
138 });
