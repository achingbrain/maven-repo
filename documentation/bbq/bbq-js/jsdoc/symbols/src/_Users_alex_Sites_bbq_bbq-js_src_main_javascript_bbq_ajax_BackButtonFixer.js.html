  3 /**
  4  * @class bbq.ajax.BackButtonFixer
  5  */
  6 bbq.ajax.BackButtonFixer = Class.create({
  7 	
  8 	_lastMethod: null,
  9 	_neediFrameFix: null,
 10 	_controller: null,
 11 	_iFrameURL: "/js/iFrameFix.html",
 12 	_emptyMethod: null,
 13 	_overrideMethodCheck: null,
 14 	
 15 	/*
 16 	 * Constructor
 17 	 * @param {Object} controller
 18 	 * @param {boolean} emptyMethod
 19 	 * 
 20 	 */
 21 	initialize: function(controller, emptyMethod) {
 22 		// the object to call methods on
 23 		this._controller = controller;
 24 		
 25 		// the method to call to return the page to it's initial state
 26 		if(emptyMethod) {
 27 			this._emptyMethod = emptyMethod;
 28 		}
 29 		
 30 		// have we been loaded with a method request
 31 		this._lastMethod = this._getMethod();
 32 		
 33 		// stupid IE
 34 		this._neediFrameFix = Browser.InternetExplorer && Browser.version < 7 ? true : false;
 35 		
 36 		if(this._neediFrameFix) {
 37 			this.createiFrame(this._lastMethod);
 38 		}
 39 		
 40 		// call the requested (bookmarked) method
 41 		this._overrideMethodCheck = true;
 42 		this._callMethod(this._lastMethod);
 43 		
 44 		// start checking to see if the URL has changed
 45 		setInterval(this._checkLocation.bind(this), 100);
 46 	},
 47 	
 48 	/*
 49 	 * Creates an iFrame to add to the page
 50 	 * @param {String} method
 51 	 */
 52 	createiFrame: function(method) {
 53 		var iframe = document.createElement("iframe");
 54 		iframe.id = "iFrameFix";
 55 		iframe.name = "iFrameFix";
 56 		iframe.style.display = "none";
 57 		document.getElementsByTagName("body")[0].appendChild(iframe);
 58 		this.loadiFrameURL(method);
 59 	},
 60 	
 61 	/** looks at the URL in the address bar */
 62 	_checkLocation: function() {
 63 		var method = this._getMethod();
 64 		
 65 		if(this._lastMethod != method) { // new method call
 66 			//alert("method = " + method + " lastMethod = " + this.lastMethod);
 67 			
 68 			if(this._neediFrameFix) {
 69 				this.loadiFrameURL(method);
 70 			} else {
 71 				this._callMethod(method);
 72 			}
 73 		}
 74 	},
 75 	
 76 	/*
 77 	 * gets the method name from the passed argument
 78 	 * @param {String} href
 79 	 * @param {boolean} acceptNullValue 
 80 	 */
 81 	_getMethod: function() {
 82 		var method = document.location.href.split("#");
 83 		
 84 		if(method.length > 1) {
 85 			return method[1];
 86 		}
 87 	
 88 		return null;
 89 	},
 90 	
 91 	/*
 92 	 * loads the iFrame for crap browsers or calls the callMethod method
 93 	 * @param {String} method
 94 	 */
 95 	loadiFrameURL: function(method) {
 96 		$("iFrameFix").src = this._iFrameURL + "?time="+(new Date()).getTime() + (method != null ? "#" + method : "");
 97 	},
 98 	
 99 	/*
100 	 * tells the iFrame to examine it's URL and tell us what the method name is
101 	 * @param {Object} frameWindow
102 	 */
103 	iFrameLoaded: function(frameWindow) {
104 		//Log.info("iframe callback");
105 		frameWindow.owner = this;
106 		frameWindow.readURLandCallMethod();
107 	},
108 	
109 	/* calls the passed method name - override this method
110 	 * @param {boolean} passed
111 	 */
112 	_callMethod: function(passed) {
113 		//Log.info("calling " + passed + ", last method was " + this.lastMethod);
114 		
115 		if(passed) {
116 			var query = passed.split("/");
117 			var method = query[0];
118 			var args = [];
119 			
120 			// don't call the same method twice
121 			if(!this._overrideMethodCheck && method == this._lastMethod) {
122 				this._overrideMethodCheck = false;
123 				return;
124 			}
125 			
126 			this._lastMethod = passed;
127 			
128 			// build arguments
129 			for(var i = 1, iCount=query.length; i < iCount; i++) {
130 				args[(i-1)] = query[i];
131 			}
132 			
133 			// make sure method exists
134 			if(this._controller[method]) {
135 				var methodToCall = this._controller[method].bind(this._controller);
136 				methodToCall.apply(this, args);
137 			} else { // tried to call non-existant method
138 				//Log.info("Method  " + method + " not found on controller");
139 			}
140 		} else if(this._emptyMethod) {
141 			//Log.info("Called empty method");
142 			this._emptyMethod.call();
143 			this._lastMethod = null;
144 		}
145 	}
146 });
