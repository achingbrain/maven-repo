1 include(bbq.gui.GUIWidget);
  2 include(bbq.util.BBQUtil);
  3 
  4 /**
  5  * GUIButton
  6  * 
  7  * This class creates a link which can be used to receive on click events.
  8  * 
  9  * @class bbq.gui.button.GUIButton
 10  * @extends bbq.gui.GUIWidget
 11  * 
 12  */
 13 bbq.gui.button.GUIButton = Class.create(bbq.gui.GUIWidget, {
 14 	_disabled: false,
 15 	_down: false,
 16 	_mouseDownOverButton: false,
 17 
 18 	/**
 19 	 * Constructor!
 20 	 * 
 21 	 * Supports the following options: 
 22 	 * options: {
 23 	 *		text: String,
 24 	 *		toolTip: String,
 25 	 *		anchor: String,
 26 	 * 		startDisabled: boolean,
 27 	 * 		startDown: boolean,
 28 	 * 		rememberDownState: boolean,
 29 	 *		onClick: function,
 30 	 *		onMouseOut: function,
 31 	 *		onMouseOver: function,
 32 	 *		onMouseDown: function,
 33 	 *		onMouseUp: function,
 34 	 *		onButtonDown: function
 35 	 * }
 36 	 *
 37 	 * Supports observers for the following events
 38 	 *
 39 	 * onClick
 40 	 * onMouseOut
 41 	 * onMouseOver
 42 	 * onMouseDown
 43 	 * onMouseUp
 44 	 * onButtonDown
 45 	 */
 46 	initialize: function($super, options) {
 47 		$super(options);
 48 
 49 		this._setUpRootNode();
 50 
 51 		this.addClass("GUIButton");
 52 		this.setDisabled(this.options.startDisabled);
 53 		this.setDown(this.options.startDown);
 54 		this.getRootNode().onclick = this.buttonClicked.bindAsEventListener(this);
 55 		this.getRootNode().onmouseout = this.mouseOut.bindAsEventListener(this);
 56 		this.getRootNode().onmouseover = this.mouseOver.bindAsEventListener(this);
 57 		this.getRootNode().onmousedown = this.mouseDown.bindAsEventListener(this);
 58 		this.getRootNode().onmouseup = this.mouseUp.bindAsEventListener(this);
 59 
 60 		if(this.options.onClick) {
 61 			this.registerListener("onClick", this.options.onClick);
 62 		}
 63 
 64 		if (this.options.onMouseOut) {
 65 			this.registerListener("onMouseOut", this.options.onMouseOut);
 66 		}
 67 
 68 		if (this.options.onMouseOver) {
 69 			this.registerListener("onMouseOver", this.options.onMouseOver);
 70 		}
 71 
 72 		if (this.options.onMouseDown) {
 73 			this.registerListener("onMouseDown", this.options.onMouseDown);
 74 		}
 75 
 76 		if (this.options.onMouseUp) {
 77 			this.registerListener("onMouseUp", this.options.onMouseUp);
 78 		}
 79 
 80 		if (this.options.onButtonDown) {
 81 			this.registerListener("onButtonDown", this.options.onButtonDown);
 82 		}
 83 	},
 84 
 85 	_setUpRootNode: function() {
 86 		this.setRootNode("a");
 87 
 88 		if (this.options.text) {
 89 			this.appendChild(this.options.text);
 90 		}
 91 
 92 		if (this.options.toolTip) {
 93 			this.setAttribute("title", this.options.toolTip);
 94 		}
 95 
 96 		if (this.options.anchor) {
 97 			this.setAttribute("href", this.options.anchor);
 98 		} else {
 99 			this.setAttribute("href", ".");
100 		}
101 	},
102 
103 	/**
104 	 * Button clicked handler function
105 	 * @param {Event} event
106 	 */
107 	buttonClicked: function(event) {
108 		if(event) {
109 			if(this.options.onClick) {
110 				Event.stop(event);
111 			}
112 
113 			BBQUtil.clearFocus(event);
114 		}
115 
116 		if(this._disabled || this._down) {
117 			return false;
118 		}
119 
120 		if(this.options.rememberDownState) {
121 			this.setDown(true);
122 		}
123 
124 		this.notifyListeners("onClick");
125 
126 		// should we let the link take us to another page?
127 		if(this.options.anchor) {
128 			return true;
129 		}
130 
131 		return false;
132 	},
133 
134 	/**
135 	 * Mouse out handler function
136 	 * @param {Event} event
137 	 */
138 	mouseOut: function(event) {
139 		if(this._disabled) {
140 			return false;
141 		}
142 
143 		this.removeClass("buttonOver");
144 		this.notifyListeners("onMouseOut");
145 
146 		// mouse button was clicked, but user dragged out of button before releasing
147 		if(this._mouseDownOverButton) {
148 			this.removeClass("buttonDown");
149 		}
150 	},
151 
152 	/**
153 	 * Mouse over handler function
154 	 * @param {Event} event
155 	 */
156 	mouseOver: function(event) {
157 		if(this._disabled) {
158 			return false;
159 		}
160 
161 		this.addClass("buttonOver");
162 		this.notifyListeners("onMouseOver");
163 	},
164 
165 	/**
166 	 * Mouse down handler function
167 	 * @param {Event} event
168 	 */
169 	mouseDown: function(event) {
170 		if(this._disabled || this._down) {
171 			return false;
172 		}
173 
174 		if(!this.options.rememberDownState) {
175 			this.addClass("buttonDown");
176 		}
177 
178 		this._mouseDownOverButton = true;
179 
180 		this.notifyListeners("onMouseDown");
181 	},
182 
183 	/**
184 	 * Mouse up handler function
185 	 * @param {Event} event
186 	 */
187 	mouseUp: function(event) {
188 		if(this._disabled || this._down) {
189 			return false;
190 		}
191 
192 		this._mouseDownOverButton = false;
193 
194 		this.removeClass("buttonDown");
195 		this.notifyListeners("onMouseUp");
196 	},
197 
198 	/**
199 	 * Down state setter
200 	 * @param {boolean} down
201 	 */
202 	setDown: function(down) {
203 		if(this._down == down) {
204 			return;
205 		}
206 
207 		this._down = down;
208 		this[(this._down ? "add" : "remove") + "Class"]("buttonDown");
209 		this.notifyListener("onButton" + (this._down ? "Down" : "Up"));
210 	},
211 
212 	/**
213 	 * Clears button down state
214 	 */
215 	clearDown: function() {
216 		this._down = false;
217 		this.removeClass("buttonDown");
218 	},
219 
220 	/**
221 	 * Checks if button is down.
222 	 * @return {boolean}
223 	 */
224 	isDown: function() {
225 		return this._down;
226 	},
227 
228 	/**
229 	 * Button disabled status setter
230 	 * @param {boolean} disabled
231 	 */
232 	setDisabled: function(disabled) {
233 		this._disabled = disabled;
234 		this[(this._disabled ? "add" : "remove") + "Class"]("buttonDisabled");
235 		this[(this._disabled ? "remove" : "add") + "Class"]("buttonEnabled");
236 	},
237 
238 	isDisabled: function() {
239 		return this._disabled;
240 	},
241 
242 	setText: function(text) {
243 		this.empty();
244 		this.anchor.appendChild(document.createTextNode(text));
245 	}
246 });
