1 include(bbq.gui.updateables.UpdateableFieldWithInstruction);
  2 
  3 /**
  4  * @class bbq.gui.updateables.UpdateableTextField A single line HTML text input field.
  5  * @extends bbq.gui.updateables.UpdateableFieldWithInstruction 
  6  */
  7 bbq.gui.updateables.UpdateableTextField = new Class.create(bbq.gui.updateables.UpdateableFieldWithInstruction, {
  8 	
  9 	inputField: null,
 10 	
 11 	/**
 12 	 * @param {Object} options 
 13 	 * @example
 14 	 * Supports the following options:
 15 	 * 
 16 	 * options: {
 17 	 * 		searchBox: boolean				// if true will render a mac like search box on Safari
 18 	 * }
 19 	 */
 20 	initialize: function($super, options) {
 21 		$super(options);
 22 		
 23 		this.addClass("UpdateableTextField");
 24 	},
 25 	
 26 	/**
 27 	 * Creates an editable representation of this field
 28 	 */
 29 	createEditField: function() {
 30 		this.inputField = this._createEditField();
 31 		
 32 		if(this.options.inlineInstruction) {
 33 			if(Browser.forms.placeholderText) {
 34 				this.inputField.placeholder = this.options.inlineInstruction;
 35 			} else {
 36 				if (this.inputField.value == this.options.inlineInstruction || this.inputField.value == "") {
 37 					this._showInstruction();
 38 					Event.observe(this.inputField, "blur", this.blurInstructionField.bindAsEventListener(this));
 39 					Event.observe(this.inputField, "focus", this.focusInstructionField.bindAsEventListener(this));
 40 				}
 41 			}
 42 		}
 43 		
 44 		Event.observe(this.inputField, "blur", this._processEventCallback.bindAsEventListener(this, "onblur"));
 45 		Event.observe(this.inputField, "focus", this._processEventCallback.bindAsEventListener(this, "onfocus"));
 46 		Event.observe(this.inputField, "change", this._processEventCallback.bindAsEventListener(this, "onchange"));
 47 		Event.observe(this.inputField, "keydown", this._processEventCallback.bindAsEventListener(this, "onkeydown"));
 48 		Event.observe(this.inputField, "keyup", this._processEventCallback.bindAsEventListener(this, "onkeyup"));
 49 		Event.observe(this.inputField, "keypress", this._processEventCallback.bindAsEventListener(this, "onkeypress"));
 50 		Event.observe(this.inputField, "click", this._processEventCallback.bindAsEventListener(this, "onclick"));
 51 		
 52 		Event.observe(this.inputField, "blur", function() {
 53 			this.inputField.hasFocus = false;
 54 		}.bind(this));
 55 		Event.observe(this.inputField, "focus", function() {
 56 			this.inputField.hasFocus = true;
 57 		}.bind(this));
 58 		
 59 		return this.inputField;
 60 	},
 61 	
 62 	_processEventCallback: function(event, type) {
 63 		setTimeout(function() {
 64 			if(this.options[type] && this.options[type] instanceof Function) {
 65 				this.options[type]();
 66 			}
 67 			
 68 			this.notifyListeners(type, event);
 69 		}.bind(this), 100);
 70 	},
 71 	
 72 	/**
 73 	 * Sets a timeout to call updateLocalValue - necessary as Internet Explorer is stupid
 74 	 * 
 75 	 */
 76 	preUpdateLocalValue: function(event) {
 77 		setTimeout(this.updateLocalValue.bind(this), 10);
 78 	},
 79 	
 80 	/**
 81 	 * @private
 82 	 */
 83 	_createEditField: function() {
 84 		return DOMUtil.createElement("input", {
 85 			type: (this.options.searchBox && Browser.forms.types.search ? "search" : "text"),
 86 			value: this._getDisplayValue(),
 87 			onchange: this.updateLocalValue.bindAsEventListener(this),
 88 			onkeydown: this.preUpdateLocalValue.bindAsEventListener(this),
 89 			onfocus: this._clearFocusWatcher.bind(this),
 90 			disabled: this.disabled,
 91 			autocomplete: "off",
 92 			oncomplete: "off"
 93 		});		
 94 	}
 95 });
