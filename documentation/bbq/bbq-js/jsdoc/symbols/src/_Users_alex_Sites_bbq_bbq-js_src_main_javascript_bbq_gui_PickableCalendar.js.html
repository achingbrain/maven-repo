1 include(bbq.gui.GUIWidget);
  2 include(bbq.util.BBQUtil);
  3 
  4 /**
  5  * @class bbq.gui.PickableCalendar
  6  * @extends bbq.gui.GUIWidget
  7  */
  8 bbq.gui.PickableCalendar = new Class.create(bbq.gui.GUIWidget, {
  9 	
 10 	monthDays: new Array(31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31),
 11 	_today: null,
 12 	_date: null,
 13 	_initialDate: null,
 14 	
 15 	/**
 16 	 * @param {mixed} options
 17 	 * @example
 18 	 * Supports the following options:
 19 	 * 
 20 	 * options: {
 21 	 * 		startDate:	Date		// the initially selected date (optional)
 22 	 * 		owner:		Object		// should have a method datePicked(Date:date)
 23 	 * 		title:				String		// the calendar title
 24 	 * 		createCell:	function(day, cell, callbackFunction rowNumber, columnNumber)
 25 	 * 		prevCallback: function(date)
 26 	 * 		nextCallback: function(date)
 27 	 * }
 28 	 */
 29 	initialize: function($super, options) {
 30 		$super(options);
 31 		
 32 		if(this.options.startDate instanceof Date) {
 33 			this._date = this.options.startDate;
 34 		} else {
 35 			this._date = new Date(this.options.startDate);
 36 		}
 37 		
 38 		this.setRootNode("div");
 39 		this.addClass("pickableCalendar");
 40 		
 41 		this._today = new Date();
 42 		this._initialDate = new Date(this._date.getTime());
 43 		this.render();
 44 	},
 45 	
 46 	render: function() {
 47 		// check for leap years
 48 		if(this._date.getMonth() == 1 && (this._date.getFullYear() % 4 == 0 || (this._date.getFullYear() % 100 == 0 && this._date.getFullYear() % 400 == 0))) {
 49 			this.monthDays[1] = 29;
 50 		} else {
 51 			this.monthDays[1] = 28;
 52 		}
 53 		
 54 		DOMUtil.emptyNode(this.getRootNode());
 55 		
 56 		var dateTable = document.createElement("table");
 57 		var dateTableBody = document.createElement("tbody");
 58 		dateTable.appendChild(dateTableBody);
 59 		
 60 		// title
 61 		if(typeof(this.options.title) != "undefined") {
 62 			var titleRow = DOMUtil.createTableHeaderRow(1);
 63 			titleRow.childNodes[0].appendChild(document.createTextNode(this.options.title));
 64 			titleRow.childNodes[0].colSpan = 7;
 65 			DOMUtil.addClass(titleRow.childNodes[0], "titlecell");
 66 			dateTableBody.appendChild(titleRow);
 67 		}
 68 		
 69 		var topRow = DOMUtil.createTableHeaderRow(3);
 70 		
 71 		// back
 72 		var backMonthLink = DOMUtil.createTextElement("a", "<", {href: ".", onclick: this.previousMonth.bindAsEventListener(this), title: Language.get("bbq.gui.PickableCalendar.previousmonth")});
 73 		DOMUtil.addClass(topRow.childNodes[0], "prevnext");
 74 		topRow.childNodes[0].appendChild(backMonthLink);
 75 		
 76 		// month
 77 		topRow.childNodes[1].appendChild(document.createTextNode(Language.get("bbq.gui.PickableCalendar.months" + this._date.getMonth()) + " " + this._date.getFullYear()));
 78 		topRow.childNodes[1].colSpan = 5;
 79 		
 80 		// forwards
 81 		var forwardMonthLink = DOMUtil.createTextElement("a", ">", {href: ".", onclick: this.nextMonth.bindAsEventListener(this), title: Language.get("bbq.gui.PickableCalendar.nextmonth")});
 82 		DOMUtil.addClass(topRow.childNodes[2], "prevnext");
 83 		topRow.childNodes[2].appendChild(forwardMonthLink);
 84 		
 85 		dateTableBody.appendChild(topRow);
 86 		
 87 		// day headers
 88 		var dayRow = DOMUtil.createTableHeaderRow(7);
 89 		
 90 		for(var i = 0; i < 7; i++) {
 91 			dayRow.childNodes[i].appendChild(document.createTextNode(Language.get("bbq.gui.PickableCalendar.days" + i).substr(0, 1)));
 92 		}
 93 		
 94 		dateTableBody.appendChild(dayRow);
 95 		
 96 		// pad out the first row
 97 		var tempDate = new Date(this._date.getTime());
 98 		tempDate.setDate(1);
 99 		
100 		var padding = tempDate.getDay();
101 		var row = document.createElement("tr");
102 		
103 		var n = 0;
104 		
105 		for(; n < padding; n++) {
106 			var cell = document.createElement("td");
107 			cell.appendChild(document.createTextNode(" "));
108 			row.appendChild(cell);
109 		}
110 		
111 		// loop through the days of the month
112 		var j = 1;
113 		var datePickedCallback = this._datePicked.bindAsEventListener(this);
114 		
115 		for(var k = 1, kCount=this.monthDays[this._date.getMonth()]; k <= kCount; k++) {
116 			if(n == 7) {
117 				dateTableBody.appendChild(row);
118 				row = document.createElement("tr");
119 				
120 				if(j % 2 == 0) {
121 					DOMUtil.addClass(row, "even");
122 				}
123 				
124 				n = 1;
125 				j++;
126 			} else {
127 				n++;
128 			}
129 			
130 			var cell = document.createElement("td");
131 			
132 			if(this.options.createCell) {
133 				cell.appendChild(this.options.createCell(k, cell, DOMUtil.createElement("a",  {href: ".", onclick: datePickedCallback}), j, n));
134 			} else {
135 				cell.appendChild(DOMUtil.createTextElement("a", k, {href: ".", onclick: datePickedCallback}));
136 			}
137 			
138 			tempDate.setDate(k);
139 			
140 			DOMUtil.addClass(cell, "day");
141 			
142 			if(this.options.showStartDate && this.equals(tempDate, this._initialDate)) {
143 				DOMUtil.addClass(cell, "originalDate");
144 			} else if(this.options.showToday && this.equals(tempDate, this._today)) {
145 				DOMUtil.addClass(cell, "today");
146 			}
147 			
148 			row.appendChild(cell);
149 		}
150 		
151 		// pad out the last row if necessary
152 		if(n != 0) {
153 			for(; n < 7; n++) {
154 				var cell = document.createElement("td");
155 				cell.appendChild(document.createTextNode(" "));
156 				row.appendChild(cell);
157 			}
158 			
159 			dateTableBody.appendChild(row);
160 		}
161 		
162 		this.getRootNode().appendChild(dateTable);
163 	},
164 	
165 	equals: function(date1, date2) {
166 		return (date1.getFullYear() == date2.getFullYear() && date1.getMonth() == date2.getMonth() && date1.getDate() == date2.getDate());
167 	},
168 	
169 	previousMonth: function(event) {
170 		BBQUtil.clearFocus(event);
171 		
172 		try {
173 			this._date.setMonth(this._date.getMonth() - 1);
174 			
175 			if(this.options.prevCallback) {
176 				this.options.prevCallback(new Date(this._date));
177 			}
178 			
179 			this.render();	
180 			
181 			this.notifyListeners("onMonthChanged");
182 		} catch(e) {
183 			Log.error("Error going to previous month", e);
184 		}
185 		return false;
186 	},
187 	
188 	nextMonth: function(event) {
189 		BBQUtil.clearFocus(event);
190 		
191 		try {
192 			this._date.setMonth(this._date.getMonth() + 1);
193 			
194 			if(this.options.nextCallback) {
195 				this.options.nextCallback(new Date(this._date));
196 			}
197 			
198 			this.render();
199 			
200 			this.notifyListeners("onMonthChanged");
201 		} catch(e) {
202 			Log.error("Error going to next month", e);
203 		}
204 		
205 		return false;
206 	},
207 	
208 	_datePicked: function(event) {
209 		this.notifyListeners("onDatePicked");
210 		Log.info('_datePicked in PickableCalendar');
211 		//Log.dir(event);
212 		var element = BBQUtil.clearFocus(event);
213 		
214 		this._date.setDate(element.innerHTML);
215 		
216 		if(this.options.owner && this.options.owner.datePicked) {
217 			this.options.owner.datePicked(this._date);
218 		}
219 		
220 		return false;
221 	}
222 });
