1 include(bbq.gui.ModalWindow);
  2 include(bbq.gui.button.ButtonHolder);
  3 
  4 /**
  5  * @class bbq.gui.error.FatalError
  6  * @extends bbq.gui.ModalWindow
  7  */
  8 bbq.gui.error.FatalError = new Class.create(bbq.gui.ModalWindow, {
  9 	/**
 10 	 * @param {mixed} options
 11 	 * @example	
 12 	 * Supports the following options:
 13 	 * 
 14 	 * options: {
 15 	 * 		showChrome: boolean						// Whether to be draggable, show close window button, etc
 16 	 * }
 17 	 * 
 18 	 */
 19 	initialize: function($super, options) {
 20 		if(!options) {
 21 			options = {};
 22 		}
 23 		
 24 		options.modal = true;
 25 		
 26 		if(!options.width) {
 27 			options.width = 400;
 28 		}
 29 		
 30 		if(!options.height) {
 31 			options.height = 500;
 32 		}
 33 		
 34 		options.hideChrome = true;
 35 		
 36 		$super(options);
 37 		
 38 		this.addClass("FatalError");
 39 		
 40 		var content = DOMUtil.createTextElement("div", [ 
 41 			DOMUtil.createTextElement("h2", this.getHeader()),
 42 			this.getMessage()
 43 		], {className: "FatalErrorContent"});
 44 		
 45 		if(!this.options.noReload) {
 46 			content.appendChild(DOMUtil.createTextElement("p", Language.get("bbq.gui.error.FatalError.belowmessage")));
 47 			
 48 			var reload = new bbq.gui.button.ButtonHolder();
 49 			reload.addButton(new bbq.gui.button.NativeButton({buttonText: Language.get("bbq.gui.error.FatalError.button"), onclick: function() {
 50 				window.location.reload();
 51 			}.bind(this)}));
 52 			
 53 			reload.appendTo(content);
 54 		}
 55 		
 56 		this.setContent(content);
 57 		
 58 		// stop edit watchers from throwing up a dialogue if the user refreshes the page.
 59 		if(typeof(EditWatcher) != "undefined") {
 60 			EditWatcher.deRegisterAllEditors();
 61 		}
 62 	},
 63 	
 64 	getHeader: function() {
 65 		return Language.get("bbq.gui.error.FatalError.header");
 66 	},
 67 	
 68 	getMessage: function() {
 69 		return Language.get("bbq.gui.error.FatalError.abovemessage");
 70 	}
 71 });
