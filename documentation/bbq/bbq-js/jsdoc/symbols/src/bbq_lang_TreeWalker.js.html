  2 /**
  3  * @class bbq.lang.TreeWalker
  4  */
  5 bbq.lang.TreeWalker = new Class.create({
  6 
  7 	/**
  8 	 * @param {Object} options
  9 	 *
 10 	 * Supports the following options:
 11 	 *
 12 	 * options: {
 13 	 *		 target: Object			The object to walk
 14 	 *		 callback: Function		A function taking the graph member as an argument and returning the new value
 15 	 * }
 16 	 */
 17 	initialize: function(options) {
 18 		this._options = options;
 19 
 20 		if(this._options.target) {
 21 			this._walk(this._options.target);
 22 		}
 23 	},
 24 
 25 	_walk: function(args) {
 26 		if (!args ||
 27 				Object.isFunction(args) || 
 28 				Object.isString(args) ||
 29 				Object.isNumber(args) ||
 30 				args === true ||
 31 				args === false
 32 		) {
 33 			return;
 34 		}
 35 
 36 		if (Object.isArray(args)) {
 37 			for (var i = 0; i < args.length; i++) {
 38 				args[i] = this._options.callback(args[i]);
 39 
 40 				this._walk(args[i]);
 41 			}
 42 		} else if (Object.isHash(args)) {
 43 			args.keys().each(function(key) {
 44 				args.set(key, this._options.callback(args.get(key)));
 45 
 46 				this._walk(args.get(key));
 47 			}.bind(this));
 48 		} else {
 49 			for (var key in args) {
 50 				args[key] = this._options.callback(args[key]);
 51 
 52 				this._walk(args[key]);
 53 			}
 54 		}
 55 	}
 56 });
