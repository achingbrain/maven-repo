  5 /**
  6  * @class bbq.gui.updateables.UpdateableDateField
  7  * @extends bbq.gui.updateables.UpdateableField  
  8  */
  9 bbq.gui.updateables.UpdateableDateField = new Class.create(bbq.gui.updateables.UpdateableField, {
 10 	_datePicker: null,
 11 	_dayDropDown: null,
 12 	_hourDropDown: null,
 13 	_minuteDropDown: null,
 14 	_monthDropDown: null,
 15 	_yearDropDown: null,
 16 	
 17 	/**
 18 	 * @param {Object} options
 19 	 * @example
 20 	 * Supports the following options
 21 	 * 
 22 	 * options: {
 23 	 * 		pickerTitle:		String
 24 	 * 		initialValue:	Date, int or undefined
 25 	 * 		mustBeGreaterThan:	bbq.gui.updateables.UpdateableDateField
 26 	 * 		mustBeLessThan:	bbq.gui.updateables.UpdateableDateField
 27 	 * 		shortDate: boolean		if true, hours and minutes will not be shown
 28 	 * }
 29 	 * 
 30 	 */
 31 	initialize: function($super, options) {
 32 		$super(options);
 33 		
 34 		this.addClass("UpdateableDateField");
 35 	},
 36 	
 37 	_setInitialValues: function() {
 38 		if(typeof(this.options.propertyDisplay) != "undefined") {
 39 			this._setCurrentValue(new Date(this.options.propertyDisplay.entity.getProperty(this.options.propertyDisplay.key)));
 40 			this._setOriginalValue(new Date(this.options.propertyDisplay.entity.getProperty(this.options.propertyDisplay.key)));
 41 		} else {
 42 			if(typeof(this.options.initialValue) == "undefined" || this.options.initialValue == null) {
 43 				this.options.initialValue = new Date();
 44 				
 45 				// set to midday
 46 				this.options.initialValue.setHours(12, 0, 0, 0);
 47 			} else if(this.options.initialValue instanceof Date) {
 48 				// create copy of passed date object
 49 				this.options.initialValue = new Date(this.options.initialValue);
 50 			} else {
 51 				// timestamp passed, hopefully
 52 				this.options.initialValue = new Date(parseInt(this.options.initialValue));
 53 			}
 54 			
 55 			this._setCurrentValue(new Date(this.options.initialValue));
 56 			this._setOriginalValue(new Date(this.options.initialValue));
 57 		}
 58 		
 59 		this._setDisplayValue(DateFormatter.getFormattedDate({date: this._getCurrentValue(), withStrings: true, shortDate: this.options.shortDate}));
 60 	},
 61 	
 62 	_updateInitialValues: function() {
 63 		this._setCurrentValue(new Date(this.options.propertyDisplay.entity.getProperty(this.options.propertyDisplay.key)));
 64 		this._setOriginalValue(new Date(this.options.propertyDisplay.entity.getProperty(this.options.propertyDisplay.key)));
 65 		this._setDisplayValue(DateFormatter.getFormattedDate({date: this._getCurrentValue(), withStrings: true, shortDate: this.options.shortDate}));
 66 	},
 67 	
 68 	_createViewFieldFromPropertyDisplay: function() {
 69 		return this.options.propertyDisplay.entity.getPropertyDisplay({property: this.options.propertyDisplay.key, formatter: function(date) {
 70 			return DateFormatter.getFormattedDate({date: date, withStrings: true, shortDate: this.options.shortDate});
 71 		}.bind(this)});
 72 	},
 73 	
 74 	cancelEdit: function() {
 75 		this._setCurrentValue(new Date(this._getOriginalValue()));
 76 		this._setDisplayValue(DateFormatter.getFormattedDate({date: this._getCurrentValue(), withStrings: true, shortDate: this.options.shortDate}));
 77 	},
 78 	
 79 	createEditField: function() {
 80 		var datePicker = new bbq.gui.button.DatePickerButton({
 81 			onclick: this.openDatePicker.bind(this),
 82 			toolTip: Language.get("bbq.gui.updateables.UpdateableDateField.opendatepicker")
 83 		});
 84 		
 85 		var node = DOMUtil.createElement("nobr");
 86 		datePicker.appendTo(node);
 87 		
 88 		var dateOrder = DateFormatter.getOrder({shortDate: this.options.shortDate});
 89 		var dateFragment;
 90 		
 91 		for(var i = 0, iCount=dateOrder.length; i < iCount; i++) {
 92 			if(this["generate" + dateOrder[i]]) {
 93 				dateFragment = this["generate" + dateOrder[i]].call(this)
 94 				if(dateFragment.hide !== true || dateFragment.hide == "undefined") { 
 95 					node.appendChild(dateFragment);
 96 				}
 97 			} else {
 98 				Log.info("Invalid date atom - " + dateOrder[i]);
 99 			}
100 		}
101 		
102 		return node;
103 	},
104 	
105 	openDatePicker: function(event) {
106 		try {
107 			if(!this._datePicker) {
108 				this._datePicker = new bbq.gui.DatePicker({width: 142, height: 176, owner: this, title: this.options.pickerTitle, startDate: this._getCurrentValue(), pointerEvent: event, showStartDate: true});
109 				this._datePicker.appear();
110 			}
111 		} catch(e) {
112 			Log.error("Error opening date picker", e);
113 		}
114 		return false;
115 	},
116 	
117 	datePicked: function(pickedDate, autoCorrect) {
118 		if(pickedDate instanceof Date) {
119 			this._getCurrentValue().setTime(pickedDate.getTime());
120 			this.render();
121 			
122 			if(typeof(autoCorrect) == "undefined" ) { // prevent recursion
123 				this.notifyListeners("onValueChanged");
124 			}
125 		}
126 		
127 		this._datePicker = null;
128 	},
129 	
130 	generateDay: function() {
131 		this._dayDropDown = DOMUtil.createElement("select", {
132 			onfocus: this._clearFocusWatcher.bind(this),
133 			onchange: this.updateLocalValue.bindAsEventListener(this),
134 			name: "days"
135 		});
136 		
137 		var daysInMonth = this.getDaysInMonth(this._getCurrentValue());
138 		
139 		for(var i = 1; i <= daysInMonth; i++) {
140 			var option = document.createElement("option");
141 			option.value = i;
142 			option.appendChild(document.createTextNode(i));
143 			
144 			if(i == this._getCurrentValue().getDate()) {
145 				option.selected = "selected";
146 			}
147 			
148 			this._dayDropDown.appendChild(option);
149 		}
150 		
151 		this._dayDropDown.disabled = this.disabled;
152 		
153 		return this._dayDropDown;
154 	},
155 	
156 	generateDateSeparator: function() {
157 		return document.createTextNode(DateFormatter.getDateSeparator());
158 	},
159 	
160 	generateEmptySpace: function() {
161 		return document.createTextNode(" ");
162 	},
163 	
164 	generateBreakableSpace: function() {
165 		return DOMUtil.createElement("wbr");
166 	},
167 	
168 	generateHour: function() {
169 		this._hourDropDown = document.createElement("select");
170 		this._hourDropDown.onchange = this.updateLocalValue.bindAsEventListener(this);
171 		this._hourDropDown.name = "hours";
172 		
173 		for(var i = 1; i < 24; i++) {
174 			var option = document.createElement("option");
175 			option.value = i;
176 			option.appendChild(document.createTextNode((i < 10 ? "0" + i : i)));
177 			
178 			if(i == this._getCurrentValue().getHours()) {
179 				option.selected = true;
180 			}
181 			
182 			this._hourDropDown.appendChild(option);
183 		}
184 		
185 		this._hourDropDown.disabled = this.disabled;
186 		this._hourDropDown.hide = this.options.hideHours;
187 		
188 		return this._hourDropDown;
189 	},
190 	
191 	generateMinute: function() {
192 		this._minuteDropDown = DOMUtil.createElement("select", {onchange: this.updateLocalValue.bindAsEventListener(this), name: "minutes"});
193 		
194 		for(var i = 0; i < 60; i += 15) {
195 			var option = document.createElement("option");
196 			option.value = i;
197 			option.appendChild(document.createTextNode((i < 10 ? "0" + i : i)));
198 			
199 			if(i == this._getCurrentValue().getMinutes()) {
200 				option.selected = true;
201 			}
202 			
203 			this._minuteDropDown.appendChild(option);
204 		}
205 		
206 		this._minuteDropDown.disabled = this.disabled;
207 		this._minuteDropDown.hide = this.options.hideMinutes;
208 		return this._minuteDropDown;
209 	},
210 	
211 	generateMonth: function() {
212 		this._monthDropDown = DOMUtil.createElement("select", {onchange: this.updateLocalValue.bindAsEventListener(this), name: "months"});
213 		
214 		for(var i = 0; i < 12; i++) {
215 			var option = document.createElement("option");
216 			option.value = i;
217 			option.appendChild(document.createTextNode(Language.get("bbq.gui.updateables.UpdateableDateField.months" + i).substr(0, 3)));
218 			
219 			if(i == this._getCurrentValue().getMonth() || "0" + i == this._getCurrentValue().getMonth()) {
220 				option.selected = true;
221 			}
222 			
223 			this._monthDropDown.appendChild(option);
224 		}
225 		
226 		this._monthDropDown.disabled = this.disabled;
227 		
228 		return this._monthDropDown;
229 	},
230 	
231 	generateTimeSeparator: function() {
232 		if(this.options.hideMinutes) {
233 			return this.generateEmptySpace();
234 		} else {
235 			return document.createTextNode(DateFormatter.getTimeSeparator());
236 		}
237 	},
238 	
239 	generateYear: function() {
240 		this._yearDropDown = DOMUtil.createElement("select", {onchange: this.updateLocalValue.bindAsEventListener(this), name: "years"});
241 		var currentDate = new Date();
242 		
243 		for(var i = currentDate.getFullYear() - 5; i < (currentDate.getFullYear() + 10); i++) {
244 			var option = document.createElement("option");
245 			option.value = i;
246 			option.appendChild(document.createTextNode(i));
247 			
248 			if(i == this._getCurrentValue().getFullYear()) {
249 				option.selected = true;
250 			}
251 			
252 			this._yearDropDown.appendChild(option);
253 		}
254 		
255 		this._yearDropDown.disabled = this.disabled;
256 		
257 		return this._yearDropDown;
258 	},
259 	
260 	getDaysInMonth: function(date) {
261 		return 32 - new Date(date.getFullYear(), date.getMonth(), 32).getDate();
262 	},
263 	
264 	getValue: function() {
265 		return this._getCurrentValue().getTime();
266 	},
267 	
268 	saveEdit: function() {
269 		this._setOriginalValue(new Date(this._getCurrentValue()));
270 		this._setDisplayValue(DateFormatter.getFormattedDate({date: this._getCurrentValue(), withStrings: true, shortDate: this.options.shortDate}));
271 	},
272 	
273 	updateLocalValue: function(event) {
274 		// make sure we don't go off the end of the month - ie. select july 31 then select february
275 		var daysInCurrentMonth = this.getDaysInMonth(this._getCurrentValue());
276 		var daysInNewMonth = this.getDaysInMonth(new Date(Form.Element.getValue(this._yearDropDown), Form.Element.getValue(this._monthDropDown)));
277 		var day = $F(this._dayDropDown);
278 		
279 		if(day > daysInNewMonth) {
280 			day = daysInNewMonth;
281 		}
282 		
283 		if(this._hourDropDown && this._minuteDropDown) {
284 			this._setCurrentValue(new Date($F(this._yearDropDown), $F(this._monthDropDown), day, $F(this._hourDropDown), $F(this._minuteDropDown), 0));
285 		} else {
286 			this._setCurrentValue(new Date($F(this._yearDropDown), $F(this._monthDropDown), day, 0, 0, 0));
287 		}
288 		
289 		this.render();
290 		this.notifyListeners("onValueChanged");
291 	},
292 	
293 	mustBeGreaterThan: function(dateField) {
294 		this.registerListener("onValueChanged", function() {
295 			if(dateField.getValue() > this.getValue()) {
296 				dateField.datePicked(new Date(this.getValue()), true);
297 			}
298 		}.bind(this));
299 	},
300 	
301 	mustBeLessThan: function(dateField) {
302 		this.registerListener("onValueChanged", function() {
303 			if(dateField.getValue() < this.getValue()) {
304 				dateField.datePicked(new Date(this.getValue()), true);
305 			}
306 		}.bind(this));
307 	}
