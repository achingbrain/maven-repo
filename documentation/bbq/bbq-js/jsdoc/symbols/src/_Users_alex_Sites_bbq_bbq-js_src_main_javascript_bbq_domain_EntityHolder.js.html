1 include(bbq.util.BBQUtil);
  2 
  3 /**
  4  * Holds domain objects.
  5  *
  6  * @class bbq.domain.EntityHolder
  7  */
  8 bbq.domain.EntityHolder = new Class.create({
  9 
 10 	_entities: null,
 11 	options: null,
 12 
 13 	/**
 14 	 * @param {Object} options
 15 	 *
 16 	 * Supports the following options:
 17 	 *
 18 	 * options: {
 19 	 * 		entities: Array		Entities to pre-populate with
 20 	 * 		type: Object  The type of object - should be a child class of bbq.domain.BBQEntity
 21 	 * }
 22 	 */
 23 	initialize: function(options) {
 24 		this.options = options ? options : {};
 25 		this._entities = [];
 26 
 27 		if(Object.isArray(this.options.entities)) {
 28 			this.options.entities.each(function(entity) {
 29 				this.add(entity);
 30 			}.bind(this));
 31 		}
 32 	},
 33 
 34 	/**
 35 	 * @return {Number} number of elements in object array
 36 	 */
 37 	size: function() {
 38 		return this._entities.length;
 39 	},
 40 
 41 	/**
 42 	 * @param {Number} index
 43 	 */
 44 	get: function(index) {
 45 		return this._entities[index];
 46 	},
 47 
 48 	/**
 49 	 * @param * object An object or a GUID
 50 	 * @return {Number}
 51 	 */
 52 	indexOf: function(object) {
 53 		var index = -1;
 54 
 55 		this._entities.each(function(entity, i) {
 56 			if(entity.equals(object) || entity.getId() == object) {
 57 				index = i;
 58 
 59 				throw $break;
 60 			}
 61 		}.bind(this));
 62 
 63 		return index;
 64 	},
 65 
 66 	/**
 67 	 * @return {Array} array of elements
 68 	 */
 69 	getAll: function() {
 70 		return $A(this._entities);
 71 	},
 72 
 73 	/**
 74 	 * @param {Object} element
 75 	 */
 76 	add: function(element) {
 77 		if(element instanceof this.options.type) {
 78 			// passed entity
 79 			if(this.indexOf(element) != -1) {
 80 				return element;
 81 			}
 82 		} else if(Object.isString(element) || Object.isNumber(element)) {
 83 			// passed identifier
 84 			element = new this.options.type({data: {id: element}});
 85 		} else {
 86 			// passed object data
 87 			element = new this.options.type({data: element});
 88 		}
 89 
 90 		return this.put(element, this._entities.length);
 91 	},
 92 
 93 	/**
 94 	 * @param {Object} element
 95 	 * @param {Number} index
 96 	 */
 97 	put: function(element, index) {
 98 		if(!element instanceof this.options.type) {
 99 			element = new this.options.type({data: element});
100 		}
101 
102 		//Log.info(index);
103 		this._entities[index] = element;
104 
105 		return element;
106 	},
107 
108 	/**
109 	 * Removes the passed entity
110 	 *
111 	 * @param * An element of the same type contained in this list or a GUID
112 	 */
113 	remove: function(object) {
114 		var entityKey = BBQUtil.getKey(object, this.options.entityKey);
115 
116 		for(var i = 0; i < this._objectArray.length; i++) {
117 			if(BBQUtil.getKey(this._objectArray[i], this.options.entityKey) == entityKey) {
118 				this._entities.splice(i, 1);
119 				//delete this._arrayIndexes["index-" + entityKey];
120 				i--;
121 			}
122 		}
123 	},
124 
125 	/**
126 	 * removes all elements
127 	 */
128 	empty: function() {
129 		this._entities = [];
130 	}
131 });
