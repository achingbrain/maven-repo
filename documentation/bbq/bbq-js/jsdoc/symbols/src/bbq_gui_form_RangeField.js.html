  4 bbq.gui.form.RangeField = new Class.create(bbq.gui.form.FormField, /** @lends bbq.gui.form.RangeField.prototype */ {
  5 	_native: null,
  6 	_bar: null,
  7 	_handle: null,
  8 	_value: null,
  9 
 10 	/**
 11 	 * Provides an HTML5 range field with fallback for older browsers.
 12 	 *
 13 	 * @constructs
 14 	 * @extends bbq.gui.form.FormField
 15 	 * @param {Object} options
 16 	 * @param {Number} options.min The minimum value
 17 	 * @param {Number} options.max The maximum value
 18 	 * @param {Number} options.step Value increment size
 19 	 * @param {boolean} options.forceNonNative If true will not use native browser control even if it's supported
 20 	 */
 21 	initialize: function($super, options) {
 22 		try {
 23 			$super(options);
 24 
 25 			if(!this.options.step) {
 26 				this.options.step = 1;
 27 			}
 28 
 29 			if(!this.options.min) {
 30 				this.options.min = 1;
 31 			}
 32 
 33 			if(!this.options.max) {
 34 				this.options.max = 100;
 35 			}
 36 
 37 			if(Browser.forms.types.range && !this.options.forceNonNative) {
 38 				// Supports HTML5 range input type
 39 				this.setAttribute("type", "range");
 40 				this.setAttribute("min", this.options.min);
 41 				this.setAttribute("max", this.options.max);
 42 				this.setAttribute("step", this.options.step);
 43 
 44 				this._native = true;
 45 			} else {
 46 				// ack, fallback to custom version
 47 				this._native = false;
 48 
 49 				// use a hidden input node to store our current value and maintain compatibility with bbq.gui.form.Form#getValues
 50 				this._bar = DOMUtil.createElement("div", {
 51 					className: "RangeField_bar",
 52 					style: {
 53 						width: "100%"
 54 					},
 55 					onclick: this._doDrag.bind(this)
 56 				});
 57 				this._handle = DOMUtil.createElement("div", {
 58 					className: "RangeField_handle",
 59 					style: {
 60 						position: "absolute"
 61 					},
 62 					onmousedown: this._startDrag.bind(this)
 63 				});
 64 
 65 				this.setRootNode("div");
 66 				this.setStyle("position", "relative");
 67 			}
 68 
 69 			this.addClass("RangeField");
 70 		} catch(e) {
 71 			Log.error("Error constructing RangeField", e);
 72 		}
 73 	},
 74 
 75 	/**
 76 	 * @inheritDoc
 77 	 */
 78 	render: function($super) {
 79 		$super();
 80 
 81 		if(!this._native) {
 82 			this.empty();
 83 
 84 			this.appendChild(this._bar);
 85 			this.appendChild(this._handle);
 86 
 87 			// position handle after being added to the DOM
 88 			setTimeout(function() {
 89 				var width = Element.getDimensions(this._bar).width;
 90 
 91 				var range = this.options.max - this.options.min;
 92 				var position = (width/range) * (this._value - this.options.min);
 93 
 94 				if(position >= width) {
 95 					var handleWidth = Element.getDimensions(this._handle).width;
 96 					position -= handleWidth;
 97 				}
 98 
 99 				var percent = ((position/width) * 100);
100 
101 				if(percent > 100) {
102 					percent = 100;
103 				}
104 
105 				DOMUtil.setStyle(this._handle, "left", percent + "%");
106 			}.bind(this), 100);
107 		}
108 	},
109 
110 	_getRawValue: function() {
111 		if(this._native) {
112 			return this.getRootNode().value;
113 		}
114 
115 		return this._value;
116 	},
117 
118 	_setRawValue: function(value) {
119 		if(this._native) {
120 			this.getRootNode().value = value;
121 
122 			return;
123 		}
124 
125 		this._value = value;
126 	},
127 
128 	_startDrag: function() {
129 		var doDrag = this._doDrag.bindAsEventListener(this);
130 
131 		Event.observe(document, "mousemove", doDrag);
132 
133 		Event.observe(document, "mouseup", function(event) {
134 			Event.stopObserving(document, "mouseup", this);
135 			Event.stopObserving(document, "mousemove", doDrag);
136 		});
137 	},
138 
139 	_doDrag: function(event) {
140 		var mouseX = Event.pointerX(event);
141 		var elementX = Element.positionedOffset(this.getRootNode()).left;
142 		var width = Element.getDimensions(this._bar).width;
143 
144 		var x = mouseX - elementX;
145 
146 		if(x < 0) {
147 			x = 0;
148 		}
149 
150 		if(x > width) {
151 			x = width;
152 		}
153 
154 		this._moveHandle(x);
155 	},
156 
157 	_moveHandle: function(position) {
158 		var width = Element.getDimensions(this._bar).width;
159 
160 		// e.g. between 5 and 55 with a step size of 10 there are 6 potential values - 5, 15, 25, 35, 45, 55
161 		var potentialValues = Math.ceil((this.options.max - this.options.min)/this.options.step);
162 
163 		// multiples of this number are valid values for handle positioning
164 		var segmentWidth = Math.ceil(width/potentialValues) + 1;
165 		var newValue = 0;
166 
167 		//Log.info("width " + width + " step size " + this.options.step + " " + potentialValues + " steps");
168 
169 		for(var i = 0; i <= potentialValues; i++) {
170 			var segmentStart = i * segmentWidth;
171 
172 			//Log.info("segmentStart " + segmentStart + " position " + position);
173 
174 			if(position >= (segmentStart - (segmentWidth / 2)) && position <= (segmentStart + (segmentWidth / 2))) {
175 				position = segmentStart;
176 
177 				newValue = this.options.min + (this.options.step * i);
178 
179 				break;
180 			}
181 		}
182 
183 		if(position >= width) {
184 			var handleWidth = Element.getDimensions(this._handle).width;
185 			position -= handleWidth;
186 		}
187 
188 		var percent = ((position/width) * 100);
189 
190 		if(percent > 100) {
191 			percent = 100;
192 		}
193 
194 		DOMUtil.setStyle(this._handle, "left", percent + "%");
195 
196 		this.setValue(newValue);
197 		this.notifyListeners("onChange");
198 	}
