131 	/**
132 	 * @private
133 	 */
134 	_addFile: function() {
135 		if(!this.options.multiple && this._uploadedFiles.length != 0) {
136 			return;
137 		}
138 		
139 		$A(this._filesToUploadDiv.childNodes).each(function(node) {
140 			if(node.tagName.toLowerCase() == "h3") {
141 				node.style.display = "block";
142 			}
143 		});
144 		
145 		this._minimiseThumbnailArea();
146 		
147 		var iframe = DOMUtil.createElement("iframe");
148 		
149 		if(Browser.InternetExplorer) {
150 			// Internet Explorer does not like dynamically created upload boxes (it submits their contents as 
151 			// if they were text inputs) so load a file with a file input in it
152 			iframe.frameBorder = "0";
153 			iframe.src = "/js/uploadFile.html";
154 		}
155 		
156 		Event.observe(iframe, "load", function() {
157 			var iframeDoc = DOMUtil.getIFrameDocument(iframe);
158 			
159 			var input = DOMUtil.createElement("input", {type: "file", name: "assetFile", style: {width: "200px"}});
160 			
161 			if(!Browser.InternetExplorer) {
162 				var form = iframeDoc.createElement("form");
163 				form.method = "post";
164 				form.enctype = "multipart/form-data";
165 				form.appendChild(input);
166 				
167 				iframeDoc.body.appendChild(form);
168 				iframeDoc.body.style.backgroundColor = "#000";
169 				iframeDoc.body.style.margin = "0";
170 				iframeDoc.body.style.padding = "0 10px";
171 			} else {
172 				if(iframeDoc.forms && iframeDoc.forms[0] && iframeDoc.forms[0].assetFile) {
173 					input = iframeDoc.forms[0].assetFile;
174 				}
175 			}
176 			
177 			Event.observe(input, "change", function() {
178 				this._uploadButton.setDisabled(false);
179 				this.notifyListeners("onFileChosen");
180 				
181 				if(!this.options.multiple) {
182 					this._uploadFiles();
183 				}
184 			}.bind(this));
185 		}.bind(this));
186 		
187 		var row = DOMUtil.createTextElement("tr", [
188 			DOMUtil.createTextElement("td", iframe),
189 		]);
190 		
191 		if(this.options.multiple) {
192 			row.appendChild(DOMUtil.createTextElement("td", this._uploadButton));
193 		}
194 		
195 		this._uploadTable.getElementsByTagName("tbody")[0].appendChild(row);
196 		
197 		// store data
198 		this._uploadingFileData.push({
199 			fileUploading: false,
200 			row: row,
201 			iframe: iframe
202 		});
203 	},

205 	/**
206 	 * @private
207 	 */
208 	_uploadFiles: function() {
209 		var hasValidFiles = false;
210 		
211 		// loop through upload boxes and get file data
212 		for(var i = 0, iCount=this._uploadingFileData.length; i < iCount; i++) {
213 			var fileData = this._uploadingFileData[i];
214 			
215 			if(!fileData.fileUploading) {
216 				var form = DOMUtil.getIFrameDocument(fileData.iframe).forms[0];
217 				
218 				if(form && form.assetFile) {
219 					var assetLocation = form.assetFile.value.strip();
220 					
221 					if(!assetLocation) {
222 						if(this._uploadingFileData.length > 1) {
223 							// no file has been chosen, remove frame
224 							this._uploadTable.getElementsByTagName("tbody")[0].removeChild(fileData.row);
225 							this._uploadingFileData.splice(i, 1);
226 							i--;
227 						}
228 						
229 						continue;
230 					}
231 					
232 					var extension = assetLocation.split(".").pop().toLowerCase();
233 					
234 					if(this.options.extensions instanceof Array && this.options.extensions.indexOf(extension) == -1) {
235 						this.notifyListeners("onInvalidFileExtension");
236 						return;
237 					}
238 					
239 					var div = DOMUtil.createTextElement("div", DOMUtil.createTextElement("p", Language.get("bbq.gui.FileUploader.uploading")), {className: "fileUploadWaitBox emptyfileUploadWaitBox"});
240 					this._filesUploadedDiv.appendChild(div);
241 					
242 					fileData.fileUploading = true;
243 					fileData.div = div;
244 					fileData.fileName = this._getFileName(assetLocation);
245 					
246 					// ensure uploadTo is properly formatted, otherwise form.submit won't work
247 					if(this.options.uploadTo.substring(0, 4) != "http") {
248 						this.options.uploadTo = window.location.protocol + "//" + window.location.host + (this.options.uploadTo.substring(0, 1) == "/" ? "" : "/") + this.options.uploadTo;
249 					}
250 					
251 					form.action = this.options.uploadTo;
252 					form.submit();
253 					
254 					Event.observe(fileData.iframe, "load", this._getUploadedData.bind(this, fileData));
255 					
256 					hasValidFiles = true;
257 				}
258 			}
259 		}
260 		
261 		if(hasValidFiles) {
262 			if(!this._addAnother) {
263 				this._uploadButton.setDisabled(true);
264 			}
265 			
266 			var headers = this._filesToUploadDiv.getElementsByTagName("h3");
267 			
268 			for(var i = 0, iCount=headers.length; i < iCount; i++) {
269 				headers[i].style.display = "none";
270 			}
271 			
272 			for(var i = 0, iCount=this._uploadingFileData.length; i < iCount; i++) {
273 				this._uploadingFileData[i].iframe.style.height = "0px";
274 			}
275 			
276 			this._maximiseThumbnailArea();
277 		}
278 	},

280 	/**
281 	 * @private
282 	 */
283 	_getFileName: function(assetLocation) {
284 		// \ for windows, / for everything else
285 		var fileLocation = assetLocation.split(assetLocation.search("/") ? "\\" : "/");
286 					
287 		var fileName = fileLocation[fileLocation.length - 1];
288 		return (fileName.length > 19 ? fileName.substr(0, 8) + "..." + fileName.substr(fileName.length - 8, 8) : fileName);
289 	},

337 	/**
338 	 * Interrogates iFrames for uploaded file data.  Called via an onload event fired on the iFrame after form submission.
339 	 */
340 	_getUploadedData: function(fileData) {
341 		if(fileData.fileUploading) {
342 			if(fileData.iframe.contentWindow && fileData.iframe.contentWindow.getData) {
343 				var iframeWindow = fileData.iframe.contentWindow;
344 				
345 				if(iframeWindow.fileError) {
346 					this._fileUploadError(fileData.fileName, fileData.div);
347 				} else {
348 					this._fileUploaded(fileData, iframeWindow.getData());
349 				}
350 			} else {
351 				Log.dir(fileData.iframe);
352 			}
353 		}
354 	},
355 	
356 	_fileUploadError: function(fileName, outputDiv) {
357 		DOMUtil.emptyNode(outputDiv);
358 		DOMUtil.addClass(outputDiv, "uploadError");
359 		
360 		var deleteLink = new bbq.gui.button.DeleteButton({onclick: this._removeErrorFile.bind(this, outputDiv)});
361 		deleteLink.appendTo(outputDiv);
362 		
363 		outputDiv.appendChild(DOMUtil.createTextElement("p", "Error uploading " + fileName));
364 	},

366 	_removeErrorFile: function(element) {
367 		Effect.Fade(element, {duration: 0.25});
368 		
369 		if(this._uploadedFiles.length == 0) {
370 			this._addFile();
371 			this._minimiseThumbnailArea();
372 		}
373 	},

375 	/**
376 	 * @private
377 	 */
378 	_fileUploaded: function(fileData, passedData) {
379 		// remove old file name so that we can upload the same file twice
380 		var iFrameIndex = 0;
381 		
382 		Log.dir(passedData);
383 		
384 		DOMUtil.emptyNode(fileData.div);
385 		
386 		fileData.div.id = BBQUtil.generateGUID() + "_" + passedData.filetransferid;
387 		
388 		//this._filesToUploadDiv.removeChild(fileData.iframe);
389 		this._uploadTable.getElementsByTagName("tbody")[0].removeChild(fileData.row);
390 		
391 		var image = DOMUtil.createElement("img", {
392 			src: "/backend/uploads/getTemporaryThumbnail?id=" + passedData.filetransferid,
393 			style: {
394 				display: "none"
395 			}
396 		});
397 		
398 		fileData.div.appendChild(image);
399 		DOMUtil.removeClass(fileData.div, "emptyfileUploadWaitBox");
400 		
401 		var deleteLink = new bbq.gui.button.DeleteButton({onclick: this._removeUploadedFile.bind(this, fileData.div)});
402 		deleteLink.appendTo(fileData.div);
403 				
404 		fileData.div.appendChild(DOMUtil.createTextElement("p", [
405 			passedData.filesize,
406 			DOMUtil.createElement("br"),
407 			fileData.fileName
408 		]));
409 		
410 		image.onload = function() {
411 			if(image.width < 128) {
412 				image.style.marginLeft = Math.ceil((128 - image.width)/2) + "px";
413 			}
414 			
415 			if(image.height < 128) {
416 				var newMargin = Math.ceil((128 - image.height)/2) + "px";
417 				DOMUtil.setStyle(image, "marginTop", newMargin);
418 				DOMUtil.setStyle(image, "marginBottom", newMargin);
419 			}
420 		};
421 			
422 		this._uploadedFiles.push(passedData);
423 		
424 		this._updateNumFilesDisplay();
425 		
426 		Effect.Appear(image, {duration: 0.25});
427 	},
428 	
429 	/**
430 	 * @private
431 	 */
432 	_removeUploadedFile: function(element) {
433 		Effect.Fade(element, {duration: 0.25});
434 		
435 		// remove uploaded file data
436 		var keyToRemove = element.id.split("_")[1];
437 		
438 		for(var i = 0, iCount=this._uploadedFiles.length; i < iCount; i++) {
439 			if(this._uploadedFiles[i].filetransferid == keyToRemove) {
440 				this._uploadedFiles.splice(i, 1);
441 				break;
442 			}
443 		}
444 		
445 		this._updateNumFilesDisplay();
446 		
447 		if(this._uploadedFiles.length == 0) {
448 			this._addFile();
449 			this._minimiseThumbnailArea();
450 		}
451 		
452 		return false;
453 	},
