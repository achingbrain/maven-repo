1 /**
  2  * Log utility object
  3  *
  4  * @class Log
  5  */
  6 var Log = {
  7 	debugging: $$("html")[0].hasClassName("debug"),
  8 
  9 	_logIt: function(message, type, nodate, exception) {
 10 		if(!Log.debugging) {
 11 			return;
 12 		}
 13 
 14 		if(!nodate) {
 15 			var date = new Date();
 16 			message = Log._padNumber(date.getHours()) + ":" + Log._padNumber(date.getMinutes()) + ":" + Log._padNumber(date.getSeconds()) + " - " + message;
 17 		}
 18 
 19 		if(window.console) {
 20 			if(window.console[type]) {
 21 				console[type](message);
 22 			} else if(window.console.log) {
 23 				console.log(message);
 24 			}
 25 		}
 26 
 27 		if(exception) {
 28 			Log._logIt(exception, "dir", true);
 29 		}
 30 	},
 31 
 32 	debug: function(message, exception) {
 33 		Log._logIt(message, "debug", false, exception);
 34 	},
 35 
 36 	info: function(message, exception) {
 37 		Log._logIt(message, "info", false, exception);
 38 	},
 39 
 40 	warn: function(message, exception) {
 41 		Log._logIt(message, "warn", false, exception);
 42 	},
 43 
 44 	error: function(message, exception) {
 45 		Log._logIt(message, "error", false, exception);
 46 	},
 47 
 48 	dir: function(object) {
 49 		Log._logIt(object, "dir", true);
 50 	},
 51 
 52 	_padNumber: function(number) {
 53 		if(number < 10) {
 54 			number = "0" + number;
 55 		}
 56 
 57 		return number;
 58 	}
 59 }
