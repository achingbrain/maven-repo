 14 	/**
 15 	 * Removes all children of the passed node
 16 	 * 
 17 	 * @param	{Node}	node
 18 	 * @return void
 19 	 */
 20 	emptyNode: function(node) {
 21 		if(node) {
 22 			if(node.appendTo) {
 23 				node.empty();
 24 			} else {
 25 				while(node.hasChildNodes()) {
 26 					var childNode = node.childNodes[node.childNodes.length-1];
 27 					
 28 					if(childNode.owner) {
 29 						var foo = childNode.owner();
 30 						
 31 						if(foo.notifyListeners) {
 32 							foo.notifyListeners("onBeforeRemoveFromDOM");
 33 						}
 34 					}
 35 					
 36 					node.removeChild(node.childNodes[node.childNodes.length-1]);
 37 					
 38 					if(foo && foo.notifyListeners) {
 39 						foo.notifyListeners("onAfterRemoveFromDOM");
 40 					}
 41 				}
 42 			}
 43 			
 44 			return node;
 45 		}
 46 	},

237 	/**
238 	 * Removes a CSS class to the passed node, optionally recursing through it's child nodes
239 	 * 
240 	 * @param	{Node}	node
241 	 * @param	{string}	oldClass
242 	 * @param	{boolean}	recursive
243 	 * @return	{void}
244 	 */
245 	removeClass: function(node, oldClass, recursive) {
246 		if(!node) {
247 			return;
248 		}
249 		
250 		if(oldClass instanceof Array) {
251 			for(var i = 0; i < oldClass.length; i++) {
252 				this.removeClass(node, oldClass[i], recursive);
253 			}
254 		} else if(recursive) {
255 			this.recursivelyRemoveClasses(node, oldClass);
256 		} else {
257 			Element.removeClassName(node, oldClass);
258 		}
259 	},

267 	recursivelyRemoveClasses: function(node, oldClass) {
268 		if(node && node.hasChildNodes()) {
269 			this.removeClass(node, oldClass);
270 			
271 			for(var i = 0; i < node.childNodes.length; i++) {
272 				if(node.childNodes[i].hasChildNodes()) {
273 					this.recursivelyRemoveClasses(node.childNodes[i], oldClass);
274 				}
275 			}
276 		}

325 	 * @param {Object} element
326 	 * @param {Object} styleToGet
327 	 */
328 	getStyle: function(element, styleToGet) {
329 		try {

331 				// get root node of GUIWidget
332 				element = element.owner().getRootNode();
333 			}
334

336 				return element.currentStyle[styleToGet];
337 			} else if (window.getComputedStyle) {
338 				return document.defaultView.getComputedStyle(element, null).getPropertyValue(styleToGet);
339 			}
340

342 				return element.style[styleToGet];
343 			}
344 		} catch(e) {
345 			Log.error("Error getting style " + styleToGet + " from element " + element, e);

468 	getOccupiedDimensions: function(node) {
469 		if(Object.isFunction(node.getRootNode)) {
470 			node = node.getRootNode();
471 		}
472 		
473 		node = $(node);
474 		
475 		var dims = {
476 			height: 0,
477 			width: 0
478 		};
479 		
480 		if(!node.getHeight || !node.getWidth) {
481 			return dims;
482 		}
483 		
484 		dims.height = node.getHeight();
485 		dims.width = node.getWidth();
486 		
487 		var nextSibling = DOMUtil._findSibling(node, "next");
488 		
489 		// find node siblings - if margins have been set we need to adjust the dimensions
490 		if(nextSibling) {
491 			
492 			// occupied size is affected by margins
493 			if(node.getStyle("marginBottom")) {
494 				var nodeMarginBotton = parseInt(node.getStyle("marginBottom"));
495 				var nextSiblingMarginTop = parseInt(nextSibling.getStyle("marginTop"));
496 				
497 				// only add if the next node's marginTop does not trump our marginBottom
498 				if(nodeMarginBotton > nextSiblingMarginTop) {
499 					dims.height += nodeMarginBotton;
500 				}
501 			}
502 			
503 			if(node.getStyle("marginRight")) {
504 				var nodeMarginRight = parseInt(node.getStyle("marginRight"));
505 				var nextSiblingMarginLeft = parseInt(nextSibling.getStyle("marginLeft"));
506 				
507 				// only add if the next node's marginTop does not trump our marginBottom
508 				if(nodeMarginRight > nextSiblingMarginLeft) {
509 					dims.width += nodeMarginRight;
510 				}
511 			}
512 		}
513 		
514 		var previousSibling = DOMUtil._findSibling(node, "previous");
515 		
516 		if(previousSibling) {
517 			
518 			// occupied size is affected by margins
519 			if(node.getStyle("marginTop")) {
520 				var nodeMarginTop = parseInt(node.getStyle("marginTop"));
521 				var previousSiblingMarginBottom = parseInt(previousSibling.getStyle("marginBottom"));
522 				
523 				// only add if the next node's marginTop does not trump our marginBottom
524 				if(nodeMarginTop > previousSiblingMarginBottom) {
525 					dims.height += nodeMarginTop;
526 				}
527 			}
528 			
529 			if(node.getStyle("marginLeft")) {
530 				var nodeMarginLeft = parseInt(node.getStyle("marginLeft"));
531 				var previousSiblingMarginBottom = parseInt(previousSibling.getStyle("marginRight"));
532 				
533 				// only add if the next node's marginTop does not trump our marginBottom
534 				if(nodeMarginLeft > previousSiblingMarginBottom) {
535 					dims.width += nodeMarginLeft;
536 				}
537 			}
538 		}
539 		
540 		return dims;
541 	},

609 	append: function(what, toNode) {
610 		if(Object.isString(what) || Object.isNumber(what)) {
611 			// string or number
612 			toNode.appendChild(document.createTextNode(what));
613 		} else if(what.appendTo) {
614 			// GUIWidget
615 			what.appendTo(toNode);
616 		} else if(what.appendChild) {
617 			// Node
618 			toNode.appendChild(what);
619 		} else if(Object.isArray(what)) {
620 			// Array of items
621 			what.each(function(item) {
622 				DOMUtil.append(item, toNode);
623 			});
624 		}

627 	elementWasClickedOn: function(event, element) {
628 		var position = Element.positionedOffset(element);
629 		var dims = Element.getDimensions(element);
630 		var mouseX = Event.pointerX(event);
631 		var mouseY = Event.pointerY(event);
632 		var inBoxHorizontal = mouseX > position.left && mouseX < position.left + dims.width;
633 		var inBoxVertical = mouseY > position.top && mouseY < position.top + dims.height;
634 		
635 		return inBoxHorizontal && inBoxVertical;
636 	}
