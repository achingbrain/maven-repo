 12 	/**
 13 	 * <p>This class is used as a base for all objects that have GUI representations.</p>
 14 	 *
 15 	 * <p>The constructor method of this class should always be called explicitly by child classes.</p>
 16 	 *
 17 	 * <p>The rootNode property is a DOM node that is attached to the DOM tree after the object is
 18 	 * created via the appendTo method.</p>
 19 	 *
 20 	 * <pre><code class="language-javascript">
 21 	 * com.myapp.MyWidget = new Class.create(bbq.gui.GUIWidget, {
 22 	 *      initialize: function($super, options) {
 23 	 *          $super(options);
 24 	 *
 25 	 *          this.appendChild(DOMUtil.createElement("p", "Hello world!"));
 26 	 *      }
 27 	 * });
 28 	 *
 29 	 * ...
 30 	 *
 31 	 * var widget = new com.myapp.MyWidget({attributes: { className: "Foo" }});
 32 	 * widget.appendTo(document.body);
 33 	 * </code></pre>
 34 	 *
 35 	 * @memberOf bbq.gui
 36 	 * @constructs
 37 	 * @param {Object} options
 38 	 * @param {Object} [options.attributes] A key/value object of attributes to be applied to the root node
 39 	 * @extends bbq.lang.Delegator
 40 	 */

 47 	/**
 48 	 * This method should be overridden by child classes and called explicitly by the overriding method if they make use of the 
 49 	 * childWidgets array, which they should, haha.
 50 	 * 
 51 	 * The idea of the method is to create a DOM node tree representation of the widget.  The root node of the tree should be 
 52 	 * this._rootNode.  The presence of any other nodes in the DOM tree should not be relied upon.
 53 	 * 
 54 	 * This method will be called before the node tree is added to the main document tree (in a similar way to off-screen buffering
 55 	 * in graphics programming) and may be called at seeming arbitrary times.  Consequently it should always create a 
 56 	 * representation of the widget/object in it's current state but at the same time, not rely on any other portion of the DOM tree 
 57 	 * existing.
 58 	 */

 63 	/**
 64 	 * Sets the root node and registers this object for retrieval from the root node.
 65 	 * 
 66 	 * This object can then be retrieved by calling owner() on the dom node passed as the argument to this method.
 67 	 * 
 68 	 * @param {Node or String} rootNode Either a DOM Node (from document.createElement) or the String name of a node - e.g. "div"
 69 	 * @example
 70 	 * <pre><code class="language-javascript">
 71 	 * bbq.gui.form.TextField = new Class.create(bbq.gui.GUIWidget, {
 72 	 *     initialize: function($super, options) {
 73 	 *       $super(options);
 74 	 *
 75 	 *       this.setRootNode(document.createElement("input"));
 76 	 *       this.addClass("TextField");
 77 	 *       this.setAttribute("value", this.options.value);
 78 	 *   }
 79 	 * });
 80 	 * 
 81 	 * ...
 82 	 * 
 83 	 * var foo = new bbq.gui.form.TextField({value: "a value"});
 84 	 * foo.appendTo(document.body);
 85 	 * 
 86 	 * ...
 87 	 * 
 88 	 * var bar = document.getElementById("exampleObject");
 89 	 * var foo = bar.owner();
 90 	 * </code></pre>
 91 	 */
 92 	setRootNode: function(rootNode) {
 93 		if(this._rootNode) {
 94 			var oldNode = this._rootNode;
 95 		}
 96 
 97 		if(Object.isString(rootNode)) {
 98 			this._rootNode = document.createElement(rootNode);
 99 		} else {
100 			this._rootNode = rootNode;
101 		}
102 
103 		if(oldNode && oldNode.className) {
104 			this.addClass(oldNode.className.split(" "));
105 		}
106 
107 		if(this.options && this.options.attributes) {
108 			var attr = this.options.attributes;
109 			for(var key in attr) {
110 				this.setAttribute(key, attr[key]);
111 			}
112 		}
113 
114 		this._rootNode = $(this._rootNode);
115 		this.registerObject();
116 	},

201 	appendAfter: function(node) {
202 		try {
203 			node = this._getNode(node);
204 
205 			// find the parent node
206 			var parentNode = node.parentNode;
207 
208 			// find the index of the node
209 			var nodeArray = $A(parentNode.childNodes);
210 			var index = nodeArray.indexOf(node);
211 			var nextNode = nodeArray[index + 1];
212 
213 			parentNode.insertBefore(this.getRootNode(), nextNode);
214 
215 			this.render();
216 		} catch(e) {
217 			Log.error("Error while appending after " + node, e);
218 		}
219 	},

221 	/**
222 	 * Adds a node or GUIWidget to the root node of this element
223 	 * 
224 	 * @param	{Mixed}	childNode	A Node, GUIWidget or array of Node and/or GUIWidget objects
225 	 */
226 	appendChild: function(childNode) {
227 		if(!childNode) {
228 			return;
229 		}
230 		
231 		if(childNode instanceof Array) {
232 			childNode.each(function(node) {
233 				this.appendChild(node);
234 			}.bind(this));
235 		} else {
236 			if(childNode && this._rootNode) {
237 				if(childNode.appendTo instanceof Function) {
238 					childNode.appendTo(this._rootNode);
239 				} else if(childNode.toUpperCase instanceof Function) {
240 					this._rootNode.appendChild(document.createTextNode(childNode));
241 				} else {
242 					this._rootNode.appendChild(childNode);
243 				}
244 			}
245 		}
246 		
247 		return childNode;
248 	},

250 	/**
251 	 * Attempts to remove the passed node if it is a child of this._rootNode
252 	 * 
253 	 * The passed argument can be either a DOM Node object, or a GUIWidget object.
254 	 * 
255 	 * @param	{Mixed}	A child node
256 	 */
257 	removeChild: function(childNode) {
258 		if(childNode && this.getRootNode()) {
259 			try {
260 				if(childNode.getRootNode instanceof Function) {
261 					this.getRootNode().removeChild(childNode.getRootNode());
262 				} else {
263 					this.getRootNode().removeChild(childNode);
264 				}
265 			} catch(e) {
266 				Log.error("Error removing child " + childNode, e);
267 			}
268 		}
269 	},

271 	/**
272 	 * Replaces a DOM node or GUIWidget that is a child of the root node of this object
273 	 * 
274 	 * @param	{Node || GUIWidget}		oldNode		The outgoing child
275 	 * @param	{Node || GUIWidget}		newNode		The incoming child
276 	 * @return {Node || GUIWidget}		The incoming child
277 	 */
278 	replaceChild: function(oldNode, newNode) {
279 		var output = newNode;
280 
281 		if(oldNode && newNode && this.getRootNode()) {
282 			try {
283 				if(oldNode.getRootNode instanceof Function) {
284 					oldNode = oldNode.getRootNode();
285 				}
286 
287 				if(newNode.getRootNode instanceof Function) {
288 					newNode.render();
289 					newNode = newNode.getRootNode();
290 				}
291 
292 				// check that we actually contain the old node before attempting to replace it
293 				if(Element.descendantOf(oldNode, this.getRootNode())) {
294 					this.getRootNode().replaceChild(newNode, oldNode);
295 				}
296 			} catch(e) {
297 				Log.error("Error replacing child " + oldNode + " for " + newNode, e);
298 			}
299 		}
300 
301 		return output;
302 	},

346 	 * @param	{String}	attributeValue	e.g. "anElement"
347 	 */
348 	setAttribute: function(attributeName, attributeValue) {
349 		if(this._rootNode) {
350 			try {
351 				if(attributeName == "style" && attributeValue instanceof Object) {
352 					for(var key in attributeValue) {
353 						this.setStyle(key, attributeValue[key]);
354 					}
355 				} else if(this._rootNode[attributeName] != attributeValue) {
356 					this._rootNode[attributeName] = attributeValue;
357 				}
358 			} catch(e) {
359 				Log.error("exeption thrown setting " + attributeName + " to " + attributeValue, e);
360 			}
361 		}
362 	},

395 			var nodeList = $A(scrollable.parentNode.childNodes);
396 			
397 			nodeList.each(function(node){

401 				}

404 			scrollable.style[dim] = maxSize + "px";
405 			scrollable.owner().triggerEvent("onResize");
406 		}
