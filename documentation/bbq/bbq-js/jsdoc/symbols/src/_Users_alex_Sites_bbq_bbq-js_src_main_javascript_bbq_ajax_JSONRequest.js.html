  5 /**
  6  * Sends an asynchronous HTTP request to a server.  The passed arguments are serialized to a
  7  * JSON string and the response is also expected to be a JSON string.  The request goes out
  8  * as a POST request and the JSON string is sent as the request body.
  9  *
 10  *<pre>
 11  *<code class="language-javascript">
 12  * new bbq.ajax.JSONRequest({
 13  *     // The URL to send the request to - can be relative
 14  *     url: "/some/path",
 15  *
 16  *     // The arguments to send - N.B. will be JSON serialized
 17  *     args: {
 18  *          foo: "bar"
 19  *     }
 20  *
 21  *     // Invoked on success.
 22  *      // The first argument is the XMLHTTPRequest object,
 23  *      // the second is the deserialized server response
 24  *     onSuccess: function(serverResponse, json) {
 25  *
 26  *     },
 27  *
 28  *     // Invoked when the request fails - eg. a X-BBQ-ResponseCode
 29  *     // header is sent with a negative value
 30  *     onFailure: function() {
 31  *
 32  *     }
 33  * });
 34  * </code>
 35  * </pre>
 36  *
 37  * @class bbq.ajax.JSONRequest is a subclass of bbq.ajax.AJAXRequest
 38  * @extends bbq.ajax.AJAXRequest
 39  */
 40 bbq.ajax.JSONRequest = Class.create(bbq.ajax.AJAXRequest, {
 41 
 42 	/**
 43 	 * @constructor
 44 	 * @param $super
 45 	 * @param options
 46 	 */
 47 	initialize: function($super, options) {
 48 		// override content type
 49 		options.contentType = "application/json";
 50 		
 51 		$super(options);
 52 	},
 53 
 54 	/**
 55 	 * @param {String} handlerName
 56 	 * @param {Object} args
 57 	 */
 58 	_callHandler: function($super, handlerName, args) {
 59 		var serverResponse = args[0];
 60 		var json = {};
 61 		var contentType = serverResponse.getHeader("Content-Type");
 62 
 63 		if(contentType && contentType.indexOf("application/json") != -1 && serverResponse.responseText != "") {
 64 			// only attempt decode JSON if the response content type is application/json
 65 			try {
 66 				json = serverResponse.responseText.evalJSON(true);
 67 			} catch(e) {
 68 				Log.error("Error de-escaping JSON", e);
 69 			}
 70 		}
 71 
 72 		$super(handlerName, [serverResponse, json]);
 73 	},
 74 
 75 	_getPostBody: function() {
 76 		return Object.toJSON(this.options.args);
 77 	},
 78 
 79 	_createRequestHeaders: function() {
 80 		return {
 81 			"Accept": "application/json;charset=UTF-8"
 82 		};
 83 	}
 84 });
