 16 	/**
 17 	 * <p>Base class for domain objects.  Supports being partially loaded - that is, properties can be asked for from
 18 	 * this object in advance of them being loaded.  If an unloaded property is requested, a full object load
 19 	 * will be trigged.</p>
 20 	 *
 21 	 * <p>Extending classes should implement:</p>
 22 	 *
 23 	 * <ol>
 24 	 *   <li>The _retrieveURL property.  This should be a URL that will answer to a JSON request of the
 25 	 *    form {id: value} where value is returned from getId() being called on this object</li>
 26 	 *   <li>The _getDefaultObject method.  This should return an object with keys for every property this
 27 	 *      object expects to have filled after a call to _retrieveURL.</li>
 28 	 *  </ol>
 29 	 *
 30 	 * @constructs
 31 	 * @extends bbq.lang.Watchable
 32 	 * @param {Object} options
 33 	 */
 34 	initialize: function($super, options) {
 35 		$super(options);
 36 		
 37 		this.options = options ? options : {};
 38 		this._data = {};
 39 		this._propertyDisplays = new Hash();
 40 		this._loadedFields = new Hash();
 41 		
 42 		this.processData(this.options.data);
 43 	},

 45 	/**
 46 	 * Should return an object similar to the data structure of this class.
 47 	 * 
 48 	 * The object should contain properties that mirror the output of getArray() in the PHP on the server.
 49 	 * @private
 50 	 * @return {Object}
 51 	 */
 52 	_getDefaultObject: function() {
 53 		return {
 54 			
 55 		}
 56 	},

 58 	/**
 59 	 * Sets properties on this object and creates getters/setters for accessing them
 60 	 * 
 61 	 * Also used by child classes to replace IDs with references to the actual objects
 62 	 * 
 63 	 * @param	{Object}	data
 64 	 * @return	void
 65 	 */
 66 	processData: function(data) {
 67 		if(data && data instanceof Object) {
 68 			var defaultObject = this._getDefaultObject();
 69 
 70 			if(Object.isUndefined(defaultObject["id"])) {
 71 				Log.warn("Object loading data from " + this._retrieveURL + " should declare an id property in _getDefaultObject");
 72 			}
 73 
 74 			for(var key in data) {
 75 				// skip fields not defined on this object - this way we do not end up with erroneous getters and setters
 76 				if(typeof(defaultObject[key]) == "undefined") {
 77 					Log.warn("skipping " + key + " on object loading data from " + this._retrieveURL);
 78 					continue;
 79 				}
 80 
 81 				var camel = BBQUtil.capitalize(key);
 82 
 83 				if(this["set" + camel] instanceof Function) {
 84 					this["set" + camel](data[key]);
 85 				} else {
 86 					//Log.info("creating " + camel);
 87 					this["get" + camel] = (new Function("return this._get(\"" + key + "\");")).bind(this);
 88 					this["set" + camel] = (new Function("this._set(\"" + key + "\", arguments[0]);")).bind(this);
 89 				
 90 					this["set" + camel](data[key]);
 91 				}
 92 
 93 				this._loadedFields.set(key, true);
 94 			}
 95 
 96 			this._partialLoad = false;
 97 
 98 			if(Log.debugging) {
 99 				var missingProperties = [];
100 
101 				// ensure we have loaded all of our properties
102 				for(var key in defaultObject) {
103 					if(typeof(data[key]) == "undefined") {
104 						missingProperties.push(key);
105 					}
106 				}
107 
108 				if(missingProperties.length > 0) {
109 					Log.warn(this._retrieveURL + " missing properties " + missingProperties.join(", "));
110 				}
111 			}
112 
113 			// ensure we have loaded all of our properties
114 			for(var key in defaultObject) {
115 				if(Object.isUndefined(data[key])) {
116 					this._dataLoaded = false;
117 					this._partialLoad = true;
118 
119 					// missing at least one property, get out of loop
120 					break;
121 				}
122 			}
123 
124 			var keys = Object.keys(data);
125 
126 			if(keys.length == 1 &&  keys[0] == "id") {
127 				this.loadData();
128 			} else if(!this._partialLoad) {
129 				this._dataLoaded = true;
130 			}
131 		}
132 	},
133 
134 	dataLoaded: function() {
135 		return this._dataLoaded === true;
136 	},

138 	/**
139 	 * Makes an entity load it's data.
140 	 * 
141 	 * Register for listener "onDataLoaded" to interact with this object after data has been loaded.
142 	 * 
143 	 * @return void
144 	 */
145 	loadData: function() {
146 		if(this._loadingData) {
147 			return;
148 		}
149 
150 		this._loadingData = true;
151 
152 		if(!this._retrieveURL) {
153 			Log.error("Objects must specify where to load data from", this);
154 
155 			return;
156 		}
157 
158 		this._dataLoaded = false;
159 		this._partialLoad = false;
160 
161 		new bbq.ajax.JSONRequest({
162 			url: this._retrieveURL, 
163 			args: {id: this.getId()}, 
164 			onSuccess: this._loadedData.bind(this), 
165 			method: "post"
166 		});
167 	},

169 	/**
170 	 * @return void
171 	 */
172 	_loadedData: function(serverResponse, json) {
173 		try {
174 			this._loadingData = false;
175 
176 			this.processData(json);
177 
178 			if(this.dataLoaded()) {
179 				this.notifyListeners("onDataLoaded");
180 			}
181 		} catch(e) {
182 			Log.error("Error encountered while processing data", e);
183 		}
184 	},

186 	/**
187 	 * {@inheritDoc}
188 	 */
189 	registerListener: function($super, type, callback) {
190 		var callbackKey = $super(type, callback);
191 
192 		if(type == "onDataLoaded") {
193 			if(this.dataLoaded()) {
194 				// if the callback is onDataLoaded and we've already loaded our data, call the callback immediately
195 				this.notifyListener(type, callbackKey, [this]);
196 			} else {
197 				// otherwise trigger data load
198 				this.loadData();
199 			}
200 		}
201 
202 		return callbackKey;
203 	},

243 	 * @param	String		property		A property on this object
244 	 */
245 	getProperty: function(property) {
246 		try {
247 			if(property.indexOf(".") != -1) {
248 				property = "" + property.split(".", 1);
249 			}
250 
251 			if(this["get" + BBQUtil.capitalize(property)] instanceof Function) {
252 				return this["get" + BBQUtil.capitalize(property)]();
253 			}
254 		} catch(e) {
255 			Log.error("getProperty threw a wobbly on " + property + " " + this._createURL, e);
256 			Log.error("this.get" + BBQUtil.capitalize(property) + " = " + this["get" + BBQUtil.capitalize(property)]);
257 			Log.dir(this);
258 		}

261 	/**
262 	 * Returns a DOM element (by default a SPAN) that contains a textual representation of the requested property of this element.
263 	 * 
264 	 * Will be updated automatically to contain the current value.
265 	 * 
266 	 * Supports getting properties of sub objects.  For example, if we want the name of this objects's creator, we can do:
267 	 * 
268 	 * <code>
269 	 * object.getPropertyDisplay({property: "creator.fullname"});
270 	 * </code>
271 	 * 
272 	 * This will have the same effect as:
273 	 * 
274 	 * <code>
275 	 * var creator = object.getCreator();
276 	 * creator.getPropertyDisplay({property: "fullname"});
277 	 * </code>
278 	 * 
279 	 * Supports the following options:
280 	 * 
281 	 * options: {
282 	 * 		property: String				// The name of the property that is to be displayed
283 	 * 		formatter: Function			// Optional.  A function that takes the property value as an argument and returns a String or a Node
284 	 * 		nodeName: String			// Optional.  Will be used in place of SPAN
285 	 * 		className: String			// Optional.  Will be applied to the node
286 	 * 		createNode: Function		// Optional.  Should return a DOM node.  Omit this to use a SPAN.  If passed you should also pass a function for updateNode
287 	 * 		updateNode: Function		// Optional.  Expect two arguments - a node and the property value.  Return nothing.  If passed you should also pass a function for createNode
288 	 * }
289 	 * 
290 	 * @param	Object			options
291 	 */
292 	getPropertyDisplay: function(options) {
293 		try {
294 			if(options.property.indexOf(".") != -1) {
295 				// pass request on to child object
296 				var parts = options.property.split(".");
297 				var myProperty = parts.shift();
298 				var object = this.getProperty(myProperty)
299 
300 				options.property = parts.join(".");
301 
302 				if(object) {
303 					// object is valid, pass the request on
304 					return object.getPropertyDisplay(options);
305 				} else {
306 					// an attempt to call a function on a child object that has not been set.  return a placeholder
307 					var node = options.createNode instanceof Function ? options.createNode() : DOMUtil.createElement("span");
308 
309 					if(options.formatter) {
310 						var value = options.formatter(null);
311 
312 						if(value) {
313 							DOMUtil.append(value, node);
314 						}
315 					}
316 
317 					return node;
318 				}
319 			}
320 			
321 			var propertyDisplay = {
322 				node: options.createNode instanceof Function ? options.createNode() : DOMUtil.createElement(options.nodeName ? options.nodeName : "span", {className: options.className ? options.className : ""}),
323 				formatter: options.formatter,
324 				property: options.property,
325 				updateNode: options.updateNode
326 			};
327 
328 			if(this["get" + BBQUtil.capitalize(options.property)] instanceof Function) {
329 				// we've loaded our data already, update current property displays
330 				this._updatePropertyDisplay(propertyDisplay);
331 			} else {
332 				// ensure that this property display will be updated
333 				this.registerOneTimeListener("onDataLoaded", function() {
334 					this._updatePropertyDisplay(propertyDisplay);
335 				}.bind(this));
336 
337 				// if this object is partially loaded and requested property has not been loaded yet, trigger loading of this objects data
338 				if(this._partialLoad && !this._loadedFields.get(options.property)) {
339 					Log.info("Partially loaded object triggering loadData");
340 					this.loadData();
341 				}
342 			}
343 
344 			// create array for the requested property
345 			if(!this._propertyDisplays.get(options.property)) {
346 				this._propertyDisplays.set(options.property, []);
347 			}
348 
349 			// add it to our list of property diplays
350 			this._propertyDisplays.get(options.property).push(propertyDisplay);
351 
352 			// clean up previously created property displays that are no longer in the DOM
353 			this._cleanUpPropertyDisplays();
354 
355 			// return the node for addition to the DOM
356 			return propertyDisplay.node;
357 		} catch(e) {
358 			Log.error("Exception thrown while trying to retrieve property display " + this._createURL, e);
359 			Log.dir(options);
360 		}
361 	},

363 	/**
364 	 * Loops through all existing property displays and updates them.
365 	 * 
366 	 * @private
367 	 */
368 	_updatePropertyDisplays: function(property) {
369 		if(this._propertyDisplays.get(property)) {
370 			this._propertyDisplays.get(property).each(function(displayProperty) {
371 				this._updatePropertyDisplay(displayProperty);
372 			}.bind(this));
373 		}
374 	},

376 	/**
377 	 * Updates stored property displays to contain the current value as known by this object
378 	 * 
379 	 * @private
380 	 */
381 	_updatePropertyDisplay: function(propertyDisplay) {
382 		DOMUtil.emptyNode(propertyDisplay.node);
383 
384 		var value = this.getProperty(propertyDisplay.property);
385 
386 		if(propertyDisplay.formatter instanceof Function) {
387 			value = propertyDisplay.formatter(value);
388 		}
389 
390 		if(propertyDisplay.updateNode instanceof Function) {
391 			propertyDisplay.updateNode(propertyDisplay.node, value);
392 		} else {
393 			if(!value) {
394 				value = " ";
395 			}
396 
397 			// default action is to treat node as if it is a SPAN
398 			if(Object.isString(value)) {
399 				var theText = value.split("\n");
400 
401 				for(var i = 0; i < theText.length; i++) {
402 					propertyDisplay.node.appendChild(document.createTextNode(theText[i]));
403 
404 					if(theText.length > 1 && i != (theText.length - 1)) {
405 						propertyDisplay.node.appendChild(document.createElement("br"));
406 					}
407 				}
408 			} else {
409 				DOMUtil.append(value, propertyDisplay.node);
410 			}
411 		}
412 	},

414 	/**
415 	 * Loops through all stored property displays, removing any references to nodes that have been removed from the DOM
416 	 * 
417 	 * @private
418 	 */
419 	_cleanUpPropertyDisplays: function() {
420 		if(this._propertyDisplayCleanupInterval) {
421 			clearInterval(this._propertyDisplayCleanupInterval);
422 		}
423 
424 		// set an interval to allow whatever rendering action that is currently underway to complete
425 		// use interval instead of timeout so that we only do this once
426 		this._propertyDisplayCleanupInterval = setInterval(function() {
427 			this._propertyDisplays.keys().each(function(key) {
428 				for(var i = 0; i < this._propertyDisplays.get(key).length; i++) {
429 					if(!DOMUtil.isInDOM(this._propertyDisplays.get(key)[i].node)) {
430 						// element has been removed from DOM, delete our reference
431 						this._propertyDisplays.get(key).splice(i, 1);
432 						i--;
433 					}
434 				}
435 			}.bind(this));
436 
437 			clearInterval(this._propertyDisplayCleanupInterval);
438 		}.bind(this), 1000);
439 	}
