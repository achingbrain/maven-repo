  5 /**
  6  * Wrapper for Prototype Ajax class.
  7  * 
  8  * Adds extras such as an independent server timeout and error handling.
  9  * Typically you would use a subclass such as <code class="language-javascript">bbq.ajax.JSONRequest</code>.
 10  *
 11  * @class bbq.ajax.AJAXRequest
 12  * @see bbq.ajax.JSONRequest
 13  */
 14 bbq.ajax.AJAXRequest = Class.create({
 15 	options: null,
 16 	_timeOut: null,
 17 	_interval: null,
 18 
 19 	/**
 20 	 * Supports the following options:
 21 	 *
 22 	 * <pre>
 23 	 * <code class="language-javascript">
 24 	 * options {
 25 	 * 		url: String								// where to send the request to
 26 	 * 		method: String					// post or get
 27 	 * 		args: Object						// key->value pairs to convert to query string
 28 	 * 		onSuccess: Function			// Everything went as expected - eg. received HTTP 200
 29 	 * 		onFailure: Function				// The call failed - eg. did not receive HTTP 200
 30 	 * 		onException: Function			// An exception was thrown while attemping to make the call
 31 	 * 		onAnything: Function			// invoked when there's no other callback to invoke
 32 	 * }
 33 	 * </code>
 34 	 * </pre>
 35 	 *
 36 	 * @constructor
 37 	 * @param	 {Object} options
 38 	 */
 39 	initialize: function(options) {
 40 		this.options = options;
 41 
 42 		if(!this.options.method) {
 43 			this.options.method = "post";
 44 		}
 45 
 46 		if(!this.options.args) {
 47 			this.options.args = {};
 48 		}
 49 
 50 		if(!this.options.headers) {
 51 			this.options.headers = {};
 52 		}
 53 
 54 		this._sendRequest();
 55 	},
 56 
 57 	_getPostBody: function() {
 58 		return this.options.postBody;
 59 	},
 60 
 61 	/**
 62 	 * Sends the request via the specified method
 63 	 * 
 64 	 * @return	void
 65 	 */
 66 	_sendRequest: function() {
 67 		try {
 68 			var requestHeaders = this._createRequestHeaders();
 69 
 70 			for(var key in requestHeaders) {
 71 				this.options.headers[key] = requestHeaders[key];
 72 			}
 73 
 74 			var request = new Ajax.Request(this.options.url, {
 75 				method: this.options.method, 
 76 				postBody: this._getPostBody(),
 77 				onSuccess: this._onSuccess.bind(this),
 78 				onFailure: this._onFailure.bind(this),
 79 				onExcepton: this._onException.bind(this),
 80 				requestHeaders: this.options.headers,
 81 				contentType: this.options.contentType
 82 			});
 83 
 84 			if(typeof(firebug) != "undefined" && firebug.watchXHR instanceof Function) {
 85 				// enable firebug lite to watch the ajax call status
 86 				request.transport._name = this.options.method.toUpperCase() + ' ' + this.options.url;
 87 				firebug.watchXHR(request.transport);
 88 			}
 89 
 90 			// get time out data from server configuration
 91 			if(typeof(ServerConfig) != "undefined" && ServerConfig["timeout"]) {
 92 				this._timeOut = ServerConfig["timeout"];
 93 			} else {
 94 				this._timeOut = 30;
 95 			}
 96 
 97 			// check timeout every second
 98 			this._interval = setInterval(this._checkTimeOut.bind(this), 1000);
 99 
100 			if(typeof(NotificationArea) != "undefined") {
101 				NotificationArea.startLoad();
102 			}
103 		} catch(e) {
104 			Log.error("Could not send request", e);
105 		}
106 	},
107 
108 	_createRequestHeaders: function() {
109 		return {
110 
111 		};
112 	},
113 
114 	/**
115 	 * @access	protected
116 	 * @param {String} handlerName
117 	 * @param {Object} args
118 	 */
119 	_callHandler: function(handlerName, args) {
120 		try {
121 			if(typeof(NotificationArea) != "undefined") {
122 				NotificationArea.stopLoad();
123 			}
124 
125 			clearInterval(this._interval);
126 
127 			if(this.options[handlerName] && this.options[handlerName] instanceof Function) {
128 				this.options[handlerName].apply(this, args);
129 			} else if(this.options["onAnything"] && this.options["onAnything"] instanceof Function) {
130 				this.options["onAnything"].apply(this, args);
131 			}
132 		} catch(e) {
133 			Log.error("Error encountered while invoking handler " + handlerName, e);
134 		}
135 	},
136 
137 	/**
138 	 * @param {Object} serverResponse
139 	 */
140 	_onSuccess: function(serverResponse) {
141 		try {
142 			var responseType = serverResponse.getResponseHeader("X-BBQ-ResponseType");
143 
144 			if(responseType < 0) {
145 				if(this.options.onFailure) {
146 					// have been passed a failure callback
147 					this._onFailure(serverResponse);
148 				} else {
149 					// fall back to default behaviour
150 					var code = "error" + responseType;
151 					var handler = bbq.ajax.AJAXRequest.errorHandlers[code];
152 
153 					if (handler && handler instanceof Function) {
154 						handler.call(this, this, serverResponse);
155 					} else {
156 						Log.error("Received error code " + responseType + " but have no handler to handle it.");
157 						Log.error("Either pass an onFaliure callback to this bbq.ajax.AJAXRequest or define bbq.ajax.AJAXRequest.errorHandlers['" + code + "'] = function(serverResponse) { ... };");
158 					}
159 				}
160 
161 				return;
162 			}
163 
164 			this._callHandler("onSuccess", $A(arguments));
165 		} catch(e) {
166 			Log.error("Could not invoke onSuccess handler", e);
167 		}
168 	},
169 
170 	_onFailure: function() {
171 		try {
172 			Log.error('Request to ' + this.options.method.toUpperCase() + ' ' + this.options.url + " failed");
173 			this._callHandler("onFailure", $A(arguments));
174 		} catch(e) {
175 			Log.error("Could not invoke onFaliure handler", e);
176 		}
177 	},
178 
179 	_onException: function() {
180 		try {
181 			Log.error('Request to ' + this.options.method.toUpperCase() + ' ' + this.options.url + " threw exception");
182 			this._callHandler("onException", $A(arguments));
183 		} catch(e) {
184 			Log.error("Could not invoke onException handler", e);
185 		}
186 	},
187 
188 	/**
189 	 * Checks to see if the timer has reached 0.  If so the server request has timed out.
190 	 * 
191 	 * @return	void
192 	 */
193 	_checkTimeOut: function() {
194 		if(this._timeOut == 0) {
195 			this._timedOut();
196 			clearInterval(this._interval);
197 		} else {
198 			this._timeOut--;
199 		}
200 	},
201 
202 	/**
203 	 * Shows the user a warning
204 	 * 
205 	 * @return	void
206 	 */
207 	_timedOut: function() {
208 		Log.warn("Ajax call to " + this.options.url + " timed out");
209 
210 		if(typeof(NotificationArea) != "undefined" && typeof(Language) != "undefined" && Language.get instanceof Function) {
211 			NotificationArea.stopLoad();
212 			NotificationArea.setMessage(
213 				Language.get("bbq.ajax.AJAXRequest.ajaxtimeoutheader"), 
214 				Language.get("bbq.ajax.AJAXRequest.ajaxtimeoutmessage"), 
215 				"error"
216 			);
217 			NotificationArea.setMessage(
218 				Language.get("bbq.ajax.AJAXRequest.reloadpageheader"), 
219 				Language.get("bbq.ajax.AJAXRequest.reloadpagemessage"), 
220 				"error"
221 			);
222 		}
223 	}
224 });

226 /**
227  * Add custom error handlers to this map.  Error handlers should
228  * be a function with the following signature:
229  * 
230  * <pre>
231  * <code class="language-javascript">
232  * function(bbq.ajax.AJAXRequest, XMLHttpRequest);
233  * </code>
234  * </pre>
235  *
236  * @see bbq.ajax.AJAXRequest
237  */
238 bbq.ajax.AJAXRequest.errorHandlers = {};
239 bbq.ajax.AJAXRequest.errorHandlers["error-100"] = function(request, response) {
240 	var errorMessage = new bbq.gui.error.ServerError({
241 		url: request.options.url,
242 		args: request.options.args,
243 		serverResponse: BBQUtil.urlDecode(response.getResponseHeader("X-bbq-responseMessage"))
244 	});
245 	errorMessage.appear();
246 };
247 bbq.ajax.AJAXRequest.errorHandlers["error-99"] = function(request, response) {
248 	var errorMessage = new bbq.gui.error.NotLoggedIn();
249 	errorMessage.appear();
250 };
