1 /**

  4 bbq.gui.form.transformer.StringTokeniserTransformer = new Class.create(/** @lends bbq.gui.form.transformer.StringTokeniserTransformer.prototype */ {
  5 	_delimiter: ",",
  6 	_space: " ",
  7 
  8 	/**
  9 	 * Lets you take a string, "one, two, three" and return ["one", "two", "three"]
 10 	 *
 11 	 * @constructs
 12 	 * @example
 13 	 * <pre><code class="language-javascript">
 14 	 * var field = new bbq.gui.form.TextField();
 15 	 * field.setTransformer(new bbq.gui.form.transformer.StringTokeniserTransformer());
 16 	 * field.setValue("one, two, three");
 17 	 *
 18 	 * // returns ["one", "two", "three"]
 19 	 * field.getValue();
 20 	 * </code></pre>
 21 	 * @example
 22 	 * <pre><code class="language-javascript">
 23 	 * var field = new bbq.gui.form.TextField();
 24 	 * field.setTransformer(new bbq.gui.form.transformer.StringTokeniserTransformer({
 25 	 *     delimiter: "-"
 26 	 * }));
 27 	 * field.setValue("one-two-three");
 28 	 *
 29 	 * // returns ["one", "two", "three"]
 30 	 * field.getValue();
 31 	 * </code></pre>
 32 	 * @param {Object} options
 33 	 * @param {String} options.delimiter The delimiter to split the string by
 34 	 * @param {String} options.space Used to join the split values back together along with the delimiter
 35 	 */
 36 	initialize: function(options) {
 37 		if(options) {
 38 			if(!Object.isUndefined(options.delimiter)) {
 39 				this._delimiter = options.delimiter;
 40 			}
 41 
 42 			if(!Object.isUndefined(options.space)) {
 43 				this._space = options.space;
 44 			}
 45 		}
 46 	},
 47 
 48 	/**
 49 	 * @param {Object} value
 50 	 * @returns {Object} Returns the transformed value
 51 	 */
 52 	transform: function(value) {
 53 		var output = [];
 54 
 55 		value.split(this._delimiter).each(function(part) {
 56 			part = part.strip();
 57 
 58 			if(part) {
 59 				output.push(part);
 60 			}
 61 		});
 62 
 63 		return output;
 64 	},
 65 
 66 	deTransform: function(value) {
 67 		return value.join(this._delimiter + this._space);
 68 	}
 69 });
