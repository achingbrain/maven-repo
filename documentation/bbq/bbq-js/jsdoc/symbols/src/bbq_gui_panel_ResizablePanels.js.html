  5 /**
  6  * @class bbq.gui.panel.ResizablePanels
  7  * @extends bbq.gui.panel.DualPanel
  8  */
  9 bbq.gui.panel.ResizablePanels  = new Class.create(bbq.gui.panel.DualPanel, {
 10 	_resizer: null,
 11 	_resized: false,
 12 	
 13 	/**
 14 	 * @param {Object} options
 15 	 * @example
 16 	 * Todo an example
 17 	 */
 18 	initialize: function($super, options) {
 19 		$super(options);
 20 		
 21 		this.addClass("ResizeablePanels");
 22 		this.addClass((this.layout.vertical() ? "Vertical" : "Horizontal") + "ResizeablePanels");
 23 		
 24 		this._resizer = new bbq.gui.Resizer({
 25 			vertical: this.layout.vertical()
 26 		});
 27 		
 28 		if(this.options.panel1InitialSize) {
 29 			this._resized = true;
 30 		}
 31 	},
 32 	
 33 	/**
 34 	 * @return Returns the resizer widget	
 35 	 * @type {bbq.gui.Resizer}
 36 	 */
 37 	getResizer: function() {
 38 		return this._resizer;
 39 	},
 40 	
 41 	/**
 42 	 * Draws the panels
 43 	 * 
 44 	 * @return	void
 45 	 */
 46 	render: function() {
 47 		this.empty();
 48 		
 49 		try {
 50 			var panel1 = new bbq.gui.panel.Panel();
 51 			var panel2 = new bbq.gui.panel.Panel();
 52 			
 53 			if(this.layout.vertical()) {
 54 				if(this.getTopPanel()) {
 55 					panel1 = this.getTopPanel();
 56 				}
 57 				
 58 				if(this.getBottomPanel()) {
 59 					panel2 = this.getBottomPanel();
 60 					panel2.setStyle("position", "absolute");
 61 				}
 62 			} else {
 63 				if(this.getLeftPanel()) {
 64 					panel1 = this.getLeftPanel();
 65 				}
 66 				
 67 				if(this.getRightPanel()) {
 68 					panel2 = this.getRightPanel();
 69 				}
 70 			}
 71 			
 72 			if(this._resizer != null) {
 73 				this._resizer.options.div1 = panel1.getRootNode();
 74 				this._resizer.options.div1Min = (this.options.panel1Min ? this.options.panel1Min : 50);
 75 				this._resizer.options.div2 = panel2.getRootNode();
 76 				this._resizer.options.div2Min = (this.options.panel2Min ? this.options.panel2Min : 50);
 77 				
 78 				// ensure that when dragged, the intial size specificed gets set to false
 79 				this._resizer.registerListener("onResize", function() {
 80 					this.options.panel1InitialSize = false;
 81 					this._resized = true;
 82 				}.bind(this));
 83 			}
 84 			
 85 			this.appendChild(panel1);
 86 			this.appendChild(this._resizer);
 87 			this.appendChild(panel2);
 88 		} catch(e) {
 89 			Log.error("Error rendering", e);
 90 		}
 91 	},
 92 	
 93 	_getDimensions: function(element) {
 94 		var panelSizing = Element.getDimensions(element);
 95 		
 96 		var borderWidthLeft = parseInt(element.getStyle("borderLeftWidth"))
 97 		var borderWidthRight = parseInt(element.getStyle("borderRightWidth"));
 98 		var borderWidth = 0 + (isNaN(borderWidthLeft) ? 0 : borderWidthLeft) + (isNaN(borderWidthRight) ? 0 : borderWidthRight)
 99 		
100 		panelSizing.width -= borderWidth;
101 		
102 		var borderHeightTop = parseInt(element.getStyle("borderTopWidth"))
103 		var borderHeightBottom = parseInt(element.getStyle("borderBottomWidth"));
104 		var borderHeight = 0 + (isNaN(borderHeightTop) ? 0 : borderHeightTop) + (isNaN(borderHeightBottom) ? 0 : borderHeightBottom)
105 				
106 		panelSizing.height -= borderHeight;
107 		
108 		return panelSizing;
109 	},
110 	
111 	resize: function() {
112 		try {
113 			var availableSize = {
114 				width: parseInt(this.getStyle("width")),
115 				height: parseInt(this.getStyle("height"))
116 			}
117 			
118 			if(isNaN(availableSize.width) || isNaN(availableSize.height)) {
119 				return;
120 			}
121 			
122 			var panel1 = null;
123 			var panel2 = null;
124 			
125 			if(this.layout.vertical) {
126 				panel1 = this.getTopPanel();
127 				panel2 = this.getBottomPanel();
128 			} else {
129 				panel1 = this.getRightPanel();
130 				panel2 = this.getLeftPanel();
131 			}
132 			
133 			var panel1sizing = {width: 0, height: 0};
134 			var panel2sizing = {width: 0, height: 0};
135 			var resizerSizing = {width: 0, height: 0};
136 			
137 			if(!this._resized) {
138 				// if we'v never been resized, ensure that both panels are 50% of the available size
139 				if(this.layout.vertical()) {
140 					resizerSizing.width =  availableSize.width;
141 					panel1sizing.width =  availableSize.width;
142 					panel2sizing.width =  availableSize.width;
143 					
144 					panel1sizing.height = Math.round((availableSize.height - 22) / 2);
145 					panel2sizing.height = availableSize.height - panel1sizing.height;
146 					
147 					//Log.info("height " + panel1sizing.height);
148 				} else {
149 					resizerSizing.height =  availableSize.height;
150 					panel1sizing.height =  availableSize.height;
151 					panel2sizing.height =  availableSize.height;
152 					
153 					panel1sizing.width = Math.round((availableSize.width - 22) / 2);
154 					panel2sizing.width = availableSize - panel1sizing.width;
155 				}
156 			} else {
157 				if(panel1) {
158 					panel1sizing = this._getDimensions(panel1.getRootNode());
159 				}
160 				
161 				if(panel2) {
162 					panel2sizing = this._getDimensions(panel2.getRootNode());
163 				}
164 			}
165 			
166 			if(this.layout.vertical()) {
167 				//Log.info("sizing vertical resizeable") ;
168 				this._computeSizing("width", "height", availableSize, panel1sizing, panel2sizing, resizerSizing, panel1, panel2);
169 				
170 				resizerSizing.top = panel1sizing.height;
171 				resizerSizing.left = 0;
172 				panel2sizing.top = resizerSizing.top + resizerSizing.height;
173 			} else {
174 				this._computeSizing("height", "width", availableSize, panel1sizing, panel2sizing, resizerSizing);
175 				
176 				resizerSizing.top = 0;
177 				resizerSizing.left = panel1sizing.width;
178 				panel2sizing.left = resizerSizing.left + resizerSizing.width;
179 			}
180 			
181 			this._applySizing(panel1, panel1sizing);
182 			this._applySizing(panel2, panel2sizing);
183 			this._applySizing(this.getResizer(), resizerSizing);
184 			
185 			if(panel1 && Object.isFunction(panel1.resize)) {
186 				panel1.resize();
187 			}
188 			
189 			if(panel2 && Object.isFunction(panel2.resize)) {
190 				panel2.resize();
191 			}
192 			
193 			this.notifyListeners("onResize");
194 		} catch(e) {
195 			Log.error("Error resizing", e);
196 		}
197 	},

213 	_applySizing: function(panel, sizing) {
214 		if(panel) {
215 			for(var key in sizing) {
216 				if(!isNaN(sizing[key]) && sizing[key] > 0) {
217 					panel.setStyle(key, sizing[key] + "px");
218 				}
219 			}
220 			
221 			panel.resize();
222 		}
223 	},
224 	
225 	_computeSizing: function(dim1, dim2, availableSize, panel1sizing, panel2sizing, resizerSizing, panel1, panel2) {
226 		// increase non-directional size to the maximum - eg. if this.options.direction == "vertical" make width as wide as possible
227 		var maxDim = parseInt(this.getStyle(dim1));
228 		
229 		panel1sizing[dim1] = maxDim;
230 		panel2sizing[dim1] = maxDim;
231 		
232 		// make resizer 22px wide and expand height or vise versa
233 		resizerSizing[dim1] = maxDim;
234 		resizerSizing[dim2] = 22;
235 		
236 		if(panel1 && panel1.hasMinimisedChild && panel1.hasMinimisedChild()) {
237 			// if panel is minimised, set the size of the panel to reflect that
238 			panel1sizing[dim2] = panel1.getMinimisedSize();
239 		} else if(this.options.panel1Min && panel1sizing[dim2] < this.options.panel1Min) {
240 			// if minimum specified, enforce it
241 			panel1sizing[dim2] = this.options.panel1Min;
242 		}
243 		
244 		if(panel2 && panel2.hasMinimisedChild && panel2.hasMinimisedChild()) {
245 			panel2sizing[dim2] = panel2.getMinimisedSize();
246 			
247 			// available size minus dim of minimised panel minus dim of resizer
248 			panel1sizing[dim2] = availableSize[dim2] - panel2.getMinimisedSize() - resizerSizing[dim2];
249 		}
250 		
251 		// if initial size specified, enforce it.  N.B - this.options.panel1InitialSize gets set to false when the dragger is first dragged
252 		if(this.options.panel1InitialSize) {
253 			var dim = this.getRootNode()["get" + dim2.capitalize()]();
254 			
255 			if(dim - this.options.panel1InitialSize < panel2sizing[dim2]) {
256 				panel1sizing[dim2] = this.options.panel1InitialSize/2;
257 			} else {
258 				panel1sizing[dim2] = this.options.panel1InitialSize;
259 			}
260 		}
261 		
262 		panel2sizing[dim2] = availableSize[dim2] - panel1sizing[dim2] - resizerSizing[dim2];
263 		
264 		// if minimum specified, enforce it
265 		if(this.options.panel2Min && panel2sizing[dim2] < this.options.panel2Min) {
266 			panel2sizing[dim2] = this.options.panel2Min;
267 			
268 			panel1sizing[dim2] = availableSize[dim2] - panel2sizing[dim2] - resizerSizing[dim2];
269 		}
270 	},
271 	
272 	setDisabled: function(disabled) {
273 		this._resizer.setDisabled(disabled);
