 19 	/** find out which broswer we are running */
 20 	detect: function() {
 21 
 22 		if(window.opera) {
 23 			Browser.Opera = true;
 24 
 25 			if(window.XMLHttpRequest) {
 26 				Browser.version = 8;
 27 			} else if(document.getElementById) {
 28 				Browser.version = 7;
 29 			}
 30 		} else if((document.childNodes) && (!document.all) && (!navigator.taintEnabled) && (!navigator.accentColorName)) {
 31 			if(window.chrome) {
 32 				Browser.Chrome = true;
 33 			} else {
 34 				Browser.Safari = true;
 35 
 36 				if(window.devicePixelRatio) {
 37 					Browser.version = 3;
 38 				} else {
 39 					Browser.version = 2; // probably
 40 				}
 41 			}
 42 		} else if(document.getElementById && !document.all) {
 43 			Browser.Mozilla = true;
 44 
 45 			if(navigator.registerProtocolHandler) {
 46 				Browser.version = 3;
 47 			} else if(window.Iterator) {
 48 				Browser.version = 2;
 49 			} else if(Array.every) {
 50 				Browser.version = 1.5;
 51 			} else if(window.getComputedStyle) {
 52 				Browser.version = 1;
 53 			}
 54 		} else if(document.getElementById && document.all && window.ActiveXObject) {
 55 			Browser.InternetExplorer = true;
 56 
 57 			if(document.querySelectorAll) {
 58 				Browser.version = 8;
 59 			} else if(window.XMLHttpRequest) {
 60 				Browser.version = 7;
 61 			} else if(document.compatMode && document.all) {
 62 				Browser.version = 6;
 63 			} else if(window.createPopup) {
 64 				Browser.version = 5.5;
 65 			} else if(window.attachEvent) {
 66 				Browser.version = 5;
 67 			} else if(document.all) {
 68 				Browser.version = 4;
 69 			}
 70 		}
 71 
 72 		// does it support AJAX?
 73 		if(window.XMLHttpRequest) {
 74 			Browser.Ajax = true;
 75 		} else if(window.ActiveXObject) {
 76 			try {
 77 				requestObject=new ActiveXObject("Msxml2.XMLHTTP");
 78 				this.Ajax = true;
 79 			} catch (e) {
 80 				try {
 81 					requestObject=new ActiveXObject("Microsoft.XMLHTTP");
 82 					Browser.Ajax = true;
 83 				} catch (oc) {
 84 
 85 				}
 86 			}
 87 		}
 88 
 89 		// how about the DOM?
 90 		if(document.getElementById) {
 91 			Browser.DOM = true;
 92 		}
 93 
 94 		// and Flash?
 95 		if(Browser.InternetExplorer) {
 96 			try {
 97 				var obj = new ActiveXObject("ShockwaveFlash.ShockwaveFlash");
 98 
 99 				if(!obj.activeXError) {
100 					Browser.Flash = true;
101 				}
102 			} catch(e) {
103 				Browser.Flash = false;
104 			}
105 		} else {
106 			try {
107 				if (navigator.mimeTypes && navigator.mimeTypes['application/x-shockwave-flash'] && navigator.mimeTypes['application/x-shockwave-flash'].enabledPlugin) {
108 					Browser.Flash = true;
109 				}
110 			} catch(e) {
111 				if (navigator.mimeTypes ['application/x-shockwave-flash'] != undefined) {
112 					Browser.Flash = true;
113 				}
114 			}
115 		}
116 
117 		DOMUtil.addClass($$("html")[0], Browser.getBrowserName());
118 
119 		// form support
120 		var input = document.createElement("input");
121 		Browser.forms.placeholderText = ("placeholder" in input);
122 
123 		// check HTML5 input type support
124 		Browser.forms.types = {};
125 
126 		["search", "tel", "url", "email", "datetime", "date", "month", "week", "time", "datetime-local", "number", "color", "range"].each(function(type) {
127 			input.setAttribute("type", type);
128 
129 			Browser.forms.types[type] = (input.type == type);
130 		});
131 	},

137 			return "Opera";
138 		} else if(Browser.Safari) {
139 			return "Safari";
140 		} else if(Browser.Mozilla) {
141 			return "Mozilla";
142 		} else if(Browser.Chrome) {
143 			return "Chrome";
144 		}

146 		return "No idea";
147 	}
