190 	_generateRow: function(index, entity) {
191 		var row = document.createElement("tr");
192 				
193 		if(!entity) { // invalid item
194 			row = this._handleInvalidItem(row, index);
195 		} else if(this.options.createRow instanceof Function) { // custom item
196 			row = this.options.createRow(row, entity, index);

201 			if(entity[this.options.entityID] == this._selectedElementID) {
202 				DOMUtil.addClass(row, "down");
203 				this._selectedEntity.row = row;
204 				this._selectedEntity.entity = entity;
205 			}
206 		}
207 		
208 		if(index % 2 == 0) {
209 			DOMUtil.addClass(row, "oddRow");
210 		} else {
211 			DOMUtil.addClass(row, "evenRow");
212 		}

217 	_handleValidItem: function(row, rowIndex, entity) {
218 		var cData = this.options.columnData;
219 		for(var n = 0, nCount=cData.length; n < nCount; n++) { // process one entity (eg. one row)
220 			var cell = document.createElement("td");
221 			
222 			if(cData[n].iconColumn) {
223 				DOMUtil.addClass(cell, "ScrollLoadTableIconColumn");
224 			}
225 			
226 			row.appendChild(cell);
227 			DOMUtil.addClass(cell, "column" + n);
228 			
229 			if(cData[n]["text"] instanceof Function) {
230 				var cellContent = cData[n]["text"].call(this, entity, n, cell);
231 				
232 				if(cellContent) {
233 					if(cellContent.appendTo) {
234 						// passed GUIWidget
235 						cellContent.appendTo(cell);
236 					} else {
237 							this._processCallbacks(cellContent, entity, n);
238 							cell.appendChild(cellContent.appendChild ? cellContent : document.createTextNode(cellContent));
239 					}
240 				}
241 			} else {
242 				var text = this.getText(entity, cData[n]["text"]);
243 				cell.appendChild(document.createTextNode(text ? text : " "));
244 			}
245 			
246 			cell.owningEntity = entity;
247 		}
248 		
249 		row.owningEntity = entity;
250 		
251 		return row;

258 	/**
259 	 * Stores onclick and ondblclick so we can call them after setting the down state of the row
260 	 * 
261 	 * @private
262 	 * @param	{Node}	node	A node that we suspect has onclick, etc attributes defined on it
263 	 * @param	{Object}	entity	The entity that the row was made from that the node is from
264 	 */
265 	_processCallbacks: function(node, entity, columnIndex) {
266 		if(node) {
267 			try {
268 				node.columnIndex = columnIndex;
269 				
270 				for(var key in {onclick: "", ondblclick: ""}) { // check for onclick and ondblclick
271 					if(node[key]) { // if it's defined
272 						if(!this._callBacks[entity[this.options.entityID]]) { // make sure the callback holder is defined
273 							this._callBacks[entity[this.options.entityID]] = {};
274 						}
275 						
276 						if(!this._callBacks[entity[this.options.entityID]][columnIndex]) { // make sure the callback holder is defined
277 							this._callBacks[entity[this.options.entityID]][columnIndex] = {};
278 						}
279 						
280 						this._callBacks[entity[this.options.entityID]][columnIndex][key] = node[key]; // store the callback for calling later
281 						node[key] = this._boundCallBacks[key]; // replace it with our generic callback
282 					}
283 				}
284 			} catch(e) {
285 				// this will fail when something that isn't a DOM node is passed to be the contents of a cell.
286 				//  since we are only concerned with anchor tags and the like, we don't mind so swallow the exception
287 			}
288 		}

310 	 * @private
311 	 */
312 	_processEvent: function(node, eventName) {
313 		var entity = node.parentNode.owningEntity;
314 		var columnIndex = node.columnIndex;
315 		
316 		if(this._callBacks[entity[this.options.entityID]] && this._callBacks[entity[this.options.entityID]][columnIndex] && this._callBacks[entity[this.options.entityID]][columnIndex][eventName] && this._callBacks[entity[this.options.entityID]][columnIndex][eventName] instanceof Function) {
317 			// if the callback returns false, do not allow the current row to become selected
318 			if(this._callBacks[entity[this.options.entityID]][columnIndex][eventName].call(this, entity) === false) {
319 				return;
320 			}
321 		}
322 		
323 		if(!this.options.columnData[columnIndex].ignoreDownState) {
324 			this._setItemDown(node);
325 			this._selectedElementID = entity[this.options.entityID];
326 			this.render();
327 		}
328 		
329 		if(FocusWatcher) {
330 			FocusWatcher.setKeypressCallbackObject(this);
331 		}

360 	/**
361 	 * Returns either the value of the entity property using text as a key or text itself.
362 	 * 
363 	 * So getText({name: "alex"}, "name") would return "alex" but getText({name: "alex"}, "age") would return "age".  
364 	 * 
365 	 * If entityID has . characters in it, this method will step down into entity to find the key, so 
366 	 * getText({name: "alex", innerObject : {name: "bob"}}, "innerObject.bob") would return "bob"
367 	 * 
368 	 * @param	{Object}		entity
369 	 * @param	{string}		text
370 	 */
371 	getText: function(entity, text) {
372 		if(text.search(/(\.)/) > -1) {
373 			return this._findIt(entity, text.split("."));
374 		} else if(entity[text]) {
375 			return entity[text];
376 		}
377 		
378 		return text;
379 	},

382 	 * Returns the ID of the entity.  If entityID has . characters in it, this method will step down into entity to find the key.
383 	 */
384 	getEntityID: function(entity, entityID) {
385 		if(entityID.search(/\./)) {
386 			return this._findIt(entity, entityID.split("."));
387 		}

399 	_findIt: function(entity, keys) {

444 	/**
445 	 * Sorts the table entries by this.sortedByColumnIndex or a specified column - 
446 	 * 0 = first column, 1 = second column, etc
447 	 * 
448 	 * @param	{integer}	index
449 	 * @return	void
450 	 */
451 	sortByIndex: function(index) {
452 		this.notifyListeners("onBeforeSort");
453 		
454 		if(index == -1) {
455 			return;
456 		}
457 		
458 		if(this.sortedByColumnIndex == index) {
459 			this.sortedByDirection = (this.sortedByDirection == "DESC" ? "ASC" : "DESC");
460 		} else {
461 			this.sortedByDirection = "DESC";
462 		}
463 		
464 		//Log.info("sorting by " + this.sortedByDirection);
465 		
466 		this.sortedByColumnIndex = index;
467 		this.currentSortKey = this.options.columnData[this.sortedByColumnIndex]["sortField"];
468 		
469 		var entityGroup = this.getEntityGroup();
470 		
471 		if(this.sortedByDirection == "ASC") {
472 			entityGroup.sort(this.ascSortFunction.bind(this));
473 		} else {
474 			entityGroup.sort(this.descSortFunction.bind(this));
475 		}
476 		
477 		this.render();
478 		this.notifyListeners("onAfterSort");
479 		this.notifyListeners("onSort");
480 	},

541 		var nextEntity = false;
542 		var foundEntity = false;
543 		
544 		// process keypress
545 		if(event.keyCode == Event.KEY_DOWN) { // if we can go one further down the list, do it but skip the contents of closed folders
546 			$A(this.getRootNode().getElementsByTagName("tr")).each(function(row){
547 				if(foundEntity && !nextEntity) {
548 					nextEntity = row.owningEntity;
549 				}
550 				
551 				if(this.isSelectedEntity(row.owningEntity)) {
552 					foundEntity = true;
553 				}
554 			}.bind(this));
555 		} else if(event.keyCode == Event.KEY_UP) {  // if we can go one further up the list, do it but skip the contents of closed folders
556 			$A(this.getRootNode().getElementsByTagName("tr")).each(function(row){
557 				if(this.isSelectedEntity(row.owningEntity)) {
558 					foundEntity = true;
559 					throw $break;
560 				}
561 				
562 				nextEntity = row.owningEntity;
563 			}.bind(this));
564 		}
565 		
566 		if(nextEntity) {
567 			this.setDown(nextEntity);
568 			
569 			if(this._callBacks[nextEntity[this.options.entityID]] && this._callBacks[nextEntity[this.options.entityID]][0] && this._callBacks[nextEntity[this.options.entityID]][0]["onclick"] && this._callBacks[nextEntity[this.options.entityID]][0]["onclick"] instanceof Function) {
570 				this._callBacks[nextEntity[this.options.entityID]][0]["onclick"].call(this, nextEntity);
571 			}
572 		}
573 		
574 		return true;
575 	},
