1 include(bbq.gui.GUIWidget);
  2 
  3 /**
  4  * @class bbq.gui.FileUploadWidget
  5  * @extends bbq.gui.GUIWidget
  6  */
  7 bbq.gui.FileUploadWidget = new Class.create(bbq.gui.GUIWidget, {
  8 	
  9 	_iFrame: null,
 10 	_uploadInterval: null,
 11 	_uploadData: null,
 12 	
 13 	/**
 14 	 * Supoprts the following options:
 15 	 * 
 16 	 * options: {
 17 	 * 		uploadTo: String		// Where to upload the file to
 18 	 * 		extensions: Array		// Optional - an array of allowed file extensions
 19 	 * }
 20 	 * 
 21 	 * Supports the following callbacks:
 22 	 * 
 23 	 * onUploadError
 24 	 * onUploadComplete
 25 	 * onInvalidFileExtension
 26 	 */
 27 	initialize: function($super, options) {
 28 		$super(options);
 29 		
 30 		this.setRootNode("div");
 31 		this.addClass("FileUploadWidget");
 32 		
 33 		var input = DOMUtil.createElement("input", {type: "file", name: "assetFile"});
 34 		var form = DOMUtil.createTextElement("form", input, {method: "post", enctype: "multipart/form-data"});
 35 		DOMUtil.setStyle(form, "width", "100%");
 36 		
 37 		this._iFrame = DOMUtil.createElement("iframe");
 38 		
 39 		Event.observe(this._iFrame, "load", function() {
 40 			var iframeDoc = DOMUtil.getIFrameDocument(this._iFrame);
 41 			iframeDoc.body.appendChild(form);
 42 			iframeDoc.body.style.margin = "0";
 43 			iframeDoc.body.style.padding = "0 10px";
 44 		}.bind(this));
 45 		
 46 		Event.observe(input, "change", function() {
 47 			this.notifyListeners("onFileChosen");
 48 		}.bind(this));
 49 		
 50 		this.appendChild(this._iFrame);
 51 	},
 52 	
 53 	appendBefore: function(parentNode, beforeNode) {
 54 		parentNode.insertBefore(this.getRootNode(), beforeNode);
 55 	},
 56 
 57 	beginUpload: function() {
 58 		if(!this._uploadInterval) {
 59 			var form = DOMUtil.getIFrameDocument(this._iFrame).forms[0];
 60 			
 61 			if(form && form.assetFile) {
 62 				var fileLocation = form.assetFile.value;
 63 				
 64 				if(fileLocation) {
 65 									
 66 					var extension = fileLocation.split(".").pop().toLowerCase();
 67 					Log.info(extension);
 68 					if(this.options.extensions instanceof Array && this.options.extensions.indexOf(extension) == -1) {
 69 						this.notifyListeners("onInvalidFileExtension");
 70 						return;
 71 					}
 72 					
 73 					form.action = this.options.uploadTo;
 74 					
 75 					if(form.action.substring(0, 1) == "/") {
 76 						// prepend http(s)://domain
 77 						form.action = document.location.protocol + "//" + document.location.host  + form.action;
 78 					}
 79 					
 80 					Log.info("uploading to " + form.action);
 81 					
 82 					form.submit();
 83 					
 84 					this._uploadInterval = setInterval(this._getUploadData.bind(this), 2000);
 85 				}
 86 			}
 87 		}
 88 	},
 89 	
 90 	cancelUpload: function() {
 91 		clearInterval(this._uploadInterval);
 92 	},
 93 	
 94 	_getUploadData: function() {
 95 		if(this._iFrame.contentWindow && this._iFrame.contentWindow.getData) {
 96 			clearInterval(this._uploadInterval);
 97 			var iframeWindow = this._iFrame.contentWindow;
 98 			
 99 			this._uploadData = iframeWindow.getData();
100 			
101 			if(iframeWindow.fileError) {
102 				this.notifyListeners("onUploadError");
103 			} else {
104 				this.notifyListeners("onUploadComplete");
105 			}
106 		}
107 	},
108 	
109 	getUploadData: function() {
110 		return this._uploadData;
111 	},
112 	
113 	getUploadError: function() {
114 		return this._uploadData;
115 	},
116 	
117 	getFileTransferID: function() {
118 		var data = this.getUploadData();
119 		return data["filetransferid"];
120 	}
121 });
