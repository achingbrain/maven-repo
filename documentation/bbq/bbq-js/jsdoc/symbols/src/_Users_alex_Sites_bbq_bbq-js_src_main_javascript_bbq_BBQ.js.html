  6 /**
  7  * Creates the bbq.page.Page object for the current page.  Should be invoked in a document onLoad event.
  8  * @param className
  9  * @param args
 10  */
 11 function init(className, args) {
 12 	if(className) {
 13 		var parts = className.split(".");
 14 		var classToInstantiate = window;
 15 		
 16 		for(var i = 0, iCount=parts.length; i < iCount; i++) {
 17 			if(classToInstantiate[parts[i]]) {
 18 				classToInstantiate = classToInstantiate[parts[i]];
 19 			} else {
 20 				alert("Fatal error - could not load " + (i < (iCount - 1) ? "package" : "class") + " " + parts[i] + (i < (iCount - 1) ? " - undefined in namespace" : " from qualified name " + className));
 21 				return;
 22 			}
 23 		}
 24 		
 25 		currentPage = new classToInstantiate(args);
 26 	}
 27 }
