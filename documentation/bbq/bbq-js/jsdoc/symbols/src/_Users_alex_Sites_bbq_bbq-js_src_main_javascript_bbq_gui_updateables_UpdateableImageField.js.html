1 include(bbq.gui.updateables.UpdateableField);
  2 include(bbq.gui.FileUploaderFlash);

  5 /**
  6  * @class bbq.gui.updateables.UpdateableImageField
  7  * @extends bbq.gui.updateables.UpdateableField
  8  */
  9 bbq.gui.updateables.UpdateableImageField = new Class.create(bbq.gui.updateables.UpdateableField, {
 10 	
 11 	_oldImage: null,
 12 	_currentImage: null,
 13 	_chooseButtonHolder: null,
 14 	_currentImageNode: null,
 15 	_fileUploaded: null,
 16 	
 17 	/**
 18 	 * @param {Object} options
 19 	 * @example
 20 	 * Supports the following options:
 21 	 * 
 22 	 * options: {
 23 	 * 		getImage: function						// should return the path to the image
 24 	 * 		uploadTo										// Where to POST the image
 25 	 * 		language: Object							// passed to a bbq.gui.FileUploader object
 26 	 * 		deleteButtonText							// Text displayed on the delete button - defaults to "Remove"
 27 	 * }
 28 	 * 
 29 	 * Supports the following callbacks
 30 	 * 
 31 	 * onFileDeleted
 32 	 * onFileChosen
 33 	 */
 34 	initialize: function($super, options) {
 35 		$super(options);
 36 		
 37 		this.setRootNode("div");
 38 		this.addClass("UpdateableImageField");
 39 		
 40 		if(this._getImage()) {
 41 			this._oldImage = this._getImage();
 42 			this._currentImage = this._getImage();
 43 			
 44 			this.addClass("UpdateableImageField_WithImage");
 45 		}
 46 		
 47 		this._setCurrentValue(null);
 48 		
 49 		this._chooseImageButtonText = Language.get("bbq.gui.updateables.UpdateableImageField.uploadfile");
 50 		this._deleteImageButtonText = this.options.deleteButtonText ? this.options.deleteButtonText : Language.get("bbq.gui.updateables.UpdateableImageField.remove");
 51 	},
 52 	
 53 	_updateInitialValues: function() {
 54 		this._currentImage = this._getImage();
 55 		this.render();
 56 	},
 57 	
 58 	_getImage: function() {
 59 		if(this.options.getImage && this.options.getImage instanceof Function) {
 60 			var entity = this.options.propertyDisplay ? this.options.propertyDisplay.entity : null;
 61 			
 62 			return this.options.getImage(entity);
 63 		}
 64 		
 65 		return false;
 66 	},
 67 	
 68 	setPropertyDisplay: function(key, entity) {
 69 		if(this._currentImageNode) {
 70 			this._currentImageNode.onload = null;
 71 			this._currentImageNode = null;
 72 		}
 73 		
 74 		bbq.gui.updateables.UpdateableField.prototype.setPropertyDisplay.apply(this, arguments);
 75 	},
 76 	
 77 	_createImage: function() {
 78 		this.addClass("UpdateableImageField_WithImage");
 79 		this.setStyle("paddingTop", "0px");
 80 		this.setStyle("paddingBottom", "0px");
 81 		
 82 		if(!this._currentImageNode) {
 83 			if(!this._fileUploaded && this.options.propertyDisplay) {
 84 				Log.info("have property display");
 85 				this._currentImageNode = this.options.propertyDisplay.entity.getPropertyDisplay({
 86 					property: this.options.propertyDisplay.key, 
 87 					createNode: function() {
 88 						return DOMUtil.createElement("img");
 89 					},
 90 					updateNode: function(node, property) {
 91 						node.src = this._currentImage;
 92 					}.bind(this)
 93 				});
 94 			} else {
 95 				Log.info("creating from passed value");
 96 				this._currentImageNode = DOMUtil.createElement("img", {src: this._currentImage, alt: ""});
 97 			}
 98 			
 99 			this._currentImageNode.onload = function() {
100 				var height = this._currentImageNode.height;
101 				var width = this._currentImageNode.width;
102 				
103 				if(height == 0 && width == 0) {
104 					return;
105 				}
106 				
107 				if(width < 128) {
108 					DOMUtil.setStyle(this._currentImageNode, "marginLeft", Math.round((128 - width)/2) + "px");
109 				}
110 				
111 				if(height < 128) {
112 					var padding = Math.round((128 - height)/2) + "px";
113 					
114 					DOMUtil.setStyle(this._currentImageNode, "marginTop", padding);
115 					DOMUtil.setStyle(this._currentImageNode, "marginBottom", padding);
116 				}
117 			}.bind(this);
118 		}
119 		
120 		this.appendChild(this._currentImageNode);
121 	},
122 	
123 	createEditField: function() {
124 		this.empty();
125 		
126 		if(this._currentImage) {
127 			this._createImage();
128 		} else {
129 			this.removeClass("UpdateableImageField_WithImage");
130 			this.setStyle("paddingTop", "128px");
131 		}
132 		
133 		this._chooseButtonHolder = new bbq.gui.button.ButtonHolder({ignoreKeyPresses: true});
134 		
135 		if(this._currentImage) {
136 			this._chooseButtonHolder.addButton(new bbq.gui.button.NativeButton({buttonText: this._deleteImageButtonText, onclick: this._deleteFile.bind(this)}));
137 		}
138 		
139 		this._chooseButtonHolder.addButton(new bbq.gui.button.NativeButton({buttonText: this._chooseImageButtonText, onclick: this.options.uploadFunction ? this.options.uploadFunction.bind(this, this) : this._showFilePickerWindow.bind(this)}));
140 		
141 		this.appendChild(this._chooseButtonHolder);
142 	},
143 	
144 	createViewField: function() {
145 		this.empty();
146 		
147 		if(this._currentImage) {
148 			this._createImage();
149 		} else {
150 			this.removeClass("UpdateableImageField_WithImage");
151 			this.setStyle("paddingTop", "128px");
152 		}
153 	},
154 	
155 	_deleteFile: function() {
156 		this._setCurrentValue(-1);
157 		
158 		this._fileUploaded = false;
159 		this._currentImage = false;
160 		this.render();
161 		this.notifyListeners("onFileRemoved");
162 		this.notifyListeners("onValueChanged");
163 	},
164 	
165 	_showFilePickerWindow: function() {
166 		var filePicker = new deadline.gui.uploads.FileUploader({
167 			title: "Upload image",
168 			description: "Images",
169 			extensions: ["jpg", "gif", "png"],
170 			multiple: this.options.multiple
171 		});
172 		filePicker.registerListener("onFilesUploaded", this._filesUploaded.bind(this));
173 		currentPage.addModalLayer(filePicker);
174 	},
175 	
176 	_filesUploaded: function(imagePicker) {
177 		var files = imagePicker.getFiles();
178 		
179 		if(files.length > 0) {
180 			this._fileUploaded = true;
181 			this._currentImageNode = false;
182 			
183 			this._oldImage = this._currentImage;
184 			
185 			if(files.length == 1) {
186 				this._currentImage = "/backend/uploads/getTemporaryThumbnail?id=" + files[0].filetransferid;
187 			} else {
188 				//this._currentImage = "/img/icons/128x128/content-types/multiple.png";
189 				this._currentImage = "/backend/uploads/getTemporaryThumbnail?id=" + files[0].filetransferid;
190 			}
191 			
192 			if(this.options.multiple) {
193 				this._setCurrentValue(files.pluck("filetransferid"));
194 			} else {
195 				this._setCurrentValue(files[0].filetransferid);
196 			}
197 			
198 			this.render();
199 		}
200 		
201 		this.notifyListeners("onFileUploaded", files);
202 		this.notifyListeners("onValueChanged");
203 	},
204 	
205 	cancelEdit: function() {
206 		this._setCurrentValue(null);
207 		this._currentImage = this._oldImage;
208 		this._fileUploaded = false;
209 		this.createViewField();
210 	},
211 	
212 	saveEdit: function() {
213 		if(this._getCurrentValue() == -1) {
214 			this.notifyListeners("onFileDeleted");
215 		} else {
216 			if(this.options.propertyDisplay) {
217 				// set current value to null - old value was previously uploaded file so set to null so we don't try to overwrite it with one that now will not exist
218 				this.currentValue = null;
219 			}
220 		}
221 		
222 		this._oldImage = this._getImage();
223 		this._currentImage = this._getImage();
224 	},
225 	
226 	getValue: function(supressError) {
227 		return this._getCurrentValue(supressError);
228 	},
229 	
230 	setDisabled: function($super, disabled) {
231 		$super(disabled);
232 		
233 		if(this._chooseButtonHolder) {
234 			this._chooseButtonHolder.setDisabled(disabled);
235 		}
236 	}
237 });
