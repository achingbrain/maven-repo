1 /**
  2  * @class bbq.gui.Theme 
  3  */
  4 Theme = {
  5 	_styleSheets: {},
  6 	_styleSheetNodes: [],
  7 	_theme: "",
  8 	
  9 	addStyleSheet: function(sheet, media) {
 10 		if(typeof(Theme._styleSheets[sheet]) == "undefined") {
 11 			sheet = sheet.replace(/\./g, "/");
 12 			
 13 			if(!Theme._styleSheets[sheet]) {
 14 				Theme._styleSheets[sheet] = sheet;
 15 				Theme._styleSheetNodes.push(Theme._addSheetToHead(sheet, media));
 16 			}
 17 		}
 18 	},
 19 	
 20 	setTheme: function(theme) {
 21 		Theme._theme = theme;
 22 		Theme._updateStyleSheets;
 23 	},
 24 	
 25 	_updateStyleSheets: function() {
 26 		var head = document.getElementsByTagName("head")[0];
 27 		
 28 		Theme._styleSheetNodes.each(function(node) {
 29 			head.removeChild(node);
 30 		});
 31 		
 32 		Theme._styleSheets.each(function(sheet) {
 33 			Theme._addSheetToHead(sheet);
 34 		});
 35 	},
 36 	
 37 	_addSheetToHead: function(sheet, m) {
 38 		if (m == "undefined") {
 39 			m = "all";
 40 		}
 41 		Log.info("media: " + m);
 42 		var node = DOMUtil.createElement("link", {
 43 			media: m,
 44 			type: "text/css",
 45 			rel: "stylesheet",
 46 			href: "/css/theme" + Theme._theme + "/" + sheet + "/style.css"
 47 		});
 48 		
 49 		var head = document.getElementsByTagName("head")[0];
 50 		head.insertBefore(node, head.getElementsByTagName("script")[0]);
 51 		
 52 		return node;
 53 	}
 54 };
