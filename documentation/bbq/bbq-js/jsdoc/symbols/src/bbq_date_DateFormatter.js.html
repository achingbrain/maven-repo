1 /**
  2  * DateFormatter represents the date format
  3  *
  4  * Cribbed from http://blog.stevenlevithan.com/archives/date-time-format
  5  *
  6  * (c) 2007-2009 Steven Levithan <stevenlevithan.com>
  7  * MIT license
  8  *
  9  * Includes enhancements by Scott Trenda <scott.trenda.net>
 10  * and Kris Kowal <cixar.com/~kris.kowal/>
 11  *
 12  * Accepts a date, a mask, or a date and a mask.
 13  * Returns a formatted version of the given date.
 14  * The date defaults to the current date/time.
 15  * The mask defaults to dateFormat.masks.default.
 16  *
 17  * <pre><code class="language-javascript">
 18  * var date = new Date();
 19  *
 20  * // default
 21  * // Mon Jan 9 2012 11:38:00
 22  * DateFormatter.format(date);
 23  *
 24  * // just the year in long form
 25  * // 2012
 26  * DateFormatter.format(date, "yyyy");
 27  *
 28  * // Use built in ISO date mask
 29  * // 2012-01-09
 30  * DateFormatter.format(date, DateFormatter.masks.isoDate);
 31  * </code></pre>
 32  *
 33  * @class DateFormatter
 34  */
 35 DateFormatter = {
 36 	_token: /d{1,4}|m{1,4}|yy(?:yy)?|([HhMsTt])\1?|[LloSZ]|"[^"]*"|'[^']*'/g,
 37 	_timezone: /\b(?:[PMCEA][SDP]T|(?:Pacific|Mountain|Central|Eastern|Atlantic) (?:Standard|Daylight|Prevailing) Time|(?:GMT|UTC)(?:[-+]\d{4})?)\b/g,
 38 	_timezoneClip: /[^-+\dA-Z]/g,
 39 
 40 	/**
 41 	 * Various built in date format masks.  Pass one of these as the
 42 	 * second argument to DateFormatter#format
 43 	 *
 44 	 * @type {object}
 45 	 */
 46 	masks: {
 47 		"default":				"ddd mmm dd yyyy HH:MM:ss",
 48 		shortDate:			"m/d/yy",
 49 		mediumDate:		"mmm d, yyyy",
 50 		longDate:				"mmmm d, yyyy",
 51 		fullDate:				"dddd, mmmm d, yyyy",
 52 		shortTime:			"h:MM TT",
 53 		mediumTime:		"h:MM:ss TT",
 54 		longTime:				"h:MM:ss TT Z",
 55 		isoDate:				"yyyy-mm-dd",
 56 		isoTime:				"HH:MM:ss",
 57 		isoDateTime:		"yyyy-mm-dd'T'HH:MM:ss",
 58 		isoUtcDateTime:	"UTC:yyyy-mm-dd'T'HH:MM:ss'Z'"
 59 	},
 60 
 61 	/**
 62 	 * Formats a given date using the passed mask and formatter.
 63 	 *
 64 	 * If you do not specify a formatter, this function will return the date formatted
 65 	 * as a string using the passed mask.
 66 	 *
 67 	 * If you do not pass a mask it uses DateFormatter.masks.default.
 68 	 *
 69 	 * If you do not pass a date it uses the current date.
 70 	 *
 71 	 * @param date
 72 	 * @param mask
 73 	 * @param formatter
 74 	 * @param formatter.format(date, mask, utc, regex)
 75 	 */
 76 	format: function(date, mask, formatter) {
 77 		// Passing date through Date applies Date.parse, if necessary
 78 		date = date ? new Date(date) : new Date();
 79 
 80 		formatter = formatter || DateFormatter._stringFormatter;
 81 
 82 		// blech, get out
 83 		if(isNaN(date)) {
 84 			throw SyntaxError("invalid date");
 85 		}
 86 
 87 		mask = new String(DateFormatter.masks[mask] || mask || DateFormatter.masks["default"]);
 88 
 89 		var utc = false;
 90 
 91 		// Allow setting the utc argument via the mask
 92 		if(mask.slice(0, 4) == "UTC:") {
 93 			mask = mask.slice(4);
 94 			utc = true;
 95 		}
 96 
 97 		return formatter.format(date, mask, utc, DateFormatter._token);
 98 	}
 99 }
100 
101 DateFormatter._stringFormatter = {
102 	format: function(date, mask, utc, regex) {
103 		return mask.replace(regex, function ($0) {
104 			return this[$0] ? this[$0](date, utc) : $0.slice(1, $0.length - 1);
105 		}.bind(DateFormatter._stringFormatter));
106 	}.bind(DateFormatter._stringFormatter),
107 
108 	d: function(date, utc) {
109 		return this._get(date, utc, "Date");
110 	}.bind(DateFormatter._stringFormatter),
111 
112 	dd: function(date, utc) {
113 		return this.d(date, utc).toPaddedString(2);
114 	}.bind(DateFormatter._stringFormatter),
115 
116 	ddd: function(date, utc) {
117 		return Language.get("bbq.date.DateFormatter.short.days" + this.D(date, utc));
118 	}.bind(DateFormatter._stringFormatter),
119 
120 	dddd: function(date, utc) {
121 		return Language.get("bbq.date.DateFormatter.long.days" + this.D(date, utc));
122 	}.bind(DateFormatter._stringFormatter),
123 
124 	D: function(date, utc) {
125 		return this._get(date, utc, "Day");
126 	}.bind(DateFormatter._stringFormatter),
127 
128 	m: function(date, utc) {
129 		return this._get(date, utc, "Month") + 1;
130 	}.bind(DateFormatter._stringFormatter),
131 
132 	mm: function(date, utc) {
133 		return this.m(date, utc).toPaddedString(2);
134 	}.bind(DateFormatter._stringFormatter),
135 
136 	mmm: function(date, utc) {
137 		return Language.get("bbq.date.DateFormatter.short.months" + (this.m(date, utc) - 1));
138 	}.bind(DateFormatter._stringFormatter),
139 
140 	mmmm: function(date, utc) {
141 		return Language.get("bbq.date.DateFormatter.long.months" + (this.m(date, utc) - 1));
142 	}.bind(DateFormatter._stringFormatter),
143 
144 	_y: function(date, utc) {
145 		return this._get(date, utc, "FullYear");
146 	}.bind(DateFormatter._stringFormatter),
147 
148 	yy: function(date, utc) {
149 		return String(this._y(date, utc)).slice(2);
150 	}.bind(DateFormatter._stringFormatter),
151 
152 	yyyy: function(date, utc) {
153 		return this._y(date, utc);
154 	}.bind(DateFormatter._stringFormatter),
155 
156 	h: function(date, utc) {
157 		return this.H(date, utc) % 12 || 12;
158 	}.bind(DateFormatter._stringFormatter),
159 
160 	hh: function(date, utc) {
161 		return this.h(date, utc).toPaddedString(2);
162 	}.bind(DateFormatter._stringFormatter),
163 
164 	H: function(date, utc) {
165 		return this._get(date, utc, "Hours");
166 	}.bind(DateFormatter._stringFormatter),
167 
168 	HH: function(date, utc) {
169 		return this.H(date, utc).toPaddedString(2);
170 	}.bind(DateFormatter._stringFormatter),
171 
172 	M: function(date, utc) {
173 		return this._get(date, utc, "Minutes");
174 	}.bind(DateFormatter._stringFormatter),
175 
176 	MM: function(date, utc) {
177 		return this.M(date, utc).toPaddedString(2);
178 	}.bind(DateFormatter._stringFormatter),
179 
180 	s: function(date, utc) {
181 		return this._get(date, utc, "Seconds");
182 	}.bind(DateFormatter._stringFormatter),
183 
184 	ss: function(date, utc) {
185 		return this.s(date, utc).toPaddedString(2);
186 	}.bind(DateFormatter._stringFormatter),
187 
188 	_ms: function(date, utc) {
189 		return this._get(date, utc, "Milliseconds");
190 	}.bind(DateFormatter._stringFormatter),
191 
192 	l: function(date, utc) {
193 		return this._ms(date, utc).toPaddedString(3);
194 	}.bind(DateFormatter._stringFormatter),
195 
196 	L: function(date, utc) {
197 		var ms = this._ms(date, utc);
198 
199 		return ms.toPaddedString(ms > 99 ? Math.round(ms / 10) : ms);
200 	}.bind(DateFormatter._stringFormatter),
201 
202 	t: function(date, utc) {
203 		return this.H(date, utc) < 12 ? "a" : "p";
204 	}.bind(DateFormatter._stringFormatter),
205 
206 	tt: function(date, utc) {
207 		return this.H(date, utc) < 12 ? "am" : "pm";
208 	}.bind(DateFormatter._stringFormatter),
209 
210 	T: function(date, utc) {
211 		return this.t(date, utc).toUpperCase();
212 	}.bind(DateFormatter._stringFormatter),
213 
214 	TT: function(date, utc) {
215 		return this.tt(date, utc).toUpperCase();
216 	}.bind(DateFormatter._stringFormatter),
217 
218 	Z: function(date, utc) {
219 		return  utc ? "UTC" : (String(date).match(DateFormatter._timezone) || [""]).pop().replace(DateFormatter._timezoneClip, "")
220 	}.bind(DateFormatter._stringFormatter),
221 
222 	o: function(date, utc) {
223 		var o = utc ? 0 : date.getTimezoneOffset();
224 		return (o > 0 ? "-" : "+") + (Math.floor(Math.abs(o) / 60) * 100 + Math.abs(o) % 60,4).toPaddedString(2);
225 	},
226 
227 	S: function(date, utc) {
228 		var d = this.d(date, utc);
229 		return ["th", "st", "nd", "rd"][d % 10 > 3 ? 0 : (d % 100 - d % 10 != 10) * d % 10];
230 	},
231 
232 	_get: function(date, utc, property) {
233 		return date["get" + (utc ? "UTC" : "") + property]();
234 	}.bind(DateFormatter._stringFormatter)
235 };

237 /**
238  * Lets you call Date#format for convenience.
239  *
240  * <pre><code class="language-javascript">
241  * var date = new Date();
242  *
243  * // default
244  * // Mon Jan 9 2012 11:38:00
245  * date.format();
246  *
247  * // just the year in long form
248  * // 2012
249  * date.format("yyyy");
250  * </code></pre>
251  */
252 Date.prototype.format = function(mask, formatter) {
253 	return DateFormatter.format(this, mask, formatter);
254 };
