1 include(bbq.gui.GUIWidget);
  2 
  3 /**
  4  * ModalLayer
  5  * 
  6  * Greys out the page and shows the passed element in the middle.  The passed element should
  7  * extend bbq.gui.GUIWidget
  8  *
  9  * @class bbq.gui.ModalLayer
 10  * @extends bbq.gui.GUIWidget
 11  */
 12 bbq.gui.ModalLayer = new Class.create(bbq.gui.GUIWidget, {
 13 	layerContents: null,
 14 	
 15 	/**
 16 	 * @param {GUIWidget} guiWidget	bbq.gui.GUIWidget
 17 	 */
 18 	initialize: function($super, guiWidget) {
 19 		$super();
 20 		
 21 		this.setRootNode("div");
 22 		this.addClass("ModalLayer");
 23 		this.setStyle("display", "none");
 24 		
 25 		this.appendTo(document.body);
 26 		
 27 		this.layerContents = guiWidget;
 28 		this.layerContents.setStyle("display", "none");
 29 		this.layerContents.appendTo(document.body);
 30 		
 31 		Effect.Appear(this.getRootNode(), { duration: 0.25, to:0.8 });
 32 		Effect.Appear(this.layerContents.getRootNode(), { duration: 0.25, to:1 });
 33 	},
 34 	
 35 	/**
 36 	 * Fades out the layer and removes it from the DOM
 37 	 */
 38 	clearModalLayer: function() {
 39 		Effect.Fade(this.getRootNode(), { duration: 0.25 });
 40 		Effect.Fade(this.layerContents.getRootNode(), { duration: 0.25 });
 41 		
 42 		setTimeout(function() {
 43 			try {
 44 				document.body.removeChild(this.getRootNode());
 45 				document.body.removeChild(this.layerContents.getRootNode());
 46 				this.layerContents = null;			
 47 			} catch(e) {
 48 				Log.error("Error clearing modal layer", e);
 49 			}
 50 		}.bind(this), 600);
 51 	}
 52 });
