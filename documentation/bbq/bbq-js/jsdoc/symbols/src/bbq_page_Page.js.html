1 include(bbq.BBQ);
  2 include(bbq.util.Log);
  3 include(bbq.web.Browser);

  7 bbq.page.Page = new Class.create(/** @lends bbq.page.Page.prototype */ {
  8 	_modalLayer: null,
  9 	_modalLayerContents: null,
 10 	_fatalError: null,
 11 	_options: null,
 12 	
 13 	/**
 14 	 * Base class for creating per-page classes.
 15 	 *
 16 	 * @param {Object} args
 17 	 * @constructs
 18 	 */
 19 	initialize: function(args) {
 20 		//Log.info("Page constructor");
 21 		currentPage = this;
 22 		this._options = args ? args : {};
 23 		
 24 		DOMUtil.checkDOM();
 25 		
 26 		Event.observe(window, "resize", this.resizePanels.bindAsEventListener(this));
 27 	},
 28 
 29 	/**
 30 	 * Greys out the page and shows the passed element in the middle.  The passed element should
 31 	 * extend bbq.gui.GUIWidget
 32 	 * 
 33 	 * @param	 bbq.gui.GUIWidget} guiWidget
 34 	 */
 35 	addModalLayer: function(guiWidget) {
 36 		try {
 37 			if(!this._modalLayer) {
 38 				this._modalLayer = DOMUtil.createElement("div", {className: "modalLayer", style: {display: "none"}});
 39 				document.body.appendChild(this._modalLayer);
 40 			}
 41 
 42 			this._modalLayerContents = guiWidget;
 43 			this._modalLayerContents.setStyle("display", "none");
 44 			this._modalLayerContents.appendTo(document.body);
 45 
 46 			if(Browser.Mozilla && Browser.version < 3) { // stop scroll bars from showing through modal layer
 47 				$$(".ScrollableHolder").each(function(element){
 48 					element.style.overflow = "hidden";
 49 				});
 50 
 51 				$$(".FloatingWindow .ScrollableHolder").each(function(element){
 52 					element.style.overflow = "auto";
 53 				});
 54 			}
 55 
 56 			Effect.Appear(this._modalLayer, {
 57 				duration: 0.25,
 58 				to: 0.8
 59 			});
 60 			
 61 			Effect.Appear(this._modalLayerContents.getRootNode(), {
 62 				duration: 0.25,
 63 				to: 1
 64 			});
 65 
 66 			if(this._modalLayerContents.appear && this._modalLayerContents.disappear) {
 67 				// bbq.gui.FloatingWindow
 68 				this._modalLayerContents.appear();
 69 				this._modalLayerContents.registerOneTimeListener("onDisppear", this.clearModalLayer.bind(this));
 70 			}
 71 
 72 			return guiWidget;
 73 		} catch(e) {
 74 			Log.error("Error adding modal layer", e);
 75 		}
 76 	},
 77 
 78 	/**
 79 	 * Fades out the layers added by bbq.page.Page#addModalLayer
 80 	 *
 81 	 * @see bbq.page.Page#addModalLayer
 82 	 */
 83 	clearModalLayer: function() {
 84 		if(this._fatalError) {
 85 			return;
 86 		}
 87 		
 88 		if(this._modalLayer) {
 89 			Effect.Fade(this._modalLayer, {
 90 				duration: 0.25
 91 			});
 92 		}
 93 
 94 		if(this._modalLayerContents) {
 95 			Effect.Fade(this._modalLayerContents.getRootNode(), {
 96 				duration: 0.25
 97 			});
 98 		}
 99 
100 		if(Browser.Mozilla && Browser.version < 3) {// re-enable scroll bars
101 			$$(".ScrollableHolder").each(function(element){
102 				element.style.overflow = "auto";
103 			});
104 		}
105 
106 		setTimeout(this._removeModalLayer.bind(this), 600);
107 	},
108 
109 	/**
110 	 * Removes the layers added by com.proxim.page.Page.addModalLayer from the document tree
111 	 */
112 	_removeModalLayer: function() {
113 		if(this._fatalError) {
114 			return;
115 		}
116 
117 		if(!this._modalLayer || this._modalLayer.parentNode != document.body) {
118 			return;
119 		}
120 
121 		if(!this._modalLayerContents || !this._modalLayerContents.rootNode || this._modalLayerContents.rootNode.parentNode != document.body) {
122 			return;
123 		}
124 
125 		try {
126 			document.body.removeChild(this._modalLayer);
127 			document.body.removeChild(this._modalLayerContents.rootNode);
128 		} catch(e) {
129 			Log.warn("Exception thrown while removing modal layer - " + e);
130 		}
131 
132 		this._modalLayer = null;
133 		this._modalLayerContents = null;
134 	}
135 });
