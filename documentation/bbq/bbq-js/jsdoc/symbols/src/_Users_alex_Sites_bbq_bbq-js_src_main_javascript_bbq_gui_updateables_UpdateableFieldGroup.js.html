1 include(bbq.gui.GUIWidget);
  2 
  3 /**
  4  * @class bbq.gui.updateables.UpdateableFieldGroup
  5  * @extends bbq.gui.GUIWidget
  6  */
  7 bbq.gui.updateables.UpdateableFieldGroup = new Class.create(bbq.gui.GUIWidget, {
  8 	_fields: null,
  9 	_fieldOptions: null,
 10 	_editMode: null,
 11 	_preSubmitFilters: null,
 12 	
 13 	/**
 14 	 * Constructor
 15 	 * 
 16 	 * Supports the following options:
 17 	 * 
 18 	 * options: {
 19 	 * 		namespace: String		//  If passed all field values will be prefixed by namespace:  - eg. pass "person" for "person:name", etc.  Allow us to sanely submit two entities data in one go
 20 	 * }
 21 	 * 
 22 	 * @param {Object} options
 23 	  */
 24 	initialize: function($super, options) {
 25 		$super(options);
 26 		
 27 		this._fields = [];
 28 		this._fieldOptions = [];
 29 		this._preSubmitFilters = [];
 30 	},
 31 	
 32 	/**
 33 	 * 	Supports the following options:
 34 	 * 
 35 	 * options: {
 36 	 * 		name: String				// The name of the variable this field represents
 37 	 * 		label: String				// The text to be displayed next to the field
 38 	 * 		ignoreValue: boolean	//  If true the field will not be queried for a value when getValues is called
 39 	 * 		nonEditable: boolean	//  If true the field will not be put into edit mode
 40 	 * 	}
 41 	 * 
 42 	 * @param {bbq.gui.updateables.UpdateableField} field
 43 	 * @param {Object} options
 44 	 */
 45 	addField: function(field, options) {
 46 		this._fields.push(field);
 47 		this._fieldOptions.push(options ? options : {});
 48 		
 49 		return field;
 50 	},
 51 	
 52 	removeField: function(fieldName) {
 53 		for(var i = 0; i < this._fieldOptions.length; i++) {
 54 			if(this._fieldOptions[i].name == fieldName) {
 55 				var needRender = false;
 56 				
 57 				// only need update if we are removing a visible field
 58 				if(!(this._fields[i] instanceof bbq.gui.updateables.UpdateableInvisibleField)) {
 59 					needRender = true;
 60 				}
 61 				
 62 				// remove the field and it's options
 63 				this._fields.splice(i, 1);
 64 				this._fieldOptions.splice(i, 1);
 65 				
 66 				// update if needed
 67 				if(needRender && this.getRootNode() && this.getRootNode().parentNode) {
 68 					this.render();
 69 				}
 70 			}
 71 		}
 72 	},
 73 	
 74 	addPreSubmitFilter: function(filter) {
 75 		this._preSubmitFilters.push(filter);
 76 	},
 77 	
 78 	/**
 79 	 * @param {boolean} editMode
 80 	 */
 81 	setEditMode: function(editMode) {
 82 		this._editMode = editMode;
 83 		
 84 		this.render();
 85 		
 86 		for(var i = 0, iCount = this._fields.length; i < iCount; i++) {
 87 			if(this._fields[i] && !this._fieldOptions[i].nonEditable) {
 88 				this._fields[i].setEditMode(editMode);
 89 			}
 90 		}
 91 		
 92 		if(editMode) {
 93 			this.addClass("UpdateableFieldGroup_EditMode");
 94 		} else {
 95 			this.removeClass("UpdateableFieldGroup_EditMode");
 96 		}
 97 	},
 98 	
 99 	/**
100 	 * @return {boolean}
101 	 */
102 	getEditMode: function() {
103 		return this._editMode;
104 	},
105 	
106 	/**
107 	 * @param {boolean} disabled
108 	 */
109 	setDisabled: function(disabled) {
110 		for(var i = 0, iCount = this._fields.length; i < iCount; i++) {
111 			if(this._fields[i] && !this._fieldOptions[i].nonEditable) {
112 				this._fields[i].setDisabled(disabled);
113 			}
114 		}
115 	},
116 	
117 	/**
118 	 * saves edited fields
119 	 */
120 	saveEdit: function() {
121 		for(var i = 0, iCount = this._fields.length; i < iCount; i++) {
122 			if(this._fields[i] && !this._fieldOptions[i].nonEditable) {
123 				this._fields[i].saveEdit();
124 			}
125 		}
126 	},
127 	
128 	/**
129 	 * cancel edit process and revert to previous state
130 	 */
131 	cancelEdit: function() {
132 		for(var i = 0, iCount = this._fields.length; i < iCount; i++) {
133 			if(this._fields[i] && !this._fieldOptions[i].nonEditable) {
134 				this._fields[i].cancelEdit();
135 			}
136 		}
137 	},
138 	
139 	/**
140 	 * @return {Object} output
141 	 */
142 	getValues: function() {
143 		var output = {};
144 		
145 		for(var i = 0; i < this._fields.length; i++) {
146 			if(this._fields[i] && !this._fieldOptions[i].ignoreValue) {
147 				this._fields[i].clearError();
148 				
149 				output[(this.options.namespace ? this.options.namespace + ":" : "") + this._fieldOptions[i]["name"]] = this._fields[i].getValue();
150 			}
151 		}
152 		
153 		this._preSubmitFilters.each(function(filter) {
154 			var errorField = filter(output);
155 			
156 			if(errorField) {
157 				for(var i = 0; i < this._fields.length; i++) {
158 					if((this.options.namespace ? this.options.namespace + ":" : "") + this._fieldOptions[i]["name"] == errorField) {
159 						this._fields[i].triggerError();
160 					}
161 				}
162 			}
163 		}.bind(this));
164 		
165 		return output;
166 	},
167 	
168 	/**
169 	 * @return {string} output
170 	 */
171 	getValuesAsString: function() {
172 		var output = "";
173 		var values = this.getValues();
174 		
175 		for(var key in values) {
176 			output += this.parseValues(key, values[key]);
177 		}
178 		
179 		return output;
180 	},
181 	
182 	/**
183 	 * @param {string} key
184 	 * @param {mixed} value
185 	 * @return {string} output
186 	 */
187 	parseValues: function(key, value) {
188 		if(value instanceof Array) {
189 			for(var i = 0, iCount=value.length; i < iCount; i++) {
190 				output += this.parseValues(key + "[]", value[i]) + "&";
191 			}
192 		} else if(key) {
193 			output = key + "=" + encodeURIComponent(value) + "&";
194 		}
195 		
196 		return output;
197 	},
198 	
199 	/**
200 	 * @param {string} key
201 	 * @return Returns fields values
202 	 * @type {mixed}
203 	 */
204 	getValue: function(key) {
205 		for(var i = 0, iCount=this._fieldOptions.length; i < iCount; i++) {
206 			if(this._fieldOptions[i].name == key && !this._fieldOptions[i].ignoreValue) {
207 				return this._fields[i].getValue();
208 			}
209 		}
210 	},
211 	
212 	/**
213 	 * @return fields
214 	 * @type {mixed}
215 	 */
216 	getFields: function() {
217 		return this._fields
218 	},
219 	
220 	changed: function() {
221 		var changed = false;
222 		
223 		this._fields.each(function(field, index) {
224 			
225 			if(this._fieldOptions[index] && this._fieldOptions[index].ignoreValue) {
226 				return;
227 			}
228 			
229 			if(field && field.changed) {
230 				if(field.changed()) {
231 					changed = true;
232 				}
233 			}
234 		}.bind(this));
235 		
236 		return changed;
237 	},
238 	
239 	/**
240 	 * b0rk b0rk b0rk.  Because of no support for DOMNodeRemoved events in IEx, we must manually tell our child updateables to clean up after themselves.
241 	 */
242 	deregisterChildListeners: function() {
243 		this._fields.each(function(field) {
244 			if(field && field.deregisterChildListeners) {
245 				field.deregisterChildListeners();
246 			}
247 		}.bind(this));
248 	}
249 });
