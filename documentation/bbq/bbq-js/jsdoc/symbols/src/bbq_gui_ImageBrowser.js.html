1 include(bbq.gui.ScrollLoadGUIWidget);
  2 include(bbq.gui.LoadingNotification);
  3 
  4 /**
  5  * @class bbq.gui.ImageBrowser
  6  * @extends bbq.gui.table.ScrollLoadGUIWidget
  7  */
  8 bbq.gui.ImageBrowser = new Class.create(bbq.gui.ScrollLoadGUIWidget, {
  9 	
 10 	_lastNumEntities: null,
 11 	
 12 	/**
 13 	 * Constructor
 14 	 * @param {mixed} options
 15 	 */
 16 	initialize: function($super, options) {
 17 		$super(options);
 18 		
 19 		this.addClass("ImageBrowser");
 20 	},
 21 	
 22 	/**
 23 	 * @private
 24 	 */
 25 	_workOutMovement: function() {	
 26 		if(this._scrollingTimeout) {
 27 			clearTimeout(this._scrollingTimeout);
 28 		}
 29 		
 30 		var scrollableDimensions = Element.getDimensions(this._scrollableHolder.getRootNode());
 31 		
 32 		if(scrollableDimensions.height == 0) {
 33 			return;
 34 		}
 35 		
 36 		// calculate number of images in row
 37 		var offset = this._scrollableHolder.getRootNode().scrollTop;
 38 		var numImagesInRow = Math.floor((scrollableDimensions.width - 4)/(this.options.boxWidth + 7));
 39 		
 40 		if(numImagesInRow == 0) {
 41 			numImagesInRow = 1;
 42 		}
 43 		
 44 		// calculate number of visible rows
 45 		var visibleHeight = scrollableDimensions.height;
 46 		var rowHeight = this.options.boxHeight + 4
 47 		var startRow = Math.floor(offset > 0 ? (offset/rowHeight) : 1);
 48 		var endRow = Math.ceil((offset + visibleHeight)/rowHeight);
 49 		
 50 		// calculate number of visible images
 51 		var startImage = (startRow * numImagesInRow) - numImagesInRow;
 52 		var endImage = (endRow * numImagesInRow);
 53 		
 54 		if(this._lastRange.start != startImage || this._lastRange.end != endImage || this._initialLoad === true) {
 55 			this._lastRange.start = startImage;
 56 			this._lastRange.end = endImage;
 57 			
 58 			// load missing details
 59 			this._loadDetails(startImage, endImage);
 60 		}
 61 	},
 62 	
 63 	/**
 64 	 * Render
 65 	 */
 66 	render: function() {	
 67 		
 68 		var scrollTop = 0;
 69 		
 70 		if(this._scrollableHolder) {
 71 			scrollTop = this._scrollableHolder.getScrollTop();
 72 		}
 73 		
 74 		this.empty();
 75 		
 76 		if(this._initialLoad == true) {
 77 			this.appendChild(new bbq.gui.LoadingNotification({text: this.options.loadingText}));
 78 		} else if(this._entities.getElements().length == 0) {
 79 			this.appendChild(DOMUtil.createTextElement("p", this.options.noEntitiesText, {className: "ScrollLoadTableEmptyList"}));
 80 		} else {
 81 			try {	
 82 				this._entities.getElements().each(function(entity, index) {
 83 					if(entity) {
 84 						this.appendChild(this.options.createImageView(entity, index));
 85 					} else {
 86 						this.appendChild(this._renderInvalidItem(index));
 87 					}
 88 				}.bind(this));
 89 				
 90 				this._lastNumEntities = this._entities.length;
 91 			} catch(e) {
 92 				Log.error("Error while rendering", e);
 93 			}
 94 		}
 95 		
 96 		if(this._scrollableHolder) {
 97 			this._scrollableHolder.setScrollTop(scrollTop);
 98 		}
 99 	},
100 	
101 	/**
102 	 * @private
103 	 */
104 	_renderInvalidItem: function(index) {
105 		return DOMUtil.createTextElement("div", new bbq.gui.LoadingNotification(), {
106 			className: "ImageBox", style: {height: this.options.boxHeight + "px", width: this.options.boxWidth+ "px"}, invalid: true
107 		});
108 	}	
109 });
