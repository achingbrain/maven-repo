1 include(bbq.gui.panel.Panel);
  2 
  3 /**
  4  * A panel with two and only two child panels
  5  * @class bbq.gui.panel.DualPanel
  6  * @extends bbq.gui.panel.Panel
  7  * 
  8  */
  9 bbq.gui.panel.DualPanel  = new Class.create(bbq.gui.panel.Panel, {
 10 	_panel1Name: null,
 11 	_panel2Name: null,
 12 	
 13 	/**
 14 	 * @param {Object} options
 15 	 * 
 16 	 * Supports the following options: 
 17 	 * 
 18 	 * options: {
 19 	 * 		panel1: bbq.gui.panel.Panel
 20 	 * 		panel2: bbq.gui.panel.Panel
 21 	 * }
 22 	 */
 23 	initialize: function($super, options) {
 24 		// call parent method
 25 		$super(options);
 26 		
 27 		this.addClass("DualPanel");
 28 		
 29 		this._panel1Name = "panel1";
 30 		this._panel2Name = "panel2";
 31 		
 32 		if(this.options.topPanel) {
 33 			this.setTopPanel(this.options.panel1);
 34 		}
 35 		
 36 		if(this.options.topPanel) {
 37 			this.setTopPanel(this.options.topPanel);
 38 		}
 39 		
 40 		if(this.options.bottomPanel) {
 41 			this.setBottomPanel(this.options.bottomPanel);
 42 		}
 43 		
 44 		if(this.options.leftPanel) {
 45 			this.setLeftPanel(this.options.leftPanel);
 46 		}
 47 		
 48 		if(this.options.rightPanel) {
 49 			this.setRightPanel(this.options.rightPanel);
 50 		}
 51 	},
 52 	
 53 	setTopPanel: function(content, panelName) {
 54 		this.layout.setVertical(true);
 55 		this.removeClass("DualPanel_Horizontal");
 56 		this.addClass("DualPanel_Vertical");
 57 		this._setPanel(content, panelName, 1, "top");
 58 		
 59 		return content;
 60 	},
 61 	
 62 	getTopPanel: function() {
 63 		return this._getChildPanel(this._panel1Name);
 64 	},
 65 	
 66 	setBottomPanel: function(content, panelName) {
 67 		this.layout.setVertical(true);
 68 		this.removeClass("DualPanel_Horizontal");
 69 		this.addClass("DualPanel_Vertical");
 70 		this._setPanel(content, panelName, 2, "bottom");
 71 		
 72 		return content;
 73 	},
 74 	
 75 	getBottomPanel: function() {
 76 		return this._getChildPanel(this._panel2Name);
 77 	},
 78 	
 79 	setLeftPanel: function(content, panelName) {
 80 		this.layout.setVertical(false);
 81 		this.removeClass("DualPanel_Vertical");
 82 		this.addClass("DualPanel_Horizontal");
 83 		this._setPanel(content, panelName, 1, "left");
 84 		
 85 		return content;
 86 	},
 87 	
 88 	getLeftPanel: function() {
 89 		return this._getChildPanel(this._panel1Name);
 90 	},
 91 	
 92 	setRightPanel: function(content, panelName) {
 93 		this.layout.setVertical(false);
 94 		this.removeClass("DualPanel_Vertical");
 95 		this.addClass("DualPanel_Horizontal");
 96 		this._setPanel(content, panelName, 2, "right");
 97 		
 98 		return content;
 99 	},
100 	
101 	getRightPanel: function() {
102 		return this._getChildPanel(this._panel2Name);
103 	},
104 	
105 	_setPanel: function(content, panelName, panelIndex, panelClass) {
106 		DOMUtil.addClass(content, "DualPanel" + panelIndex);
107 		DOMUtil.addClass(content, "DualPanel_" + (this.layout.vertical() ? "Vertical" : "Horizontal") + "_DualPanel" + panelIndex);
108 		DOMUtil.addClass(content, "DualPanel_" + panelClass + "Panel");
109 		
110 		if(!Object.isUndefined(panelName)) {
111 			this["_panel" + panelIndex + "Name"] = panelName;
112 		}
113 		
114 		this._setChildPanel(this["_panel" + panelIndex + "Name"], content);
115 		this.render();
116 	},
117 	
118 	render: function() {
119 		this.empty();
120 		
121 		if(this.layout.vertical()) {
122 			if(this.getTopPanel()) {
123 				this.appendChild(this.getTopPanel());
124 			}
125 			
126 			if(this.getBottomPanel()) {
127 				this.appendChild(this.getBottomPanel());
128 			}
129 		} else {
130 			if(this.getRightPanel()) {
131 				this.appendChild(this.getRightPanel());
132 			}
133 			
134 			if(this.getLeftPanel()) {
135 				this.appendChild(this.getLeftPanel());
136 			}
137 		}
138 	},
139 	
140 	replacePanel: function($super, oldPanel, newPanel) {
141 		var oldName = this._getPanelName(oldPanel);
142 		
143 		$super(oldPanel, newPanel);
144 		
145 		if(oldName == this._panel1Name) {
146 			this._setPanel(newPanel, oldName, 1)
147 		} else if(oldName == this._panel2Name) {
148 			this._setPanel(newPanel, oldName, 2)
149 		}
150 	}
151 });
