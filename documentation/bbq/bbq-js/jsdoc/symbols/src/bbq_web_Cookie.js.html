1 /**
  2  * @class Cookie
  3  */
  4 Cookie = {
  5 	_keys: [
  6 		"expires",
  7 		"domain",
  8 		"path",
  9 		"secure"
 10 	],
 11 
 12 	/**
 13 	 * Set a cookie
 14 	 *
 15 	 * @param key
 16 	 * @param value
 17 	 * @param expires
 18 	 * @param domain
 19 	 * @param path
 20 	 * @param secure
 21 	 */
 22 	set: function(key, value, expires, domain, path, secure) {
 23 		if(!value) {
 24 			// remove old cookie
 25 			if(Cookie.get(key)) {
 26 				Cookie.del(key);
 27 			}
 28 			
 29 			return;
 30 		}
 31 		
 32 		var args = {
 33 			"expires": expires ? expires : new Date(new Date().getTime() + 2629743000),
 34 			"domain": domain,
 35 			"path": path,
 36 			"secure": secure
 37 		};
 38 		
 39 		var cookie = {
 40 			value: value
 41 		};
 42 		
 43 		Cookie._keys.each(function(arg) {
 44 			Cookie._applyArg(cookie, arg, args[arg]);
 45 		});
 46 		
 47 		document.cookie = Cookie._toString(key, cookie);
 48 	},
 49 	
 50 	_applyArg: function(cookie, key, value) {
 51 		if(value) {
 52 			cookie[key] = value;
 53 		}
 54 	},
 55 
 56 	/**
 57 	 * Get a cookie previously set with Cookie#set
 58 	 *
 59 	 * @param cookieName
 60 	 * @see Cookie#set
 61 	 */
 62 	get: function(cookieName) {
 63 		var cookie = Cookie._getAll()[cookieName];
 64 		
 65 		if(cookie != null) {
 66 			return cookie.value;
 67 		}
 68 		
 69 		return null;
 70 	},
 71 	
 72 	del: function(cookieName) {
 73 		var cookies = Cookie._getAll();
 74 		
 75 		if(cookies[cookieName]) {
 76 			cookies[cookieName].expires = new Date(new Date().getTime() - 1000000);
 77 			document.cookie = Cookie._toString(cookieName, cookies[cookieName]);
 78 		}
 79 	},
 80 	
 81 	_getAll: function() {
 82 		var output = {};
 83 		
 84 		var parts = document.cookie.split("; ");
 85 		var currentName;
 86 		
 87 		for(var i = 0; i < parts.length; i++) {
 88 			var keyPart = parts[i];
 89 			
 90 			if(keyPart && keyPart.strip) {
 91 				
 92 				var keyValue = keyPart.strip().split("=");
 93 				
 94 				if(keyValue[0] == "expires") {
 95 					output[currentName].expires = new Date(keyValue[1]);
 96 				} else if(keyValue[0] == "path") {
 97 					output[currentName].path = keyValue[1];
 98 				} else if(keyValue[0] == "domain") {
 99 					output[currentName].domain = keyValue[1];
100 				} else if(keyValue[0] == "secure") {
101 					output[currentName].secure = true;
102 				} else {
103 					currentName = keyValue[0];
104 					output[currentName] = {
105 						value: keyValue[1]
106 					};
107 				}
108 			}
109 		}
110 		
111 		return output;
112 	},
113 	
114 	_toString: function(cookieName, object) {
115 		var cookie = [
116 			cookieName + "=" + escape(object.value)
117 		];
118 		
119 		Cookie._keys.each(function(arg) {
120 			if(object[arg]) {
121 				if(arg == "expires") {
122 					cookie.push(arg + "=" + object[arg].toUTCString());
123 				} else if(arg == "secure") {
124 					cookie.push(arg);
125 				} else {
126 					cookie.push(arg + "=" + object[arg]);
127 				}
128 			}
129 		});
130 		
131 		return cookie.join("; ");
132 	}
133 }
