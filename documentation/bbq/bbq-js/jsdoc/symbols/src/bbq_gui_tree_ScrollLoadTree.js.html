 94 	_handleValidItem: function(entity, index) {
 95 		try {
 96 			var entityKey = BBQUtil.getKey(entity);
 97 			var treeData = this._entityTree[entityKey];
 98 			
 99 			if(!treeData) {
100 				Log.warn("Could not determing tree data for entity");
101 				Log.dir(entity);
102 				return;
103 			}
104 			
105 			var constructionData = this._listConstructionData[treeData["level"]];
106 			
107 			var output = DOMUtil.createElement("li", {
108 				className: "CollapseExpandListTitle ScrollLoadTreeLevel" + treeData["level"]
109 			});
110 					
111 			// set down if has been selcted before
112 			//if(index == this._selectedIndex) {
113 			if(entityKey == this._selectedKey) {
114 				DOMUtil.addClass(output, "down");
115 			}
116 			
117 			if(!constructionData) {
118 				Log.info("could not get construction data for level " + treeData["level"]);
119 			}
120 			
121 			// add collapse/expand button
122 			if(constructionData.collapsible) {
123 				var startCollapsed = !treeData["folderOpen"];
124 				
125 				output.collapsible = true;
126 				output.collapsed = startCollapsed;
127 				
128 				if(treeData["children"].length > 0) {
129 					output.appendChild(DOMUtil.createTextElement("a", Language.get("bbq.gui.tree.ScrollLoadTreeList.collapseexpand"), {
130 						href: ".",
131 						onclick: function(event, entity) {
132 							try {
133 								BBQUtil.clearFocus(event);
134 								this._folderClicked(entity);
135 							} catch(e) {
136 								Log.error("Error clearing focus", e);
137 							}
138 							
139 							return false;
140 						}.bindAsEventListener(this, entity),
141 						title: Language.get("bbq.gui.tree.ScrollLoadTreeList.collapseexpand"),
142 						className: "CollapseExpand  CollapseExpand" + (startCollapsed ? "Collapsed" : "Expanded"),
143 						collapsed: startCollapsed
144 					}));
145 				} else {
146 					// no children, add placeholder
147 					output.appendChild(DOMUtil.createElement("span", {
148 						className: "CollapseExpand  CollapseExpandCollapserPlaceHolder"
149 					}));
150 				}
151 			}
152 			
153 			var text = constructionData.text(entity);
154 			var folderTitle;
155 			
156 			if(constructionData.onclick || constructionData.ondblclick) {
157 				folderTitle = DOMUtil.createTextElement("a", text, {
158 					href: ".",
159 					onmousedown: this._processEvent.bindAsEventListener(this, entity),
160 					onclick: this._processEvent.bindAsEventListener(this, entity),
161 					ondblclick: this._processEvent.bindAsEventListener(this, entity),
162 					className: "CollapseExpandFolderTitle"
163 				});
164 				
165 				if(Object.isElement(text)) {
166 					Event.observe(text, "click", function() {
167 						folderTitle.blur();
168 						return true;
169 					});
170 				}
171 				
172 				if(constructionData.onclick) {
173 					this._storeCallback(entity, "click", constructionData.onclick.bind(this, entity));
174 				}
175 				
176 				if(constructionData.ondblclick) {
177 					this._storeCallback(entity, "dblclick", constructionData.ondblclick.bind(this, entity));
178 				}
179 			} else {
180 				if(Object.isElement(text)) {
181 					folderTitle = text;
182 				} else {
183 					folderTitle = DOMUtil.createTextElement("span", text);
184 				}
185 				
186 				DOMUtil.addClass(folderTitle, "CollapseExpandFolderTitle");
187 			}
188 			
189 			if(constructionData.className instanceof Function) {
190 				var className = constructionData.className(entity, [folderTitle, output]);
191 				
192 				if(Object.isString(className)) {
193 					DOMUtil.addClass(folderTitle, className);
194 					DOMUtil.addClass(output, className);
195 				}
196 			}
197 			
198 			output.appendChild(folderTitle);
199 			
200 			return output;
201 		} catch(e) {
202 			Log.error("Error handling valid item", e);
203 		}
204 		
205 		return output;
206 	},

218 	_folderClicked: function(entity, newStatus) {
219 		var treeData = this._entityTree[BBQUtil.getKey(entity)];
220 		
221 		// newStatus is passed by left/right arrow keypress handler.  abort if our status will not change.
222 		if(typeof(newStatus) != "undefined" && newStatus == !treeData["folderOpen"]) {
223 			return;
224 		}
225 		
226 		// store where we are scrolled to so that we can scroll to it after loading
227 		this._scrollTop = this._scrollableHolder.getRootNode().scrollTop;
228 		
229 		// store collapsed state
230 		new bbq.ajax.JSONRequest({
231 			url: "/backend/setFolderCollapsedStatus", 
232 			args: {
233 				entity:  this._getEntityKey(entity),
234 				closed: treeData["folderOpen"]
235 			}, 
236 			onSuccess: this.reloadEntities.bind(this)
237 		});
238 		
239 		return false;
240 	},

291 	/**
292 	 * Process key presses
293 	 * 
294 	 * @param {Event} event
295 	 * @return {Boolean} true
296 	 */
297 	processKeypress: function(event) {
298 		var increment = 1;
299 		var nextEntity = this._getNextEntity(event, increment);
300 		
301 		while(nextEntity) {
302 			if(this._invokeCallback(nextEntity, "click") === false) {
303 				increment++;
304 				nextEntity = this._getNextEntity(event, increment);
305 			} else {
306 				return true;
307 			}
308 		}
309 		
310 		return true;
311 	},
312 	
313 	_getNextEntity: function(event, increment) {
314 		var nextEntity = false;
315 		
316 		// process keypress
317 		if(event.keyCode == Event.KEY_DOWN) {
318 			nextEntity = this._entities.getElement(this._selectedIndex + increment);
319 		} else if(event.keyCode == Event.KEY_UP) {
320 			nextEntity = this._entities.getElement(this._selectedIndex - increment);
321 		} else if(event.keyCode == Event.KEY_RIGHT) { // if the currently selected level is collapsible and collapsed, expand it
322 			this._folderClicked(this._entities.getElement(this._selectedIndex), false);
323 		} else if(event.keyCode == Event.KEY_LEFT) { // if the currently selected level is collapsible and expanded, collapse it
324 			this._folderClicked(this._entities.getElement(this._selectedIndex), true);
325 		}
326 		
327 		return nextEntity;
328 	},

330 	/**
331 	 * @private
332 	 * Override
333 	 */
334 	 _processNewEntities: function(json) {
335 		try {
336 			var keyToEntity = {};
337 			
338 			for(var i = 0, iCount=json.numEntities; i < iCount; i++) {
339 				var entityData = json.entities["en" + i];
340 				
341 				// do not process missing items
342 				if(typeof(entityData) == "undefined") {
343 					continue;
344 				}
345 				
346 				// overwrite currently stored entity with entity passed from server, otherwise preserve entitiy 
347 				// if we have it already, otherwise make false to trigger placeholder
348 				if(this.options.createEntity instanceof Function) {
349 					this._entities.setElement(i, this.options.createEntity(entityData, i));
350 				} else {
351 					this._entities.setElement(i, entityData);
352 				}
353 				
354 				if(this._entities.getElement([i])) {
355 					if(!this._entities.getElement(i).getId) {
356 						Log.dir(this._entities.getElement(i));
357 					} else{
358 						keyToEntity[this._entities.getElement(i).getId()] = this._entities.getElement(i);
359 					}
360 				}
361 			}
362 			var tree = json.entityTree;
363 			// store tree construction data
364 			for(var key in tree) {
365 				if(!key) {
366 					continue;
367 				}
368 				
369 				if(this._entityTree[key]) {
370 					this._entityTree[key]["folderOpen"] = tree[key]["folderOpen"];
371 					this._entityTree[key]["children"] = this._entityTree[key]["children"].concat(tree[key]["children"]).uniq();
372 				} else {
373 					this._entityTree[key] = {
374 						folderOpen: tree[key]["folderOpen"] ? true : false,
375 						children: tree[key]["children"] ? tree[key]["children"] : [],
376 						level: tree[key]["level"]
377 					};
378 				}
379 			}
380 			
381 			this.render();
382 		} catch(e) {
383 			Log.error("Error processing new entities", e);
384 		}
385 	},

387 	/**
388 	 * @private
389 	 */
390 	_workOutMovement: function() {	
391 		if(this._scrollingTimeout) {
392 			clearTimeout(this._scrollingTimeout);
393 		}
394 		
395 		// do not try to load entities if there were none last time as chances are there still won't be any
396 		if(!this._initialLoad && this._entities.getLength() == 0) {
397 			return;
398 		}
399 		
400 		// if we've no scrollabelholder, we won't know how many to load
401 		if(!this._scrollableHolder) {
402 			return;
403 		}
404 		
405 		var offset = this._scrollableHolder.getRootNode().scrollTop;
406 		
407 		// div height or table height if not scrolling
408 		var visibleHeight = Element.getHeight(this._scrollableHolder.getRootNode());
409 		
410 		if(visibleHeight == 0) {
411 			return;
412 		}
413 		
414 		var startRow = Math.floor(offset/this.options.rowHeight);
415 		var endRow = startRow + Math.floor(visibleHeight/this.options.rowHeight);
416 		
417 		if(this._lastRange.start != startRow || this._lastRange.end != endRow || this._initialLoad) {
418 			this._lastRange.start = startRow;
419 			this._lastRange.end = endRow;
420 			
421 			// load missing details
422 			//Log.info("calling loadDetails for rows " + (startRow) + " to " + (endRow + 1) + " this._forceLoad = " + this._forceLoad);
423 			this._loadDetails(startRow, endRow + 1);
424 		}
