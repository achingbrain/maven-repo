  3 /**
  4  * @class bbq.gui.updateables.UpdateableCheckBox A single line HTML text input field.
  5  * @extends bbq.gui.updateables.UpdateableField 
  6  */
  7 bbq.gui.updateables.UpdateableCheckBox = new Class.create(bbq.gui.updateables.UpdateableField, {
  8 	inputField: null,
  9 	
 10 	/**
 11 	 * @param {Object} options 
 12 	 * @example
 13 	 * Supports the following options:
 14 	 * 
 15 	 * options: {
 16 	 * 		searchBox: boolean				// if true will render a mac like search box on Safari
 17 	 * }
 18 	 */
 19 	initialize: function($super, options) {
 20 		if(typeof(options) == "undefined") {
 21 			options = {};
 22 		}
 23 		
 24 		options.initialValue = options.initialValue ? true : false; 
 25 		
 26 		$super(options);
 27 		
 28 		this.addClass("UpdateableCheckBox");
 29 	},
 30 	
 31 	/**
 32 	 * Creates an editable representation of this field
 33 	 */
 34 	createEditField: function() {
 35 		this.inputField = DOMUtil.createElement("input", {type: "checkbox", checked: this.getValue(true)});
 36 		
 37 		Event.observe(this.inputField, "blur", this._processEventCallback.bindAsEventListener(this, "onblur"));
 38 		Event.observe(this.inputField, "focus", this._processEventCallback.bindAsEventListener(this, "onfocus"));
 39 		Event.observe(this.inputField, "change", this._processEventCallback.bindAsEventListener(this, "onchange"));
 40 		Event.observe(this.inputField, "click", this._processEventCallback.bindAsEventListener(this, "onclick"));
 41 		
 42 		return this.inputField;
 43 	},
 44 	
 45 	createViewField: function() {
 46 		if(this.getValue(true) && this.options.checkedText) {
 47 			return document.createTextNode(this.options.checkedText);
 48 		} else if(!this.getValue(true) && this.options.unCheckedText) {
 49 			return document.createTextNode(this.options.unCheckedText);
 50 		}
 51 		
 52 		return bbq.gui.updateables.UpdateableField.prototype.createViewField.apply(this);
 53 	},
 54 	
 55 	_processEventCallback: function(event, type) {
 56 		setTimeout(function() {
 57 			this.updateLocalValue();
 58 			
 59 			if(this.options[type] && this.options[type] instanceof Function) {
 60 				this.options[type]();
 61 			}
 62 			
 63 			this.notifyListeners(type);
 64 		}.bind(this), 100);
 65 	},
 66 	
 67 	updateLocalValue: function() {
 68 		var newValue = this.inputField.checked ? true : false;
 69 		var shouldNotify = false;
 70 		
 71 		if(this.currentValue != newValue) {
 72 			shouldNotify = true;
 73 		}
 74 		
 75 		this.currentValue = newValue;
 76 		
 77 		if(shouldNotify) {
 78 			this.notifyListeners("onValueChanged");
 79 		}
 80 	},
 81 	
 82 	/**
 83 	 * Ensure we return a boolean value
 84 	 */
 85 	getValue: function(supressErrorWarning) {
 86 		if(bbq.gui.updateables.UpdateableField.prototype.getValue.call(this, supressErrorWarning)) {
 87 			return true;
 88 		}
 89 		
 90 		return false;
 91 	}
 92 });
