  4 bbq.lang.Watchable = new Class.create(/** @lends bbq.lang.Watchable.prototype */ {
  5 	/**
  6 	 * Storage for callbacks registered on this object
  7 	 *
  8 	 * @type {Object}
  9 	 */
 10 	_callbacks: null,
 11 
 12 	/**
 13 	 * @constructs
 14 	 */
 15 	initialize: function() {
 16 		this._callbacks = {};
 17 	},
 18 
 19 	/**
 20 	 * Returns a callback array, initialising it where necessary
 21 	 */
 22 	_getCallbacks: function(type) {
 23 		if(Object.isUndefined(this._callbacks[type])) {
 24 			this._callbacks[type] = new Hash();
 25 		}
 26 
 27 		return this._callbacks[type];
 28 	},
 29 
 30 	/**
 31 	 * Example: this._dropDown.registerListener("onchange", this._subTypeChanged.bind(this, predicateType));
 32 	 * 
 33 	 * @param {String} type
 34 	 * @param {Function} callback
 35 	 */
 36 	registerListener: function(type, callback) {
 37 		var callbackKey = BBQUtil.generateGUID();
 38 		this._getCallbacks(type).set(callbackKey, callback);
 39 
 40 		return callbackKey;
 41 	},
 42 
 43 	/**
 44 	 * Similar to Watchable.registerListener except the callback will only be called once
 45 	 * 
 46 	 * @example
 47 	 * <pre><code type="language-javascript">
 48 	 * this._dropDown.registerOneTimeListener("onchange", this._subTypeChanged.bind(this, predicateType));
 49 	 * </code></pre>
 50 	 * @param {String} type
 51 	 * @param {Function} callback
 52 	 */
 53 	registerOneTimeListener: function(type, callback) {
 54 		callback.__oneTime = true;
 55 
 56 		return this.registerListener(type, callback);
 57 	},
 58 
 59 	/**
 60 	 * @param {String} type
 61 	 * @param {Function} callbackKey
 62 	 */
 63 	deRegisterListener: function(type, callbackKey) {
 64 		if(callbackKey) {
 65 			return this._getCallbacks(type).unset(callbackKey) ? true : false;
 66 		}
 67 
 68 		return false;
 69 	},
 70 
 71 	/**
 72 	 * Notifies this object's listeners of an event.
 73 	 * 
 74 	 * <pre><code type="language-javascript">
 75 	 * watchable.registerListener("myEvent", function() {
 76 	 *    alert("hello");
 77 	 * });
 78 	 * 
 79 	 * watchable.notifyListeners("myEvent");
 80 	 * // alerts "hello"
 81 	 * </code></pre>
 82 	 * 
 83 	 * By default the watchable is passed as the first argument to the callback function.  If you wish to pass extra arguments, do the following:
 84 	 * 
 85 	 * <pre><code type="language-javascript">
 86 	 * watchable.registerListener("myEvent", function(theWatchable, someArg) {
 87 	 *     alert(someArg);
 88 	 * });
 89 	 * 
 90 	 * watchable.notifyListeners("myEvent", "bob");
 91 	 * // alerts "bob"
 92 	 * </code></pre>
 93 	 * 
 94 	 * @param {String} type
 95 	 * @param {...} args
 96 	 */
 97 	notifyListeners: function() {
 98 		var args = [this];
 99 		var type = "";
100 
101 		for(var i = 0; i < arguments.length; i++) {
102 			if(i == 0) {
103 				type = arguments[i];
104 			} else {
105 				args.push(arguments[i]);
106 			}
107 		}
108 
109 		if(Object.isUndefined(this._getCallbacks(type))) {
110 			return;
111 		}
112 
113 		this._getCallbacks(type).keys().each(function(key) {
114 			this.notifyListener(type, key, args);
115 		}.bind(this));
116 
117 		// notify global listeners
118 		bbq.lang.Watchable.notifyGlobalListeners.apply(this, arguments);
119 	},
120 
121 	/**
122 	 * Calls the passed callback and deregisters it if it's a one time listener.
123 	 * 
124 	 * @param {String} type The event type
125 	 * @param {String} key The callback key
126 	 */
127 	notifyListener: function(type, key, args) {
128 		var callback = this._getCallbacks(type).get(key);
129 
130 		if(!callback) {
131 			return;
132 		}
133 
134 		try {
135 			callback.apply(this, args);
136 		} catch(e) {
137 			Log.error("Error invoking callback " + type + " key " + key, e);
138 		}
139 
140 		if(callback.__oneTime) {
141 			this.deRegisterListener(type, key);
142 		}
143 	}
144 });

146 /**
147  * Storage for global callbacks
148  */
149 bbq.lang.Watchable._globalCallbacks = {};
150 
151 bbq.lang.Watchable._getGlobalCallbacks = function(type) {
152 	if(!Object.isArray(bbq.lang.Watchable._globalCallbacks[type])) {
153 		bbq.lang.Watchable._globalCallbacks[type] = [];
154 	}
155 
156 	return bbq.lang.Watchable._globalCallbacks[type];
157 }
158 
159 /**
160  * Allows us to register for any type of callback called on any object
161  *
162  * @param {String} type
163  * @param {Function} callback
164  */
165 bbq.lang.Watchable.registerGlobalListener = function(type, callback) {
166 	bbq.lang.Watchable._getGlobalCallbacks(type).push(callback);
167 };
168 
169 /**
170  * Notifies global listeners.
171  */
172 bbq.lang.Watchable.notifyGlobalListeners = function() {
173 	var args = [this];
174 	var type = "";
175 
176 	for(var i = 0; i < arguments.length; i++) {
177 		if(i == 0) {
178 			type = arguments[i];
179 		} else {
180 			args.push(arguments[i]);
181 		}
182 	}
183 
184 	var callbacks = bbq.lang.Watchable._getGlobalCallbacks(type);
185 
186 	for(var i = 0; i < callbacks.length; i++) {
187 		try {
188 			callbacks[i].apply(this, args);
189 		} catch(e) {
190 			Log.error("Error invoking callback on global listeners", e);
191 		}
192 	}
193 };
