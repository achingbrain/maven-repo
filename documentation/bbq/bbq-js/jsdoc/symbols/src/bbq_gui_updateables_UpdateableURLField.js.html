1 include(bbq.gui.updateables.UpdateableTextField);
  2 
  3 /**
  4  * @class bbq.gui.updateables.UpdateableURLField
  5  * @extends behaviour.gui.updateables.UpdateableTextField 
  6  */
  7 behaviour.gui.updateables.UpdateableURLField = new Class.create(behaviour.gui.updateables.UpdateableTextField, {
  8 	
  9 	/**
 10 	 * A text field containing a URL
 11 	 * @param {Object} options
 12 	 */
 13 	initialize: function($super, options) {
 14 		if(!options.initialValue) {
 15 			options.initialValue = "http://";
 16 		}
 17 		
 18 		$super(options);
 19 		
 20 		this.addClass("UpdateableURLField");
 21 		
 22 		this.registerListener("onValueChanged", this._showIsSecure.bind(this));
 23 	},
 24 	
 25 	/**
 26 	 * Creates a view field
 27 	 */
 28 	createViewField: function() {
 29 		this.empty();
 30 		
 31 		if(this.displayValue && this.displayValue.length > 0) {
 32 			if(this.getProtocol()) {
 33 				this.appendChild(DOMUtil.createTextElement("a", this._getTruncatedValue(), {href: this.currentValue, target: "new"}));
 34 			} else {
 35 				this.appendChild(document.createTextNode(this._getTruncatedValue()));
 36 			}
 37 		} else {
 38 			this.appendChild(document.createTextNode(" "));
 39 		}
 40 	},
 41 	
 42 	/**
 43 	 * Returns current value
 44 	 */
 45 	getValue: function() {
 46 		if(this.currentValue == "http://" || this.currentValue == "https://") {
 47 			this.currentValue = "";
 48 		}
 49 		
 50 		return behaviour.gui.updateables.UpdateableTextField.prototype.getValue.apply(this, arguments);
 51 	},
 52 	
 53 	_showIsSecure: function() {
 54 		try {
 55 			this.removeClass("UpdateableURLField_FieldIsSecure");
 56 			
 57 			if(this.getProtocol() == "https" || this.getProtocol() == "ssl" || this.getProtocol() == "ssh" || this.getProtocol() == "sftp") {
 58 				this.addClass("UpdateableURLField_FieldIsSecure");
 59 			}
 60 		} catch(e) {
 61 			Log.warn(e);
 62 		}
 63 	},
 64 	
 65 	_getParts: function() {
 66 		// handles urls of the type http://user:password@domain.com/some/path
 67 		
 68 		var value = this.currentValue.strip().toLowerCase();
 69 		
 70 		var parts = {};
 71 		
 72 		// gets protocol
 73 		if(value.search("://")) {
 74 			var tmpArray = value.split("://");
 75 			parts.protocol = tmpArray[0];
 76 			value = tmpArray[1];
 77 		}
 78 		
 79 		// gets path
 80 		if(value.search("/")) {
 81 			var tmpArray = value.split("/", 1);
 82 			parts.path = tmpArray[1];
 83 			value = tmpArray[0];
 84 		}
 85 		
 86 		// gets username and password
 87 		if(value.search("@")) {
 88 			var tmpArray = value.split("@");
 89 			parts.domain = tmpArray[1];
 90 			tmpArray = tmpArray[0].split(":");
 91 			parts.username = tmpArray[0];
 92 			parts.password = tmpArray[1];
 93 		} else {
 94 			parts.domain = value;
 95 		}
 96 		
 97 		// gets port number
 98 		if(typeof(parts.domain) != "undefined" && parts.domain.search(":")) {
 99 			var tmpArray = parts.domain.split(":");
100 			parts.domain = tmpArray[0];
101 			parts.port = tmpArray[1];
102 		}
103 		
104 		return parts;
105 	},
106 	
107 	getProtocol: function() {
108 		return this._getParts()["protocol"];
109 	},
110 	
111 	getPath: function() {
112 		return this._getParts()["path"];
113 	},
114 	
115 	getDomain: function() {
116 		return this._getParts()["domain"];
117 	},
118 	
119 	getPort: function() {
120 		return this._getParts()["port"];
121 	},
122 	
123 	getUsername: function() {
124 		return this._getParts()["username"];
125 	},
126 	
127 	getPassword: function() {
128 		return this._getParts()["password"];
129 	}
130 });
