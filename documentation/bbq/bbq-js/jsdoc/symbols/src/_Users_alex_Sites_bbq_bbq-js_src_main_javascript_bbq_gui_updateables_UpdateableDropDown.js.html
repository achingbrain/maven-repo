1 include(bbq.gui.updateables.UpdateableField);
  2 
  3 /**
  4  * @class bbq.gui.updateables.UpdateableDropDown
  5  * @extends bbq.gui.updateables.UpdateableField  
  6  */
  7 bbq.gui.updateables.UpdateableDropDown = new Class.create(bbq.gui.updateables.UpdateableField, {
  8 	dropDown: null,
  9 	fields: null,
 10 	
 11 	/**
 12 	 * 
 13 	 * Supports the folloing options:
 14 	 * 
 15 	 * options: {
 16 	 * 		keys: Array
 17 	 * }
 18 	 * 
 19 	 * Supports the following callbacks
 20 	 * 
 21 	 * onValueChanged
 22 	 * 
 23 	 * @param {Object} options
 24 	 * 
 25 	 * }
 26 	 */
 27 	initialize: function($super, options) {
 28 		$super(options);	
 29 		
 30 		this.addClass("UpdateableDropDown");
 31 	},
 32 	
 33 	_setInitialValues: function() {
 34 		this.fields = this.options.keys;
 35 		this.options.initialValue = this.options.initialValue ? this.options.initialValue : 0;
 36 		bbq.gui.updateables.UpdateableField.prototype._setInitialValues.call(this);
 37 		this._setDisplayValue(this.fields[this._getCurrentValue()]);
 38 	},
 39 	
 40 	_updateInitialValues: function() {
 41 		bbq.gui.updateables.UpdateableField.prototype._updateInitialValues.call(this);
 42 		this._setDisplayValue(this.fields[this._getCurrentValue()]);
 43 	},
 44 	
 45 	_createViewFieldFromPropertyDisplay: function() {
 46 		return this.options.propertyDisplay.entity.getPropertyDisplay({property: this.options.propertyDisplay.key, formatter: function(key) {
 47 			return this.fields[key]
 48 		}.bind(this)});
 49 	},
 50 	
 51 	/**
 52 	 * Cancels drop down change to the previous state
 53 	 */
 54 	cancelEdit: function() {
 55 		this._setCurrentValue(this._getOriginalValue());
 56 		this._setDisplayValue(this.fields[this._getCurrentValue()]);
 57 	},
 58 	
 59 	/**
 60 	 * Creates drop down fields
 61 	 */
 62 	createEditField: function() {
 63 		this.dropDown = DOMUtil.createElement("select", {
 64 			onfocus: this._clearFocusWatcher.bind(this),
 65 			onchange: this.updateLocalValue.bindAsEventListener(this)
 66 		});
 67 		
 68 		if(this.fields instanceof Array) {
 69 			for(var i = 0, iCount=this.fields.length; i < iCount; i++) {
 70 				var node = document.createElement("option");
 71 				node.appendChild(document.createTextNode(this.fields[i]));
 72 				node.value = i;
 73 				
 74 				if(i == this._getCurrentValue()) {
 75 					node.selected = "selected";
 76 				}
 77 				
 78 				this.dropDown.appendChild(node);
 79 			}
 80 		} else {
 81 			for(var key in this.fields) {
 82 				var node = document.createElement("option");
 83 				node.appendChild(document.createTextNode(this.fields[key]));
 84 				node.value = key;
 85 				
 86 				if(key == this._getCurrentValue()) {
 87 					node.selected = "selected";
 88 				}
 89 				
 90 				this.dropDown.appendChild(node);
 91 			}
 92 		}
 93 		
 94 		this.dropDown.disabled = this.disabled;
 95 		return this.dropDown;
 96 	},
 97 	
 98 	/**
 99 	 * Saves drop down change
100 	 */
101 	saveEdit: function() {
102 		this._setOriginalValue(this._getCurrentValue());
103 		this._setDisplayValue(this.fields[this._getCurrentValue()]);
104 	},
105 	
106 	/**
107 	 * Updaates local value and triggers Event "onchange"
108 	 * @return {void}
109 	 */
110 	updateLocalValue: function() {
111 		if(this.dropDown && this.dropDown.selectedIndex != null) {
112 			this._setCurrentValue(this.dropDown.selectedIndex);
113 			this._setDisplayValue(this.fields[this._getCurrentValue()]);
114 		}
115 		
116 		this.notifyListeners("onValueChanged");
117 	},
118 	
119 	getValue: function(supressErrorWarning) {
120 		if(!this.getEditMode() && this.fields[this._getCurrentValue()] && this.fields[this._getCurrentValue()].strip() == "") {
121 			// if the visual representation of this field is an empty string, return null so we will be skipped in tables with hideEmptyFields set to true
122 			return null;
123 		}
124 		
125 		if(this.options.values && this.options.values[this._getCurrentValue()]) {
126 			return this.options.values[this._getCurrentValue()];
127 		}
128 	
129 		return this._getCurrentValue();
130 	}
131 });
