  5  * @class bbq.gui.FloatingWindow
  6  * @extends bbq.gui.GUIWidget
  7  */
  8 bbq.gui.FloatingWindow = new Class.create(bbq.gui.GUIWidget, {
  9 	
 10 	_contentDiv: null,
 11 	_content: null,
 12 	_lastMouseX: null,
 13 	_lastMouseY: null,
 14 	_escapeKeyObserver: null,

 16 	/**
 17 	 * @param {mixed} options
 18 	 * @example	
 19 	 * Supports the following options:
 20 	 * 
 21 	 * options: {
 22 	 * 		x: int						// initial x position, optional
 23 	 * 		y: int						// initial y position, optional
 24 	 * 		width: int				// inital width
 25 	 * 		height: int				// initial height
 26 	 * 		owner: Object		// should implement a method called windowClosed
 27 	 * 		title: String				// the window title
 28 	 * 		modal: boolean		// whether the window should be modal
 29 	 * 		nearPointer: boolean	// whether the window should appear near to where the user clicked
 30 	 * 		pointerEvent: Event		// the mouse click
 31 	 * 		shouldCloseWindow: Function // a function that can return false to prevent the window from disappearing
 32 	 * }
 33 	 * 
 34 	 * Supports the following events:
 35 	 * 
 36 	 * onAppear
 37 	 * onDisappear
 38 	 * 
 39 	 */
 40 	initialize: function($super, options) {
 41 		$super(options);
 42 		
 43 		this.setRootNode("div");
 44 		this.addClass("FloatingWindow");
 45 		this.setStyle("position", "absolute");
 46 		this.setStyle("width", this.options.width + "px");
 47 		this.setStyle("height", this.options.height + "px");
 48 		
 49 		this.setStyle("display", "none");
 50 		
 51 		if(!this.options.hideChrome) {
 52 			this._windowTitle = DOMUtil.createTextElement("a", this.options.title ? this.options.title : ".", {
 53 				title: Language.get("bbq.gui.FloatingWindow.clickandholdtodrag"), 
 54 				href: ".", 
 55 				onclick: this.blurIt.bindAsEventListener(this), 
 56 				onmousedown: this.startDrag.bindAsEventListener(this),
 57 				className: "FloatingWindowTitle"
 58 			});
 59 			
 60 			if(Browser.InternetExplorer) {
 61 				this._windowTitle.ondrag = function() {return false};
 62 				this._windowTitle.onselectstart = function() {return false};
 63 			}
 64 			
 65 			var row = DOMUtil.createTableRow(2);
 66 			row.childNodes[0].appendChild(this._windowTitle);
 67 			DOMUtil.addClass(row.childNodes[0], "dragLinkCell");
 68 			
 69 			row.childNodes[1].appendChild(DOMUtil.createTextElement("a", Language.get("bbq.gui.FloatingWindow.close"), {href: ".", onclick: this.closeWindow.bindAsEventListener(this)}));
 70 			DOMUtil.addClass(row.childNodes[1], "closeLinkCell");
 71 			
 72 			this.appendChild(DOMUtil.createTextElement("div", DOMUtil.createTextElement("table", DOMUtil.createTextElement("tbody", row), {className: "chrome"}), {className: "FloatingWindowHeader"}));
 73 			
 74 			this._escapeKeyObserver = function(event) {
 75 				// if escape key was pressed and we are the front-most window
 76 				if(event.keyCode == Event.KEY_ESC && !DOMUtil.hasClass(this.getRootNode(), "Backgrounded")) {
 77 					this.closeWindow();
 78 				}
 79 			}.bind(this);
 80 			
 81 			Event.observe(document, "keypress", this._escapeKeyObserver);
 82 		}
 83 		
 84 		this._contentDiv = DOMUtil.createElement("div", {className: "FloatingWindowContent"}); 
 85 		
 86 		this.appendChild(this._contentDiv);
 87 		
 88 		this._resizeContent();
 89 		
 90 		Event.observe(window, "resize", this._windowResized.bind(this));
 91 		
 92 		if(this.options.modal) {
 93 			currentPage.addModalLayer(this);
 94 		}
 95 		
 96 		// register ourselves with the window list
 97 		bbq.gui.FloatingWindow.windows.push(this);
 98 		
 99 		// watch the content window for clicks to bring the window to front
100 		var bringToFront = this.bringToFront.bind(this);
101 		Event.observe(this._contentDiv, "click", bringToFront);
102 		
103 		this.registerListener("onWindowClosed", function() {
104 			// stop observing content window to prevent memory leak
105 			Event.stopObserving(this._contentDiv, "click", bringToFront);
106 			
107 			// remove ourselves from the window list
108 			for(var i = 0; i < bbq.gui.FloatingWindow.windows.length; i++) {
109 				if(bbq.gui.FloatingWindow.windows[i] == this) {
110 					bbq.gui.FloatingWindow.windows.splice(i, 1);
111 					i--;
112 				}
113 			}
114 		}.bind(this));
115 	},
116 	
117 	isFrontMostWindow: function() {
118 		return this.getStyle("zIndex") == bbq.gui.FloatingWindow.topZIndex;
119 	},
120 	
121 	/**
122 	 * Brings this window in front of all other instances of FloatingWindow
123 	 */
124 	bringToFront: function() {
125 		bbq.gui.FloatingWindow.topZIndex++;
126 		this.setStyle("zIndex", bbq.gui.FloatingWindow.topZIndex);
127 		this.removeClass("Backgrounded");
128 		
129 		for(var i = 0; i < bbq.gui.FloatingWindow.windows.length; i++) {
130 			if(bbq.gui.FloatingWindow.windows[i] != this) {
131 				bbq.gui.FloatingWindow.windows[i].addClass("Backgrounded");
132 			}
133 		}
134 	},
135 	
136 	/**
137 	 * Shows the passed content in the window
138 	 */
139 	setContent: function(content) {
140 		if(content) {
141 			this._content = content;
142 			DOMUtil.emptyNode(this._contentDiv);
143 			
144 			if(this._content.appendTo) {
145 				this._content.setID(null);
146 				//this._content.setStyle("height", DOMUtil.getStyle(this._contentDiv, "height"));
147 				this._content.appendTo(this._contentDiv);
148 			} else {
149 				this._content.id = null;
150 				//DOMUtil.setStyle(this._content, "height", DOMUtil.getStyle(this._contentDiv, "height"));
151 				this._contentDiv.appendChild(this._content);
152 			}
153 			
154 			this._resizeContent();
155 		}
156 	},
157 	
158 	/**
159 	 * @private
160 	 */
161 	_minusStyle: function(node, style) {
162 		var dim = parseInt(node.getStyle(style))
163 		
164 		if(!isNaN(dim)) {
165 			return dim;
166 		}
167 		
168 		return 0;
169 	},
170 	
171 	/**
172 	 * @private
173 	 */
174 	_resizeContent: function() {
175 		var dims = Element.getDimensions(this.getRootNode());
176 		
177 		dims.width -= this._minusStyle(this._contentDiv, "paddingLeft");
178 		dims.width -= this._minusStyle(this._contentDiv, "paddingRight");
179 		dims.height -= this._minusStyle(this._contentDiv, "paddingTop");
180 		dims.height -= this._minusStyle(this._contentDiv, "paddingBottom");
181 		
182 		if(!this.options.hideChrome) {
183 			dims.height -= 30;
184 			dims.width -= 4;
185 		}
186 		
187 		DOMUtil.setStyle(this._contentDiv, "width", Math.abs(dims.width) + "px");
188 		DOMUtil.setStyle(this._contentDiv, "height", Math.abs(dims.height) + "px");
189 		
190 		if(this._content) {
191 			DOMUtil.setStyle(this._content, "height", Math.abs(dims.height - 2) + "px");
192 			//make sure that hight of content and contentDiv are different to please IE
193 			DOMUtil.setStyle(this._contentDiv, "height", Math.abs(dims.height) + "px");
194 			
195 			if(this._content.resize) {
196 				this._content.resize();
197 			}
198 		}
199 	},

201 	/**
202 	 * Sets the window title to the passed string
203 	 * 
204 	 * @param String title The new title
205 	 */
206 	setWindowTitle: function(title) {
207 		if(!this._windowTitle) {
208 			return;
209 		}
210 		
211 		DOMUtil.emptyNode(this._windowTitle);
212 		
213 		if(Object.isString(title)) {
214 			this._windowTitle.appendChild(document.createTextNode(title ? title : " "));
215 		} else {
216 			this._windowTitle.appendChild(title);
217 		}
218 	},

220 	/**
221 	 * @override
222 	 */
223 	render: function() {
224 		if(this._content && this._content.render) {
225 			this._content.render();
226 		}
227 	},

229 	/**
230 	 * Makes the window visible
231 	 */
232 	appear: function() {
233 		this.appendTo(document.body);
234 		
235 		this._resizeContent();
236 		
237 		Effect.Appear(this.getRootNode(), {
238 			duration: 0.1,
239 			afterFinish: function() {
240 				this._resizeContent();
241 				this.notifyListeners("onAppear");
242 				this.bringToFront();
243 			}.bind(this)
244 		});
245 	},
246 	
247 	/**
248 	 * Makes the window invisible
249 	 */
250 	disappear: function() {
251 		Effect.Fade(this.getRootNode(), { duration: 0.25 });
252 		setTimeout(function() {
253 			if(!this.options.modal) {
254 				var parentNode = this.getRootNode().parentNode;
255 		
256 				if(parentNode) {
257 					parentNode.removeChild(this.getRootNode());
258 				}
259 			}
260 			
261 			this.notifyListeners("onDisppear");
262 		}.bind(this), 260);
263 		
264 		Event.stopObserving(document, "keypress", this._escapeKeyObserver);
265 	},
266 	
267 	/**
268 	 * @override
269 	 */
270 	appendTo: function(node) {
271 		bbq.gui.GUIWidget.prototype.appendTo.apply(this, arguments);
272 		
273 		Effect.Appear(this.getRootNode(), { duration: 0.1 });
274 		
275 		this._positionWindow();
276 	},

278 	/**
279 	 * @private
280 	 */
281 	_positionWindow: function() {
282 		var dims = Element.getDimensions(this.getRootNode());
283 		var windowDims = BBQUtil.getWindowSize();
284 		
285 		if(this.options.x && this.options.y) {
286 			this.setStyle("left", this.options.x + "px");
287 			this.setStyle("top", this.options.y + "px");
288 		} else if(this.options.pointerEvent) {
289 			var mouseDims = {
290 				x: Event.pointerX(this.options.pointerEvent),
291 				y: Event.pointerY(this.options.pointerEvent)
292 			}
293 			
294 			var windowPosition = {x: 0, y: 0};
295 			
296 			if(mouseDims.x < (windowDims.width/2)) { //left hand side of screen
297 				if(mouseDims.y < (windowDims.height/2)) { // top half of screen
298 					windowPosition.x = mouseDims.x + 10;
299 					windowPosition.y = mouseDims.y - 10;
300 				} else { // bottom half of screen
301 					windowPosition.x = mouseDims.x + 10;
302 					windowPosition.y = (mouseDims.y - dims.height) + 10;
303 				}
304 			} else { // right hand side of screen
305 				if(mouseDims.y < (windowDims.height/2)) { // top half of screen
306 					windowPosition.x = (mouseDims.x - dims.width) - 10;
307 					windowPosition.y = mouseDims.y - 10;
308 				} else { // bottom half of screen
309 					windowPosition.x = (mouseDims.x - dims.width) - 10;
310 					windowPosition.y = (mouseDims.y - dims.height) + 10;
311 				}
312 			}
313 					
314 			this.setStyle("left", windowPosition.x + "px");
315 			this.setStyle("top", windowPosition.y + "px");
316 		} else {
317 			this.setStyle("left", Math.abs((windowDims.width - dims.width)/2) + "px");
318 			this.setStyle("top", Math.abs((windowDims.height - dims.height)/2) + "px");
319 		}
320 	},

322 	/**
323 	 * Makes the window disappear unless options.shouldCloseWindow is set and returns false.
324 	 */
325 	closeWindow: function(event) {
326 		BBQUtil.clearFocus(event);
327 		
328 		if(this.options.shouldCloseWindow instanceof Function && !this.options.shouldCloseWindow()) {
329 			return false;
330 		}
331 		
332 		this.notifyListeners("onWindowClosed");
333 		this.disappear();
334 		
335 		return false;
336 	},
337 	
338 	/**
339 	 * @private
340 	 */
341 	blurIt: function(event) {
342 		BBQUtil.clearFocus(event);
343 		return false;
344 	},
345 	
346 	/**
347 	 * @private
348 	 */
349 	startDrag: function(event) {
350 		BBQUtil.clearFocus(event);
351 		Event.stop(event);
352 		
353 		this.bringToFront();
354 		this._lastMouseX = Event.pointerX(event);
355 		this._lastMouseY = Event.pointerY(event);
356 		
357 		document.onmousemove = this._doDrag.bindAsEventListener(this);
358 		document.onmouseup = this._stopDrag.bindAsEventListener(this);
359 	},
360 	
361 	/**
362 	 * @private
363 	 */
364 	_stopDrag: function() {
365 		document.onmousemove = null;
366 		document.onmouseup = null;
367 	},
368 	
369 	/**
370 	 * @private
371 	 */
372 	_doDrag: function(event) {
373 		var x = parseInt(this.getStyle("left").replace("px", "")) - (this._lastMouseX - Event.pointerX(event));
374 		var y = parseInt(this.getStyle("top").replace("px", "")) - (this._lastMouseY - Event.pointerY(event))
375 		var windowDims = BBQUtil.getViewportSize();
376 		
377 		if(x >= 0 && x <= (windowDims.width - this.options.width - 10)) {
378 			this.setStyle("left", x + "px");
379 			this._lastMouseX = Event.pointerX(event);
380 		}
381 		
382 		if(y >= 0  && y <= (windowDims.height - this.options.height - 10)) {
383 			this.setStyle("top", y + "px");
384 			this._lastMouseY = Event.pointerY(event);
385 		}
386 	},

388 	/**
389 	 * Animates a change in the window size.
390 	 */
391 	setWindowSize: function(width, height, duration) {
392 		this.options.width = width;
393 		this.options.height = height;
394 		
395 		if(Object.isUndefined(duration)) {
396 			duration = 0.5;
397 			
398 			new Effect.Morph(this.getRootNode(), { 
399 				style: {
400 					width: this.options.width + "px",
401 					height: this.options.height + "px"
402 				},
403 				afterUpdate: function() {
404 					this._resizeContent();
405 					
406 					this._windowResized(true);
407 				}.bind(this),
408 				afterFinish: function() {
409 					this.notifyListeners("onAfterWindowResize");
410 				}.bind(this),
411 				duration: duration
412 			});
413 		} else {
414 			if(this._contentDiv) {
415 				this._contentDiv.style.display = "none";
416 			}
417 			
418 			new Effect.Morph(this.getRootNode(), {
419 				style: {
420 					width: this.options.width + "px",
421 					height: this.options.height + "px"
422 				},
423 				afterUpdate: function() {
424 					this._resizeContent();
425 					
426 					this._windowResized(true);
427 				}.bind(this),
428 				afterFinish: function() {
429 					if(this._contentDiv) {
430 						Effect.Appear(this._contentDiv, {duration: duration/2});
431 					}
432 					
433 					this.notifyListeners("onAfterWindowResize");
434 				}.bind(this),
435 				duration: duration
436 			});
437 		}
438 	},
439 	
440 	/**
441 	 * @private
442 	 */
443 	_windowResized: function(animate) {
444 		this._resizeContent();
445 		
446 		if(this.options.centerWindow) {
447 			// move window into center of screen
448 			this._positionWindow();
449 		} else {
450 			// make sure window is still entirely on the screen
451 			var windowDims = BBQUtil.getViewportSize();
452 			var x = parseInt(this.getStyle("left"));
453 			var y = parseInt(this.getStyle("top"));
454 			var onScreenX = (windowDims.width - this.options.width - 10);
455 			var onScreenY = (windowDims.height - this.options.height - 10);
456 			
457 			if(onScreenX < 0) {
458 				onScreenX = 0;
459 			}
460 			
461 			if(x > onScreenX) {
462 				if(animate) {
463 					new Effect.Morph(this.getRootNode(), {
464 						style: {
465 							left: onScreenX + "px"
466 						}, 
467 						duration: 0.15
468 					});
469 				} else {
470 					this.setStyle("left", onScreenX + "px");
471 				}
472 			}
473 			
474 			if(onScreenY < 0) {
475 				onScreenY = 0;
476 			}
477 			
478 			if(y > onScreenY) {
479 				if(animate) {
480 					new Effect.Morph(this.getRootNode(), {
481 						style: {
482 							top: onScreenY + "px"
483 						}, 
484 						duration: 0.15
485 					});
486 				} else {
487 					this.setStyle("top", onScreenY + "px");
488 				}
489 			}
490 		}
491 	}
