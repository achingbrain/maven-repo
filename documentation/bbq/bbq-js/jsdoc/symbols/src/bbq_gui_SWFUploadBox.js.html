1 include(bbq.gui.GUIWidget);
  2 include(bbq.gui.updateables.UpdateableField);
  3 include(bbq.gui.ProgressBar);
  4 include(bbq.web.FileUploader);
  5 include(bbq.util.BBQUtil);
  6 
  7 /**
  8  * @class bbq.gui.SWFUploadBox
  9  * @extends bbq.gui.GUIWidget
 10  */
 11 bbq.gui.SWFUploadBox = new Class.create(bbq.gui.GUIWidget, {
 12 	
 13 	_fileUploader: null,
 14 	_progressBar: null,
 15 	_percentage: null,
 16 	
 17 	_uploadButton: null,
 18 	_deleteButton: null,
 19 	
 20 	_uploadData: null,
 21 	_fileName: null,
 22 	_fileSize: null,
 23 	
 24 	/**
 25 	 * Supoprts the following options:
 26 	 * 
 27 	 * options: {
 28 	 * 		extensions: Array		// Optional - an array of allowed file extensions: ["csv", "tsv", "txt"] 
 29 	 * 		description: String		// Optional - file extensions description: "Contacts, *.csv, *.tsv, *.txt"
 30 	 * 		uploadButton: Boolean		// Optional - restricts creation of upload button, 
 31 	 *  	uploadTo: String		// Required - Upload path		
 32 	 * }
 33 	 * 
 34 	 * Supports the following callbacks:
 35 	 * 
 36 	 * onFileSelected
 37 	 * onUploadCancel
 38 	 * onUploadProgress
 39 	 * onUploadComplete
 40 	 * onUploadError
 41 	 * onInvalidFileExtension
 42 	 */
 43 	initialize: function($super, options) {
 44 		$super(options);
 45 		
 46 		this.setRootNode("ul");
 47 		this.addClass("SWFUploadBox");
 48 		
 49 		this._fileName = null;
 50 		this._fileSize = null;
 51 		
 52 		this._fileUploader = new bbq.web.FileUploader({transparent: true, description: this.options.description, extensions: this.options.extensions});	
 53 		this._progressBar = new bbq.gui.ProgressBar({barText: Language.get("bbq.gui.SWFUploadBox.choosefile")});
 54 		this._percentage = new bbq.gui.updateables.UpdateableField({initialValue: "0%"});
 55 		
 56 		if(Object.isUndefined(this.options.uploadButton)) {
 57 			this._uploadButton = false;
 58 		} else {
 59 			this._uploadButton = this.options.uploadButton;
 60 		}
 61 		
 62 		this.appendChild(DOMUtil.createTextElement("li", this._progressBar));
 63 		this.appendChild(DOMUtil.createTextElement("li", this._percentage, {className:"Percentage"}));
 64 		this.appendChild(DOMUtil.createTextElement("li", this._fileUploader));
 65 		
 66 		this._fileUploader.registerListener("onLoad", function() {
 67 			this._fileUploader.setButtonText(Language.get("bbq.gui.SWFUploadBox.browse"));
 68 		}.bind(this));
 69 		
 70 		this._fileUploader.registerListener("onFileSelected", function() {
 71 			this.notifyListeners("onFileChosen");
 72 			
 73 			this._fileName = this._fileUploader.getFileName();
 74 			this._fileSize = this._fileUploader.getBytesTotal();
 75 			
 76 			this._progressBar.setBarTextAndSize(this._getFileName(this._fileName) +" "+ BBQUtil.formatFileSize(this._fileSize), 0);
 77 			
 78 			if(this._uploadButton) {
 79 				this._uploadButton.setDisabled(false);
 80 			}else{
 81 				this.beginUpload();
 82 			}
 83 		}.bind(this));
 84 		
 85 		this._fileUploader.registerListener("onUploadCancel", function() {
 86 			Log.info("onUploadCancel");
 87 			this.notifyListeners("onUploadCancel");
 88 			
 89 			if(this._uploadButton) {
 90 				this._uploadButton.setDisabled(true);
 91 			}
 92 			
 93 			this._fileName = Language.get("bbq.gui.SWFUploadBox.choosefile");
 94 			
 95 			this._percentage.setValue("0%");
 96 			this._progressBar.setBarTextAndSize(this._getFileName(this._fileName), 0);
 97 			
 98 			this._fileUploader.setButtonText(Language.get("bbq.gui.SWFUploadBox.browse"));
 99 			this._fileUploader.setBrowse(true);
100 		}.bind(this));
101 		
102 		this._fileUploader.registerListener("onUploadComplete", function() {
103 			this._percentage.setValue("100%");
104 			this._progressBar.setBarSize(200);
105 			this._progressBar.render();
106 			this._getUploadData();
107 			
108 			this.notifyListeners("onUploadComplete");
109 			if(this._uploadButton) {
110 				this._uploadButton.setDisabled(true);
111 			}
112 		}.bind(this));
113 		
114 		this._fileUploader.registerListener("onUploadError", function() {
115 			this.notifyListeners("onUploadError");
116 			if(this._uploadButton) {
117 				this._uploadButton.setDisabled(true);
118 			}
119 		}.bind(this));
120 		
121 		this._fileUploader.registerListener("onUploadProgress", function() {
122 			this.notifyListeners("onUploadProgress");
123 			
124 			if(this._uploadButton) {
125 				this._uploadButton.setDisabled(true);
126 			}
127 			
128 			var completed = Math.round((this._fileUploader.getBytesSent()/this._fileSize)*100);
129 			this._percentage.setValue(completed+"%");
130 			this._progressBar.setBarSize(Math.round(completed*2));
131 			this._progressBar.render();
132 		}.bind(this));
133 	},
134 	
135 	appendBefore: function(parentNode, beforeNode) {
136 		parentNode.insertBefore(this.getRootNode(), beforeNode);
137 	},
138 	
139 	beginUpload: function() {
140 		if(this._fileName) {
141 			var extension = this._fileName.split(".").pop().toLowerCase();
142 			if(this.options.extensions instanceof Array && this.options.extensions.indexOf(extension) == -1) {
143 				this.notifyListeners("onInvalidFileExtension");
144 				Log.info("Invalid file extensions");
145 				return;
146 			}
147 			
148 			this._fileUploader.setBrowse(false);
149 			this._fileUploader.setButtonText(Language.get("bbq.gui.SWFUploadBox.cancel"));
150 			this._fileUploader.startUpload();
151 		}
152 	},
153 	
154 	cancelUpload: function() {
155 		this._fileUploader.cancelUpload();
156 	},
157 	
158 	_getUploadData: function() {		
159 		this._uploadData = this._fileUploader.getData();
160 	},
161 	
162 	getUploadData: function() {
163 		return this._uploadData;
164 	},
165 	
166 	getUploadError: function() {
167 		return this._uploadData;
168 	},
169 	
170 	/**
171 	 * @private
172 	 */
173 	_getFileName: function(fileName) {
174 		return (fileName.length > 19 ? fileName.substr(0, 8) + "..." + fileName.substr(fileName.length - 8, 8) : fileName);
175 	},
176 	
177 	getFileName: function() {
178 		return this._fileName;
179 	},
180 	
181 	getFileSize: function() {
182 		return this._fileSize;
183 	},
184 	
185 	getFileTransferID: function() {
186 		var data = this.getUploadData();
187 		Log.dir(data);
188 		return data["filetransferid"];
189 	}	
190 });
