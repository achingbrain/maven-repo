 16 	/**
 17 	 * @constructs
 18 	 * @extends bbq.gui.GUIWidget
 19 	 * @param {mixed} options
 20 	 * @example
 21 	 * The URL passed as part of the options object should return in the following format:
 22 	 * 
 23 	 *  { numEntities: int, entities: {"en7": data, "en8": data, "en9": data,...}}     // <-  enN is zero indexed!
 24 	 * 
 25 	 * Supports the following options:
 26 	 * 
 27 	 * options: {
 28 	 * 		rowHeight: int								// roughly the expected height of one row.
 29 	 * 		createEntity: function(data)		// should return one new entity
 30 	 * 		url: String										// where to load from
 31 	 * 		args: Object									// arguments to pass
 32 	 * 		numToLoad: int							// how many entities to load initially and on every column sort
 33 	 * 		loadingText: String						// shown while doin the inital load
 34 	 * 		noEntitiesText:	String					// shown when there is nothing to show
 35 	 * 		entityKey: String							// the key field on each entity.  Pass this OR getEntityKey
 36 	 * 		getEntityKey: Function					// should take an entity and return it's id Pass this OR entityKey
 37 	 * }
 38 	 * 
 39 	 * Supports observers for the following events
 40 	 * 
 41 	 * onEntitiesLoad
 42 	 * onEmptyList
 43 	 * onEntitiesLoaded
 44 	 * onSelectedEntityChange
 45 	 */
 46 	initialize: function($super, options) {
 47 		$super(options);
 48 		
 49 		this.setRootNode("div");
 50 		this.addClass("ScrollLoadGUIWidget");
 51 		this._lastRange = {start: 0, end: 0};
 52 		
 53 		if(typeof(this.options.args) == "undefined") {
 54 			this.options.args = {};
 55 		}
 56 		
 57 		if(typeof(this.options.startSorted) != "undefined") {
 58 			this.options.args.sortBy = parseInt(this.options.startSorted.column);
 59 			this.options.args.sortDirection = this.options.startSorted.direction == bbq.constants.PAGINATOROPTIONS_SORTDESC ? bbq.constants.PAGINATOROPTIONS_SORTDESC : bbq.constants.PAGINATOROPTIONS_SORTASC;
 60 			this.options.args.includeArchived = false;
 61 		} else {
 62 			this.options.args.sortBy = 0;
 63 			this.options.args.sortDirection = bbq.constants.PAGINATOROPTIONS_SORTDESC;
 64 			this.options.args.includeArchived = false;
 65 		}
 66 		
 67 		this._entities = new bbq.lang.DataHolder({entityKey: this.options.entityKey});
 68 		this._loading = false;
 69 		this._initialLoad = true;
 70 		
 71 		this._callbacks = {};
 72 	},

 74 	_getEntityKey: function(entity) {
 75 		if(this.options.getEntityKey) {
 76 			return this.options.getEntityKey(entity);
 77 		}
 78 		
 79 		return BBQUtil.getKey(entity, this.options.entityKey);
 80 	},
 81 	
 82 	_storeCallback: function(entity, action, callback) {
 83 		if(!(callback instanceof Function)) {
 84 			return;
 85 		}
 86 		
 87 		var key = this._getEntityKey(entity);
 88 		
 89 		if(typeof(this._callbacks[key]) == "undefined") {
 90 			this._callbacks[key] = {};
 91 		}
 92 		
 93 		this._callbacks[key][action] = callback;
 94 	},
 95 	
 96 	_processEvent: function(event, entity) {
 97 		BBQUtil.clearFocus(event);
 98 		this._invokeCallback(entity, event.type);
 99 		
100 		if(FocusWatcher) {
101 			FocusWatcher.setKeypressCallbackObject(this);
102 		}
103 		
104 		return false;
105 	},
106 	
107 	_invokeCallback: function(entity, action) {
108 		try {
109 			var key = this._getEntityKey(entity);
110 			
111 			if(this._callbacks[key] && this._callbacks[key][action] instanceof Function) {
112 				// only set the item down if the callback does not return false
113 				var result = this._callbacks[key][action](entity);
114 				
115 				if(result !== false) {
116 					this.setDown(entity);
117 				}
118 				
119 				return result;
120 			}
121 		} catch(e) {
122 			Log.error("Error invoking callback " + action, e);
123 		}
124 	},
125 	
126 	setDown: function(entity) {
127 		this.notifyListeners("onSelectedEntityChange");
128 	},

141 		}.bind(this));

151 	/**
152 	 * @protected
153 	 * 
154 	 * Callback for when the containing scrollable is scrolled.
155 	 */
156 	_scrollableScrolled: function() {
157 		if(this._scrollingTimeout) {
158 			clearTimeout(this._scrollingTimeout);
159 		}
160 		
161 		this._scrollingTimeout = setTimeout(this._workOutMovement.bind(this), 500);
162 	},

164 	/**
165 	 * @protected
166 	 * 
167 	 * Override this method in your subclass.  It should work out which elements (loaded or not) 
168 	 * are currently visible and call:
169 	 * 
170 	 * this._loadDetails(startElement, endElement);
171 	 * 
172 	 * Where the arguments to _loadDetails are their indexes in the _entities array
173 	 */
174 	_workOutMovement: function() {	
175 		Log.error("Non overriden call to ScrollLoadGUIWidget._workOutMovement");
176 	},

178 	_loadDetails: function(start, end) {
179 		try {
180 			if(this._scrollingTimeout) {
181 				clearTimeout(this._scrollingTimeout);
182 			}
183 			
184 			//Log.info("passed " + start + " to " + end);
185 			if(typeof(start) == "undefined" || start  < 0) {
186 				start = 0;
187 			}
188 			
189 			if(start > end) {
190 				start = end - 1;
191 			}
192 			
193 			var needLoad = false;
194 			
195 			if(this._initialLoad) {
196 				needLoad = true;
197 			} else {
198 				//Log.info("checking " + start + " to " + end + " of " + this._entities.length);
199 				
200 				// see which entities we need to load data for
201 				for(var i = start; i <= end; i++) {
202 					if(!this._entities.getElement(i)) {
203 						//Log.info("item " + i + " triggered load");
204 						needLoad = true;
205 						break;
206 					}
207 				}
208 			}
209 			
210 			if(!this._loading && needLoad) {
211 				this.options.args["entityOffset"] = start;
212 				this.options.args["entityLength"] = end - start;
213 				
214 				new bbq.ajax.JSONRequest({
215 					url: this.options.url, 
216 					args: this.options.args,
217 					onSuccess: this._loadedDetails.bind(this), 
218 					method: "post"
219 				});
220 				
221 				this._loading = true;
222 				
223 				this.notifyListeners("onEntitiesLoad");
224 			} else {
225 				this.notifyListeners("onAllEntitiesLoadedAlready");
226 			}
227 		} catch(e) {
228 			Log.error("Error loading details", e);
229 		}
230 	},
231 	
232 	dataLoaded: function() {
233 		if(this._initialLoad) {
234 			return false;
235 		}
236 		
237 		return !this._loading;
238 	},

240 	_loadedDetails: function(serverResponse, json) {
241 		try {
242 			this._loading = false;
243 			this._initialLoad = false;
244 			var numEntities = json.numEntities;
245 			if(!(json.entities instanceof Array)) {
246 				if(this._entities.getLength() != numEntities) {
247 					if(numEntities > this._entities.getLength()) {
248 						// pad the array
249 						for(var i = numEntities-1; i >= 0; i--) {
250 							if(typeof(this._entities.getElement(i)) == "undefined") {
251 								this._entities.setElement(i, false);
252 							}
253 						}
254 					} else {
255 						// truncate list
256 						this._entities.setLength(numEntities);
257 					}
258 				}
259 				
260 				this._processNewEntities(json);
261 			}
262 			
263 			this.render();
264 			
265 			this.notifyListeners("onEntitiesLoaded");
266 			
267 			if(this._entities.getLength() == 0) {
268 				this.notifyListeners("onEmptyList");
269 			}
270 		} catch(e) {
271 			Log.error("thrown _loadedDetails", e);
272 		}
273 	},
274 	
275 	_processNewEntities: function(json) {
276 		var entities = json.entities;
277 		for(var key in entities) {
278 			var index = parseInt(key.replace("en", ""));
279 					
280 			if(this.options.createEntity) {
281 				this._entities.setElement(index, this.options.createEntity(entities[key], index));
282 			} else {
283 				this._entities.setElement(index, entities[key]);
284 			}
285 		}
286 	},

290 		clearTimeout(this._scrollingTimeout);
291 		this._scrollingTimeout = setTimeout(this._workOutMovement.bind(this), 1000);
292 	},
