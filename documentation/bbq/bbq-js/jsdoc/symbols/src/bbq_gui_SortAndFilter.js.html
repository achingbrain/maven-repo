 89 		if(items) {

104 		this.sortNames[this.sortNames.length] = text;
105 		this.sortKeys[this.sortKeys.length] = key;
106 		
107 		var optionNode = document.createElement("option");
108 		optionNode.appendChild(document.createTextNode(text));
109

117 	addNewItem: function(item) {
118 		this.originalItems[this.originalItems.length] = item;
119 		this.sortIt();
120 	},
121 	
122 	sortIt: function(event) {
123 		var element = BBQUtil.clearFocus(event);
124 		this.currentSortKey = this.sortKeys[this.sortCriteria.selectedIndex];
125 		
126 		if(this.sortKeys[this.sortCriteria.selectedIndex].search(/(\.)/) > -1) {
127 			var keys = this.sortKeys[this.sortCriteria.selectedIndex].split(".");
128 			this.currentSortKey = keys[keys.length - 1];
129 			
130 			for(var i = 0, iCount=this.items.length; i < iCount; i++) {
131 				if(this.ascRadio.checked) {
132 					this.items[i][this.searchItems].sort(this.ascSortFunction.bind(this));
133 				} else {
134 					this.items[i][this.searchItems].sort(this.descSortFunction.bind(this));
135 				}
136 			}
137 		} else {
138 			this.currentSortKey = this.sortKeys[this.sortCriteria.selectedIndex];
139 			
140 			if(this.ascRadio.checked) {
141 				this.items.sort(this.ascSortFunction.bind(this));
142 			} else {
143 				this.items.sort(this.descSortFunction.bind(this));
144 			}
145 		}
146 		
147 		this.controller.renderTreeList();
148 	},
149 	
150 	preFilter: function(event) {
151 		DOMUtil.addClass(this.filterText, "filtering");
152 		DOMUtil.addClass($("div_dataTree"), "filtering");
153 		
154 		setTimeout(this.filterIt.bind(this), 100);
155 	},
156 	
157 	filterIt: function() {
158 		if(this.filterText.value != "") {
159 			var expression = new RegExp(this.filterText.value, "i");
160 			var n = 0;
161 			
162 			this.items.splice(0, this.items.length);
163 			
164 			if(this.searchItems) {
165 				for(var i = 0, iCount=this.originalItems.length; i < iCount; i++) {
166 					// TODO: rewrite this to store the order of IDs, not clone the objects themselves
167 					//this.items[i] = BBQUtil.cloneObject(this.originalItems[i]);
168 					this.items[i][this.searchItems].splice(0, this.items[i][this.searchItems].length);
169 						
170 					for(var k = 0, kCount=this.originalItems[i][this.searchItems].length; k < kCount; k++) {
171 						if(this.filterItems(this.originalItems[i][this.searchItems][k], this.filterText.value, expression, 5)) {
172 							this.items[n][this.searchItems][k] = this.originalItems[i][this.searchItems][k];	
173 						}
174 					}
175 				}
176 			} else {
177 				for(var i = 0, iCount=this.originalItems.length; i < iCount; i++) {
178 					if(this.filterItems(this.originalItems[i], this.filterText.value, expression, 5)) {
179 						this.items[n] = this.originalItems[i];
180 						n++;
181 					}
182 				}
183 			}
184 			
185 			currentPage.renderTreeList();
186 		} else {
187 			this.clearFilter();
188 		}
189 	},
190 	
191 	filterItems: function(item, text, expression, level) {
192 		if(level == 0 || !item) {
193 			return false;
194 		}
195 		
196 		if(item instanceof Array) {
197 			for(var i = 0, iCount=item.length; i < iCount; i++) {
198 				if(this.filterItems(item[i], text, expression, level - 1)) {
199 					return true;			
200 				}
201 			}
202 		} else if(item instanceof Object) {
203 			for(var key in item) {
204 				if(this.filterItems(item[key], text, expression, level - 1)) {
205 					return true;			
206 				}
207 			}
208 		} else {
209 			return (item == text || (item.search && item.search(expression) > -1));
210 		}
211 		
212 		return false;
213 	},
214 	
215 	clearFilter: function(event) {
216 		var element = BBQUtil.clearFocus(event);
217 		
218 		DOMUtil.removeClass(this.filterText, "filtering");
219 		DOMUtil.removeClass($("div_dataTree"), "filtering");
220 		
221 		this.filterText.value = "";
222 		
223 		for(var i = 0, iCount=this.originalItems.length; i < iCount; i++) {
224 			this.items[i] = this.originalItems[i];
225 		}
226 		
227 		currentPage.renderTreeList();
228 	}
