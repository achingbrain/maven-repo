1 include(bbq.gui.form.FormField);
  2 
  3 /**
  4  * Supports the following options:
  5  *
  6  * options: {
  7  *      options: [
  8  *          key: String
  9  *          value: Object
 10  *      ]
 11  * }
 12  *
 13  * @class bbq.gui.form.DropDown
 14  * @extends bbq.gui.form.FormField
 15  */
 16 bbq.gui.form.DropDown = new Class.create(bbq.gui.form.FormField, {
 17 	initialize: function($super, args) {
 18 		try {
 19 			$super(args);
 20 
 21 			this.setRootNode("select");
 22 			this.addClass("DropDown");
 23 
 24 			if(Object.isArray(this.options.options)) {
 25 				this.setOptions(this.options.options);
 26 			}
 27 		} catch(e) {
 28 			Log.error("Error constructing DropDown", e);
 29 		}
 30 	},
 31 
 32 	_getRawValue: function() {
 33 		var index = this.getRootNode().value;
 34 
 35 		return this.options.options[index].value;
 36 	},
 37 
 38 	_setRawValue: function(value) {
 39 		if(!Object.isArray(this.options.options)) {
 40 			return;
 41 		}
 42 
 43 		var optionElements = $A(this.getRootNode().getElementsByTagName("option"));
 44 
 45 		optionElements.each(function(option) {
 46 			delete option.selected;
 47 		});
 48 
 49 		this.options.options.each(function(option, index) {
 50 			if(option.value == value) {
 51 				if(optionElements[index]) {
 52 					optionElements[index].selected = true;
 53 				}
 54 			}
 55 		}.bind(this));
 56 	},
 57 
 58 	setOptions: function(options) {
 59 		// preserve currently selected index
 60 		var selectedIndex = 0;
 61 
 62 		$A(this.getRootNode().getElementsByTagName("option")).each(function(option, index) {
 63 			if (option.selected) {
 64 				selectedIndex = index;
 65 
 66 				throw $break;
 67 			}
 68 		});
 69 
 70 		this.empty();
 71 
 72 		options.each(function(option, index) {
 73 			var node = DOMUtil.createElement("option", option.key, {value: index});
 74 
 75 			if (index == selectedIndex) {
 76 				node.selected = true;
 77 			}
 78 
 79 			this.appendChild(node);
 80 		}.bind(this));
 81 
 82 		this.options.options = options;
 83 	}
 84 });
